x-4 & x\geq 2 \end{cases}$}{\mbox{}\\[-\baselineskip]\begin{tikzpicture}[alt={ALT-TEXT-TO-BE-DETERMINED},scale=.5]
\begin{axis}[axis y line=middle, axis x line=middle, xmin=-4, xmax=4, ymin=-4, ymax=4,name=myplot]
\addplot[draw={\colorone}, domain=-4:2, thick, smooth]{x*x-3};

\draw[draw={\colorone}] (axis cs:2,-2) circle (2pt);
\addplot[draw={\colorone}, domain=2:4, thick, smooth]{x-4};
\end{axis}

\node [above] at (myplot.above origin) {\scriptsize $y$};
\end{tikzpicture}}
