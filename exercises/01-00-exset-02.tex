\exercise{$f(x)=\begin{cases} x^2-3 & x<2\\
x-4 & x\geq 2 \end{cases}$}{\mbox{}\\[-\baselineskip]\begin{tikzpicture}[alt={A parabola starting near (-3.5,4), decreasing to bottom out at (0,-3) before increasing to finish at a hollow dot at (2,1).  A solid dot at (2,-2) connects to (4,0).},scale=.5]
\begin{axis}[axis y line=middle, axis x line=middle, xmin=-4, xmax=4, ymin=-4, ymax=4,name=myplot]
\addplot[draw={\colorone}, domain=-4:2, thick, smooth]{x*x-3};
\draw[fill={\colorone},draw={\colorone}] (axis cs:2,1) circle (2pt);
\draw[draw={\colorone}] (axis cs:2,-2) circle (2pt);
\addplot[draw={\colorone}, domain=2:4, thick, smooth]{x-4};
\end{axis}
\node [right] at (myplot.right of origin) {\scriptsize $x$};
\node [above] at (myplot.above origin) {\scriptsize $y$};
\end{tikzpicture}}
