\exercisesetinstructions[Use the following limits to evaluate the limits given in Exercises]{, where possible. If it is not possible, state so.
\begin{align*}
& \lim_{x\to1}f(x) = f(1) = 2 && \lim_{x\to10} f(x) = f(10) = 1\\
& \lim_{x\to1}g(x) = g(1) = 0 && \lim_{x\to10} g(x) = g(10) = \pi
\end{align*}}

\exercise{$\ds \lim_{x\to1}f(x)^{g(x)}$}{$1$}

\exercise{$\ds \lim_{x\to10}\cos \big(g(x)\big)$}{$-1$}

\exercise{$\ds \lim_{x\to1}f(x)g(x)$}{$0$}

\exercise{$\ds \lim_{x\to1}g\big(5f(x)\big)$}{$\pi$}
