\exercise{\begin{minipage}[t]{.4\linewidth}
$\ds f(x) = \frac{1}{(x+1)^2}$
\begin{enumext}[start=1]
\item		$\ds \lim_{x\to -1^-} f(x)$
\item		$\ds \lim_{x\to -1^+} f(x)$
\end{enumext}
\end{minipage}%
\begin{minipage}[t]{.6\linewidth}
\mbox{}\\[-\baselineskip]
\begin{tikzpicture}[alt={A curve starting near (-2,0) and (2,0) and getting large near x=-1.}]
\begin{axis}[width=\marginparwidth,tick label style={font=\scriptsize},
axis y line=middle,axis x line=middle,name=myplot,
ymin=-5,ymax=105,xmin=-2.5,xmax=.5]
\addplot [thick,draw={\colorone},domain=-2.5:-1.1,samples=50] {1/(x+1)^2};
\addplot [thick,draw={\colorone},domain=-.9:0.5,samples=50] {1/(x+1)^2};
\draw [draw={\colorone},dashed] (axis cs:-1,105) -- (axis cs:-1,-5);
\end{axis}
\node [right] at (myplot.right of origin) {\scriptsize $x$};
\node [above] at (myplot.above origin) {\scriptsize $y$};
\end{tikzpicture}
\end{minipage}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
\item	$\infty$
\item	$\infty$
\end{enumext}}}
