\exercisesetinstructions{, numerically approximate the following limits:
	\begin{enumerate}[label=(\alph*)]
	\item $\ds \lim_{x\to 3^-} f(x)$
	\item $\ds \lim_{x\to 3^+} f(x)$
	\item $\ds \lim_{x\to 3} f(x)$
	\end{enumerate}}

\exercise{$\ds f(x)=\frac{x^2-1}{x^2-x-6}$
}{Tables will vary.
\begin{enumerate}
\item	
\begin{tabular}{cc}
$x$ & $f(x)$ \\ \hline
$2.9$ & $-15.1224$ \\
 $2.99$ & $-159.12$ \\
 $2.999$ & $-1599.12$
 \end{tabular}\\
 It seems $\lim_{x\to3^-}f(x) =-\infty$.
\item
\begin{tabular}{cc}
$x$ & $f(x)$ \\ \hline
$ 3.1$ & $16.8824$ \\
 $3.01$ & $160.88$ \\
 $3.001$ & $1600.88$
 \end{tabular}\\
  It seems $\lim_{x\to3^+}f(x) =\infty$.
  \item		It seems $\lim_{x\to3}f(x)$ does not exist.
\end{enumerate}}
