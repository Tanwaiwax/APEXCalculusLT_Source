\begin{exerciseset}{In Exercises}{, approximate the given limits both numerically and graphically.}

\exercise{$\ds\lim_{x\to 1}\left(x^2+3x-5\right)$}{$-1$}

\exercise{$\ds\lim_{x\to 0}\left(x^3-3x^2+x-5\right)$}{$-5$}

\exercise{$\ds\lim_{x\to 0} \frac{x+1}{x^2+3x}$}{Limit does not exist}

\exercise{$\ds\lim_{x\to 3} \frac{x^2-2x-3}{x^2-4x+3}$}{$2$}

\exercise{$\ds\lim_{x\to -1} \frac{x^2+8x+7}{x^2+6x+5}$}{$1.5$}

\exercise{$\ds\lim_{x\to 2} \frac{x^2+7x+10}{x^2-4x+4}$}{Limit does not exist.}

\exercise{$\ds\lim_{x\to 2} f(x)$, where 
$f(x)=\begin{cases}x+2 & x\leq 2 \\ 3x-5 & x>2 \end{cases}$.}{Limit does not exist.}

\exercise{$\ds\lim_{x\to 3} f(x)$, where 
$f(x)=\begin{cases}x^2-x+1 & x\leq 3 \\ 2x+1 & x>3 \end{cases}$.}{$7$}

\exercise{$\ds\lim_{x\to 0} f(x)$, where 
$f(x)=\begin{cases}\cos x & x\leq 0 \\ x^2+3x+1 & x>0 \end{cases}$.}{$1$}

\exercise{$\ds\lim_{x\to \pi/2} f(x)$, where 
$f(x)=\begin{cases}\sin x & x\leq \pi/2 \\ \cos x & x>\pi/2 \end{cases}$.}{Limit does not exist.}

\end{exerciseset}
