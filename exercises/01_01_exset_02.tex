\exerciseset{In Exercises}{, a function $f$ and a value $a$ are giv\-en. Approximate the limit of the difference quotient, $\displaystyle \lim_{h\to 0}\frac{f(a+h)-f(a)}{h}$, using $h = \pm 0.1, \pm 0.01$.}{

\exercise{$f(x) = -7x+2$,\quad  $a=3$}{\begin{tabular}{cc}
$h$ & $\frac{f(a+h)-f(a)}{h}$\\ \hline 
 $-0.1$ & $-7$ \\
 $-0.01$ & $-7$ \\
 $0.01$ & $-7$ \\
 $0.1$ & $-7$
\end{tabular}
The limit seems to be exactly $-7$.}
