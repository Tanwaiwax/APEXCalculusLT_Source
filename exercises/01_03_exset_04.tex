\exercise{$\ds \lim_{x\to3}x^2-3x+7$}{$7$}

\exercise{$\ds \lim_{x\to\pi}\left(\frac{x-3}{x-5}\right)^7$}{$-0.000000015\approx 0$}

\exercise{$\ds \lim_{x\to\pi/4}\cos x\sin x$}{$1/2$}

\exercise{$\ds \lim_{x\to0}\ln x$}{Limit does not exist}

\exercise{$\ds \lim_{x\to3}4^{x^3-8x}$}{$64$}

\exercise{$\ds \lim_{x\to\pi/6} \csc x$}{$2$}

\exercise{$\ds \lim_{x\to0} \ln(1+x)$}{$0$}

\exercise{$\ds \lim_{x\to\pi}\frac{x^2+3x+5}{5x^2-2x-3}$}{$\frac{\pi^2+3\pi+5}{5\pi^2-2\pi-3} \approx 0.6064$}

\exercise{$\ds \lim_{x\to\pi}\frac{3x+1}{1-x}$}{$\frac{3\pi+1}{1-\pi}$}

\exercise{$\ds \lim_{x\to6} \frac{x^2-4 x-12}{x^2-13 x+42}$}{$-8$}

\exercise{$\ds \lim_{x\to0} \frac{x^2+2 x}{x^2-2 x}$}{$-1$}

\exercise{$\ds \lim_{x\to2} \frac{x^2+6 x-16}{x^2-3 x+2}$}{$10$}

\exercise{$\ds \lim_{x\to2}\frac{x^2-10 x+16}{x^2-x-2}$}{$-2$}

\exercise{$\ds \lim_{x\to-2}\frac{x^2-5 x-14}{x^2+10 x+16}$}{$-3/2$}

\exercise{$\ds \lim_{x\to-1}\frac{x^2+9 x+8}{x^2-6 x-7}$}{$-7/8$}

\exercise{$\ds \lim_{t\to 9} \frac{\sqrt{t}-3}{t-9}$}{$1/6$}

\exercise{$\ds \lim_{x\to 0} \frac{\sqrt{x^2+4}-2}{x^2}$}{$1/4$}

\exercise{$\ds \lim_{t\to 3} \frac{\frac{1}{t}-\frac{1}{3}}{t-3}$}{$-1/9$}

\exercise{$\ds \lim_{x\to 0} \frac{1}{x}-\frac{1}{x^2+x}$}{$1$}

\exercise{$\ds \lim_{t\to 0} \frac{(t-4)^2-16}{t}$}{$-8$}
