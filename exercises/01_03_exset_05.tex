\exerciseset{Exercises}{ challenge your understanding of limits that can be evaluated using the knowledge gained in this section.}{

\exercise{$\ds \lim_{x\to0}\frac{\sin 3x}{x}$}{$3$}

\exercise{$\ds \lim_{x\to0}\frac{\sin 5x}{8x}$}{$5/8$}

\exercise{$\ds \lim_{x\to0}\frac{\ln (1+x)}{x}$}{$1$}

\exercise{$\ds \lim_{x\to0}\frac{\sin x}{x}$, where $x$ is measured in degrees, not radians.}{$\pi/180$}

\exercise{$\ds \lim_{x\to 0} \frac {\tan 4x}{\tan 3x}$}{$4/3$}

\exercise{$\ds \lim_{x\to 0} \frac {\tan 5x}{\sin 7x}$}{$5/7$}

}
