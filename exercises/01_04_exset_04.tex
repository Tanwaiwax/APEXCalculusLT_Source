\exerciseset{In Exercises}{, sketch the graph of a function $f$ that satisfies all of the given conditions.}{

\exercise{$\ds \lim_{x\to 1^-} f(x)=2$, \quad $\ds \lim_{x\to 1^+} f(x)=-3$, \quad $f(1)=0$.}{Answers will vary.}

\exercise{$\ds \lim_{x\to -1} f(x)=3$, \quad $\ds \lim_{x\to 3^-} f(x)=1$, \quad $\ds \lim_{x\to 3^+} f(x)=-2$, \quad $f(-1)=1$, \quad $f(3)=-2$.}{Answers will vary.}

\exercise{$\ds \lim_{x\to -2^-} f(x)=1$, \quad  $\ds \lim_{x\to-2^+} f(x)=0$, \quad $\ds \lim_{x\to 0^-} f(x)=3$, \quad $\ds \lim_{x\to 0^+} f(x)=-1$, \quad $f(-2)=4$, \quad $f(0)=-3$.}{Answers will vary.}

\exercise{$\ds \lim_{x\to 0^-} f(x)=0$, \quad $\ds \lim_{x\to 0^+} f(x)=2$, \quad $\ds \lim_{x\to 4^-} f(x)=-2$, \quad $\ds \lim_{x\to 4^+} f(x)=1$, \quad $f(0)=2$, \quad $f(4)=-2$.}{Answers will vary.}
