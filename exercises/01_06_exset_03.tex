\exerciseset{In Exercises}{, identify the horizontal and vertical asymptotes, if any, of the given function.
}{

\exercise{$\ds f(x)=\frac{2 x^2-2 x-4}{x^2+x-20}$
}{Horizontal asymptote at $y=2$; vertical asymptotes at $x=-5,4$.
}

\exercise{$\ds f(x)=\frac{-3 x^2-9 x-6}{5 x^2-10 x-15}$
}{Horizontal asymptote at $y=-3/5$; vertical asymptote at $x=3$.
}

\exercise{$\ds f(x)=\frac{x^2+x-12}{7 x^3-14 x^2-21 x}$
}{Horizontal asymptote at $y=0$; vertical asymptotes at $x=-1,0$.
}

\exercise{$\ds f(x)=\frac{x^2-9}{9 x-9}$
}{No horizontal asymptote; vertical asymptote at $x=1$.
}

\exercise{$\ds f(x)=\frac{x^2-9}{9 x+27}$
}{No horizontal or vertical asymptotes.
}

\exercise{$\ds f(x) = \frac{x^2-1}{-x^2-1}$
}{Horizontal asymptote at $y=-1$; no vertical asymptotes
}
}
