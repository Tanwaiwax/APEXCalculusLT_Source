\exercise{If $f(x)=x^2+2x$ and $g(x)=x-4$, find
\begin{multicols}{2}
\begin{enumerate}
\item $(f \circ g)(6)$
\item $(g \circ f)(6)$
\item $(f \circ g)(x)$
\item $(g \circ f)(x)$
\end{enumerate}
\end{multicols}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item 8
\item 44
\item $x^2-6x+8$
\item $x^2+2x-4$
\end{enumerate}}}

\item $(g \circ f)(x)$

\end{enumerate}}}

\exercise{If $F(x) = f(g(x))$, identify $f(x)$ and $g(x)$.
\begin{enumerate}
\item $F(x) = \frac5{x+4}$
\item $F(x)=\abs{4-x^2}$
\item $F(x)=\sqrt{(x+2)^2-5}$
\end{enumerate}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item Possible solution: $f(x)=\frac5{x}$ and $g(x)=x+4$
\item Possible solution: $f(x)=\abs x$ and $g(x)=4-x^2$
\item Possible solution: $f(x)=\sqrt{x-5}$ and $g(x)=(x+2)^2$
\end{enumerate}}}
