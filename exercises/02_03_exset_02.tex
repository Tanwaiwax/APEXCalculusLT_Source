\begin{exerciseset}{In Exercises}{, compute the first four derivatives of the given function.}

\exercise{$f(x) = x^6$}{$\fp(x) = 6x^5$ $\fpp(x) = 30x^4$ $\fpp'(x) = 120x^3$ $f^{(4)}(x) = 360x^2$}

\exercise{$g(x) = 2\cos x$}{$g'(x) = -2\sin x$ $g''(x) = -2\cos x$ $g'''(x) = 2\sin x$ $g^{(4)}(x) = 2\cos x$}

\exercise{$h(t) = t^2 - e^t$}{$h'(t) = 2t-e^t$ $h''(t) = 2-e^t$ $h'''(t) = -e^t$ $h^{(4)}(t) = -e^t$}

\exercise{$p(\theta) = \theta^4-\theta^3$}{$p'(\theta) = 4\theta^3-3\theta^2$ $p''(\theta) = 12\theta^2 - 6\theta$ $p'''(\theta) = 24\theta-6$ $p^{(4)}(\theta) = 24$}

\exercise{$f(\theta) = \sin\theta-\cos\theta$}{$\fp(\theta) = \cos\theta+\sin\theta$ $\fp'(\theta) = -\sin\theta+\cos\theta$ $\fp''(\theta) = -\cos\theta-\sin\theta$ $f^{(4)}(\theta) = \sin\theta-\cos\theta$}

\exercise{$f(x)=1{,}100$}{$\fp(x) = \fp'(x) = \fp''(x) = f^{(4)}(x) = 0$}

\end{exerciseset}
