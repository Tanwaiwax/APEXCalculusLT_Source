\exerciseset{In Exercises}{, an implicitly defined function is given. Find $\ds \frac{d^2y}{dx^2}$. Note: these are the same problems used in Exercises \ref{exer:02_06_ex_09} through \ref{exer:02_06_ex_12}.
}{

\exercise{$x^4+y^2+y=7$}{$\frac{d^2y}{dx^2} = \frac{(2y+1)(-12x^2) + 4x^3\left(2\frac{-4x^3}{2y+1}\right)}{(2y+1)^2}$}

\exercise{$x^{2/5}+y^{2/5}=1$}{$\frac{d^2y}{dx^2} = \frac35\frac{y^{3/5}}{x^{8/5}} +\frac35\frac{1}{yx^{6/5}}$}

\exercise{$\cos x+\sin y=1$}{$\frac{d^2y}{dx^2} = \frac{\cos x\cos y+\sin^2x\tan y}{\cos^2y}$}

\exercise{$\ds\frac{x}{y}=10$}{$\frac{d^2y}{dx^2} = 0$}

}
