\exerciseset{In Exercises}{, use logarithmic differentiation to find $\ds \frac{dy}{dx}$, then find the equation of the tangent line at the indicated $x$--value.}{

\exercise{$\ds y=(1+x)^{1/x}$,\quad $x=1$}{$y'= (1+x)^{1/x}\big(\frac{1}{x(x+1)}-\frac{\ln(1+x)}{x^2}\big)$

Tangent line: $y = (1-2\ln 2)(x-1)+ 2$}

\exercise{$\ds y=(2x)^{x^2}$,\quad $x=1$}{$y'= (2x)^{x^2}\big(2x\ln (2x)+x\big)$

Tangent line: $y = (2+4\ln 2)(x-1)+ 2$}

\exercise{$\ds y=\frac{x^{x}}{x+1}$,\quad $x=1$}{$y'= \frac{x^x}{x+1}\big(\ln x+1-\frac1{x+1}\big)$

Tangent line: $y = (1/4)(x-1)+ 1/2$}

\exercise{$\ds y=x^{\sin (x)+2}$,\quad $x=\pi/2$}{$y'= x^{\sin (x)+2}\big(\cos x\ln x+\frac{\sin x+2}{x}\big)$

Tangent line: $y = (3\pi^2/4)(x-\pi/2)+(\pi/2)^3$}

\exercise{$\ds y=\frac{x+1}{x+2}$,\quad $x=1$}{$y'= \frac{x+1}{x+2}\big(\frac1{x+1}-\frac1{x+2}\big)$

Tangent line: $y = 1/9(x-1)+2/3$}

\exercise{$\ds y=\frac{(x+1)(x+2)}{(x+3)(x+4)}$,\quad $x=0$}{$y'= \frac{(x+1)(x+2)}{(x+3)(x+4)}\big(\frac1{x+1}+\frac1{x+2}-\frac1{x+3}-\frac1{x+4}\big)$

Tangent line: $y = 11/72x+1/6$}

}
