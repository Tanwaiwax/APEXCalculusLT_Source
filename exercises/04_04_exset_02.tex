\begin{exerciseset}{In Exercises}{, compute the differential $\dd y$.}

\exercise{$y=x^2+3x-5$}{$\dd y = (2x+3)\dd x$}

\exercise{$y=x^7-x^5$}{$\dd y = (7x^6-5x^4)\dd x$}

\exercise{$\ds y=\frac{1}{4x^2}$}{$\dd y = \frac{-2}{4x^3}\dd x$}

\exercise{$\ds y=(2x+\sin x)^2$}{$\dd y = 2(2x+\sin x)(2+\cos x)\dd x$}

\exercise{$\ds y=x^2e^{3x}$}{$\dd y = \big(2xe^{3x}+3x^2e^{3x}\big)\dd x$}

\exercise{$\ds y=\frac{4}{x^4}$}{$\dd y = \frac{-16}{x^5}\dd x$}

\exercise{$\ds y=\frac{2x}{\tan x + 1}$}{$\dd y = \frac{2(\tan x+ 1)-2x\sec^2x}{(\tan x+1)^2}\dd x$}

\exercise{$\ds y=\ln (5x)$}{$\dd y = \frac{1}{x}\dd x$}

\exercise{$\ds y=e^x\sin x$}{$\dd y = (e^x\sin x+ e^x\cos x)\dd x$}

\exercise{$\ds y=\cos (\sin x)$}{$\dd y = (-\sin(\sin x)\cos x)\dd x$}

\exercise{$\ds y=\frac{x+1}{x+2}$}{$\dd y = \frac{1}{(x+2)^2}\dd x$}

\exercise{$\ds y=3^x\ln x$}{$\dd y = ((\ln 3)3^x\ln x + \frac{3^x}{x})\dd x$}

\exercise{$\ds y=x\ln x-x$}{$\dd y = (\ln x)\dd x$}

\exercise{$y=\ln(\sec x)$}{$\dd y=(\tan x)\dd x$}

\end{exerciseset}
