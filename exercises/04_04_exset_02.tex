\exerciseset{In Exercises}{, compute the differential $dy$.}{

\exercise{$y=x^2+3x-5$}{$dy = (2x+3)dx$}

\exercise{$y=x^7-x^5$}{$dy = (7x^6-5x^4)dx$}

\exercise{$\ds y=\frac{1}{4x^2}$}{$dy = \frac{-2}{4x^3}dx$}

\exercise{$\ds y=(2x+\sin x)^2$}{$dy = 2(2x+\sin x)(2+\cos x)dx$}

\exercise{$\ds y=x^2e^{3x}$}{$dy = \big(2xe^{3x}+3x^2e^{3x}\big)dx$}

\exercise{$\ds y=\frac{4}{x^4}$}{$dy = \frac{-16}{x^5}dx$}

\exercise{$\ds y=\frac{2x}{\tan x + 1}$}{$dy = \frac{2(\tan x+ 1)-2x\sec^2x}{(\tan x+1)^2}dx$}

\exercise{$\ds y=\ln (5x)$}{$dy = \frac{1}{x}dx$}

\exercise{$\ds y=e^x\sin x$}{$dy = (e^x\sin x+ e^x\cos x)dx$}

\exercise{$\ds y=\cos (\sin x)$}{$dy = (-\sin(\sin x)\cos x)dx$}

\exercise{$\ds y=\frac{x+1}{x+2}$}{$dy = \frac{1}{(x+2)^2}dx$}

\exercise{$\ds y=3^x\ln x$}{$dy = ((\ln 3)3^x\ln x + \frac{3^x}{x})dx$}

\exercise{$\ds y=x\ln x-x$}{$dy = (\ln x)dx$}

\exercise{$y=\ln(\sec x)$}{$dy=(\tan x)dx$}

}
