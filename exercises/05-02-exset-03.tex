\exercisesetinstructions{, a graph of the velocity function of an object moving in a straight line is given. Answer the questions based on that graph.}

\exercise{\noindent
\begin{minipage}{\linewidth}
\pdftooltip{\begin{tikzpicture}
\begin{axis}[width=\marginparwidth,tick label style={font=\scriptsize},
axis y line=middle,axis x line=middle,name=myplot,axis on top,
xtick={1,2,3},ymin=-1.1,ymax=2.2,xmin=-.2,xmax=3.1]
\addplot [thick, draw={\colorone},domain=0:3] {-x+2};
\end{axis}
\node [right] at (myplot.right of origin) {\scriptsize $t$ (s)};
\node [above] at (myplot.above origin) {\scriptsize $y$ (ft/s)};
\end{tikzpicture}}{ALT-TEXT-TO-BE-DETERMINED}
\end{minipage}
\begin{enumerate}
\item		What is the object's maximum velocity?
\item		What is the object's maximum displacement?
\item		What is the object's total displacement on $[0,3]$?
\end{enumerate}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item		2ft/s
\item		2ft
\item		1.5ft
\end{enumerate}}}
