\exercisesetinstructions{, find a value $c$ guaranteed by the Mean Value Theorem.}

\exercise{$\ds \int_0^2 x^2\dd x$}{$c=2/\sqrt{3}$}

\exercise{$\ds \int_{-2}^2 x^2\dd x$}{$c=\pm 2/\sqrt{3}$}

\exercise{$\ds \int_{0}^1 e^x\dd x$}{$c=\ln(e-1)\approx 0.54$}

\exercise{$\ds \int_{0}^{16} \sqrt{x}\dd x$}{$c=64/9\approx 7.1$}

\exercisesetend
