\exercisesetinstructions{, a definite integral is given. 
\begin{enumerate}
\renewcommand{\theenumi}{(\alph{enumi})}
\item Approximate the definite integral with the Trapezoidal Rule and $n=4$.
\item Approximate the definite integral with Simpson's Rule and $n=4$.
\item Find the exact value of the integral.
\end{enumerate}}

\exercise{$\ds \int_{-1}^1 x^2\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $3/4$
\item $2/3$
\item $2/3$
\end{enumext}}}

\exercise{$\ds \int_{0}^{10} 5x\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $250$
\item $250$
\item $250$
\end{enumext}}}

\exercise{$\ds \int_{0}^{\pi} \sin x\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $\frac14(1+\sqrt2)\pi\approx 1.896$
\item $\frac1{6}(1+2\sqrt{2})\pi\approx 2.005$
\item $2$
\end{enumext}}}

\exercise{$\ds \int_{0}^{4} \sqrt x\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $2+\sqrt2+\sqrt3\approx 5.15$
\item $2/3(3+\sqrt2+2\sqrt3)\approx 5.25$
\item $16/3\approx 5.33$
\end{enumext}}}

\exercise{$\ds \int_{0}^{3} (x^3+2x^2-5x+7)\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $38.5781$
\item $147/4\approx 36.75$
\item $147/4\approx 36.75$
\end{enumext}}}

\exercise{$\ds \int_{0}^{1} x^4\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $0.2207$
\item $0.2005$
\item $1/5$
\end{enumext}}}

\exercise{$\ds \int_{0}^{2\pi} \cos x\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $0$
\item $0$
\item $0$
\end{enumext}}}

\exercise{$\ds \int_{-3}^{3} \sqrt{9-x^2}\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $9(1+\sqrt3)/2\approx 12.294$
\item $3+6\sqrt3\approx 13.392$
\item $9\pi/2\approx 14.137$
\end{enumext}}}

\exercisesetend
