\exercisesetinstructions{, find $n$ such that the error in approximating the given definite integral is less than $0.0001$ when using:
\begin{enumerate}
\renewcommand{\theenumi}{(\alph{enumi})}
\item the Trapezoidal Rule
\item Simpson's Rule
\end{enumerate}}

\exercise{$\ds \int_{0}^{\pi} \sin x\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $n=161$ (using $\max\big(\fpp(x)\big)=1$)
\item $n=12$	(using $\max\big(f\,^{(4)}(x)\big)=1$)
\end{enumext}}}

\exercise{$\ds \int_{1}^{4} \frac{1}{\sqrt x}\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $n=150$ (using $\max\big(\fpp(x)\big)=1$)
\item $n=18$	(using $\max\big(f\,^{(4)}(x)\big)=7$)
\end{enumext}}}

\exercise{$\ds \int_{0}^{\pi} \cos \big(x^2\big)\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $n=1004$ (using $\max\big(\fpp(x)\big)=39$)
\item $n=62$	(using $\max\big(f\,^{(4)}(x)\big)=800$)
\end{enumext}}}

\exercise{$\ds \int_{0}^{5} x^4\dd x$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item $n=5591$ (using $\max\big(\fpp(x)\big)=300$)
\item $n=46$	(using $\max\big(f\,^{(4)}(x)\big)=24$)
\end{enumext}}}

\exercisesetend
