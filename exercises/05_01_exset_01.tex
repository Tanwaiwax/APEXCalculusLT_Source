\begin{exerciseset}{In Exercises}{, evaluate the given indefinite integral.}

\exercise{$\ds \int 3x^3\dd x$}{$3x^4/4+C$}

\exercise{$\ds \int x^8\dd x$}{$x^9/9+C$}

\exercise{$\ds \int (10x^2-2)\dd x$}{$10x^3/3-2x+C$}

\exercise{$\ds \int\dd t$}{$t+C$}

%\exercise{$\ds \int 1\dd s$}{$s+C$}

\exercise{$\ds \int \frac{1}{3t^2}\dd t$}{$-1/(3t)+C$}

\exercise{$\ds \int \frac{3}{t^2}\dd t$}{$-3/(t)+C$}

\exercise{$\ds \int \frac{1}{\sqrt{x}}\dd x$}{$2\sqrt{x}+C$}

\exercise{$\ds \int \sec^2\theta\dd\theta$}{$\tan \theta+C$}

\exercise{$\ds \int \sin\theta\dd\theta$}{$-\cos \theta+C$}

\exercise{$\ds \int (\sec x\tan x + \csc x\cot x)\dd x$}{$\sec x - \csc x+C$}

\exercise{$\ds \int 5e^\theta\dd\theta$}{$5e^\theta+C$}

%\exercise{$\ds \int e^t\dd t$}{$e^t+C$}

\exercise{$\ds \int \frac{e^t}{2}\dd t$}{$\frac{e^t}{2}+C$}

\exercise{$\ds \int (2t+3)^2\dd t$}{$4/3t^3+6t^2+9t+C$}

\exercise{$\ds \int (t^2+3)(t^3-2t)\dd t$}{$t^6/6+t^4/4-3t^2+C$}

\exercise{$\ds \int x^2x^3\dd x$}{$x^6/6+C$}

\exercise{$\ds \int e^\pi\dd x$}{$e^\pi x+C$}

%\exercise{$\ds \int a\dd x$}{$ax+C$}

\exercise{$\ds \int \frac3{x^4}\dd x$}{$-x^{-3}+C$}

\exercise{$\ds \int \frac{4x^5-7}{x^3}\dd x$}{$\dfrac43x^3+\frac72x^{-2}+C$}

\exercise{$\ds \int \sqrt{x^7}\dd x$}{$\dfrac29x^{9/2}+C$}

\exercise{$\ds \int \frac{x^3-7x}{\sqrt x}\dd x$}{$\dfrac27x^{7/2}-\frac{14}3x^{3/2}+C$}

\exercise{$\ds \int 5-\frac23 x^2+\frac34 x^3\dd x$}{$5x-\frac29 x^3+\frac{3}{16}x^4+C$}

\exercise{$\ds \int u^6-2u^5-u^3+\frac27\dd u$}{$\frac17 u^7-\frac13 u^6-\frac14 u^4+\frac27 u+C$}

\exercise{$\ds \int (u+4)(2u+1)\dd u$}{$\frac23 u^3+\frac92 u^2+4u+C$}

\exercise{$\ds \int \sqrt t(t^2+3t+2)\dd t$}{$\frac27 t^{7/2}+\frac65 t^{5/2}+\frac43 t^{3/2}+C$}

\exercise{$\ds \int \frac{1+\sqrt x +x}{\sqrt x}\dd x$}{$2\sqrt x+x+\frac23 x\sqrt x+C$}

\exercise{$\ds \int \sin^2 x+\cos^2 x \dd x$}{$x+C$}

\exercise{$\ds \int 2+\tan^2 \theta \dd\theta$}{$\theta+\tan \theta +C$}

\exercise{$\ds \int \sec t(\sec t+\tan t)\dd t$}{$\tan t+\sec t+C$}

\exercise{$\ds \int \frac{1-\sin^2 t}{\sin^2 t}\dd t$}{$-\cot t-t+C$}

\exercise{$\ds \int \frac {\sin 2x}{\sin x}\dd x$}{$2\sin x+C$}

\exercise{$\ds \int \frac{4+6u}{\sqrt u}\dd u$}{$8\sqrt u+4u\sqrt u+C$}

\exercise{$\ds \int \frac{\sin\theta+\sin\theta\tan^2\theta}{\sec^2\theta}\dd\theta$}{$-\cos t +C$}

\exercise{$\ds \int \frac{2+t}{\root 3 \of{t^2}}\dd t$}{$6t^{1/3}+\frac34 t^{4/3}+C$}

\exercise{$\ds \int \sqrt[4]{x^5} +\sqrt[5]{x^4}\dd x$}{$\frac49 x^{9/4}+\frac49 x^{9/5}+C$}

\end{exerciseset}
