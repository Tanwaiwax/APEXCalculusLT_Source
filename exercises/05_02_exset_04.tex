\exerciseset{In Exercises}{, let
\begin{itemize}
\item $\ds \int_0^2 f(x)\ dx = 5,$
\item $\ds \int_0^3 f(x) \ dx=7,$
\item $\ds \int_0^2 g(x)\ dx = -3,$ and
\item $\ds \int_2^3 g(x)\ dx = 5.$
\end{itemize}
Use these values to evaluate the given definite integrals.}{

\exercise{$\ds \int_0^2 \big(f(x)+g(x)\big) \ dx$}{2}

\exercise{$\ds \int_0^3 \big(f(x)-g(x)\big) \ dx$}{5}

\exercise{$\ds \int_2^3 \big(3f(x)+2g(x)\big) \ dx$}{16}

\exercise{Find non-zero values for $a$ and $b$ such that 

$\ds \int_0^3 \big(af(x)+bg(x)\big) \ dx=0$}{Answers can vary; one solution is $a=-2$, $b=7$}

}
