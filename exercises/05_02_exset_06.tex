\begin{exerciseset}{In Exercises}{, evaluate the given indefinite integral.}

\exercise{$\ds \int \big(x^3-2x^2+7x-9\big)\ dx$}{$1/4x^4-2/3x^3+7/2x^2-9x+C$}

\exercise{$\ds \int \big(\sin x - \cos x + \sec^2 x\big)\ dx$}{$-\cos x - \sin x + \tan x+C$}

\exercise{$\ds \int \big(\sqrt[3]{t} + \frac{1}{t^2}+ 2^t\big)\ dt$}{$3/4t^{4/3}-1/t+2^t/\ln 2+C$}

\exercise{$\ds \int \left(\frac{1}{x} -\csc x\cot x\right)\ dx$}{$\ln\abs x + \csc x+C$}

\end{exerciseset}
