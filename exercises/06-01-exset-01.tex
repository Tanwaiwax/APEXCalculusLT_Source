\exercisesetinstructions{, evaluate the indefinite integral.}

% was set 1
\exercise{$\ds \int 3 x^2 \left(x^3-5\right)^7\dd x$}{$\frac18(x^3-5)^8+C$}

\exercise{$\ds \int (2 x-5) \left(x^2-5 x+7\right)^3\dd x$}{$\frac14(x^2-5x+7)^4+C$}

\exercise{$\ds \int x \left(x^2+1\right)^8\dd x$}{$\frac{1}{18} \left(x^2+1\right)^9+C$}

\exercise{$\ds \int (12 x+14) \left(3 x^2+7 x-1\right)^5\dd x$}{$\frac13(3 x^2+7 x-1)^6+C$}

\exercise{$\ds \int \frac{1}{2 x+7}\dd x$}{$\frac{1}{2} \ln\abs{2 x+7}+C$}

\exercise{$\ds \int \frac{1}{\sqrt{2 x+3}}\dd x$}{$\sqrt{2 x+3}+C$}

\exercise{$\ds \int \frac{x}{\sqrt{x+3}}\dd x$}{$\frac23(x+3)^{3/2}-6(x+3)^{1/2}+C = \frac{2}{3} (x-6) \sqrt{x+3}+C$}

\exercise{$\ds \int \frac{x^3-x}{\sqrt{x}}\dd x$}{$\frac{2}{21} x^{3/2} \left(3 x^2-7\right)+C$}

\exercise{$\ds \int \frac{e^{\sqrt{x}}}{\sqrt{x}}\dd x$}{$2 e^{\sqrt{x}}+C$}

\exercise{$\ds \int \frac{x^4}{\sqrt{x^5+1}}\dd x$}{$\frac{2 \sqrt{x^5+1}}{5}+C$}

\exercise{$\ds \int \frac{\frac{1}{x}+1}{x^2}\dd x$}{$-\frac{1}{2 x^2}-\frac{1}{x}+C$}

\exercise{$\ds \int \frac{\ln (x)}{x}\dd x$}{$\frac{\ln ^2(x)}{2}+C$}

% was set 2
\exercise{$\ds \int \sin ^2(x) \cos (x)\dd x$}{$\frac{\sin^3(x)}{3}+C$}

\exercise{$\ds\int\cos^3(x)\sin(x)\dd x$}{$-\frac{\cos^4(x)}4+C$}

\exercise{$\ds \int \cos (3-6 x)\dd x$}{$-\frac16\sin(3-6x)+C$}

\exercise{$\ds \int \sec ^2(4-x)\dd x$}{$-\tan (4-x)+C$}

\exercise{$\ds \int \sec (2 x)\dd x$}{$\frac{1}{2} \ln\abs{\sec (2x)+\tan(2x)}+C$}

% cut for parity
%\exercise{$\ds \int \tan ^2(x) \sec ^2(x)\dd x$}{$\frac{\tan ^3(x)}{3}+C$}

\exercise{$\ds \int x \cos \left(x^2\right)\dd x$}{$\frac{\sin \left(x^2\right)}{2}+C$}

\exercise{$\ds\int\tan^2(x)\dd x$}{$\tan(x)-x+C$}

\exercise{$\ds \int \cot x\dd x$. Do not just refer to \autoref{thm:triganti} for the answer; justify it through Substitution.}{The key is to rewrite $\cot x$ as $\cos x/\sin x$, and let $u=\sin x$.}

\exercise{$\ds \int \csc x\dd x$. Do not just refer to \autoref{thm:triganti} for the answer; justify it through Substitution.}{The key is to multiply $\csc x$ by 1 in the form $(\csc x+\cot x)/(\csc x+\cot x)$.}

\exercise{$\ds \int e^{x^3} x^2\dd x$}{$\frac{e^{x^3}}{3}+C$}

\exercise{$\ds \int e^{x^2-2 x+1} (x-1)\dd x$}{$\frac{1}{2} e^{(x-1)^2}+C$}

\exercise{$\ds \int \frac{\ln \left(x^3\right)}{x}\dd x$}{$\frac32 \ln^2(x)+C$}

\exercise{$\ds \int \frac{1}{x \ln \left(x^2\right)}\dd x$}{$\frac12 \ln\abs{\ln \left(x^2\right)}+C$}

\exercise{$\ds \int \left(3 x^2+2 x\right) \left(5 x^3+5 x^2+2\right)^8\dd x$}{$\frac{1}{45}(5x^3+5x^2+2)^9+C$}

\exercise{$\ds \int \frac{x}{\sqrt{1-x^2}}\dd x$}{$-\sqrt{1-x^2}+C$}

\exercise{$\ds \int \frac{2 x+7}{x^2+7 x+3}\dd x$}{$\ln \abs{x^2+7 x+3}+C$}

\exercise{$\ds \int \frac{3 x-3}{\sqrt{x^2-2 x-6}}\dd x $}{$3 \sqrt{x^2-2 x-6}+C$}

\exercise{$\ds\int \frac{\cos\sqrt x}{\sqrt x}\dd x$}{$2\sin\sqrt x+C$}

\exercise{$\ds\int x\sqrt{2x+3}\dd x$}{$\frac1{10}(2x+3)^{5/2}-\frac12(2x+3)^{3/2}+C$}

\exercise{$\ds\int\frac{x^3}{(x^2+1)^3}\dd x$}{$-\frac1{2(x^2+1)}+\frac1{4(x^2+1)^2}+C$}

\exercise{$\ds\int\frac{2x^5}{x^2+1}\dd x$}{$\frac12(x^2+1)^2-2(x^2+1)+\ln(x^2+1)+C$}

\exercise{$\ds\int3x^8(x^3+2)^8\dd x$}{$\frac1{11}(x^3+2)^{11}-\frac25(x^3+2)^{10}+\frac49(x^3+2)^9+C$}

\exercise{$\ds\int\sin\bigl(\frac{x}{3}\bigr)\dd x$}{$-3 \cos(\frac{x}{3}) + C$}

\exercise{$\ds\int\sin^5\bigl(\frac{x}{4}\bigr)\cos\bigl(\frac{x}{4}\bigr)\dd x$}{$\frac{2}{3}\sin^6(\frac{x}{4}) + C$}
