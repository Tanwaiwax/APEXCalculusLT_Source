\begin{exerciseset}{In Exercises}{, evaluate the indefinite integral.}

\exercise{$\ds \int \frac{7x+7}{x^2+3x-10}\dd x$}{$3 \ln\abs{x-2}+4 \ln\abs{x+5}+C$}

\exercise{$\ds \int \frac{7x-2}{x^2+x}\dd x$}{$9 \ln\abs{x+1}-2 \ln\abs{x}+C$}

\exercise{$\ds \int \frac{-4}{3x^2-12}\dd x$}{$\frac{1}{3} (\ln\abs{x+2}-\ln\abs{x-2})+C$}

\exercise{$\ds \int \frac{x+7}{(x+5)^2}\dd x$}{$\ln\abs{x+5}-\frac{2}{x+5}+C$}

\exercise{$\ds \int \frac{-3x-20}{(x+8)^2}\dd x$}{$-\frac{4}{x+8}-3 \ln\abs{x+8}+C$}

\exercise{$\ds \int \frac{9x^2+11x+7}{x(x+1)^2}\dd x$}{$\frac{5}{x+1}+7 \ln\abs x+2 \ln \abs{x+1}+C$}

\exercise{$\ds \int \frac{-12x^2-x+33}{(x-1)(x+3)(3-2x)}\dd x$}{$-\ln\abs{2x-3}+5 \ln\abs{x-1}+2 \ln\abs{x+3}+C$}

\exercise{$\ds \int \frac{94x^2-10x}{(7x+3)(5x-1)(3x-1)}\dd x$}{$-\frac{1}{5} \ln\abs{5x-1}+\frac{2}{3} \ln\abs{3 x-1}+\frac{3}{7} \ln\abs{7x+3}+C$}

\exercise{$\ds \int \frac{x^2+x+1}{x^2+x-2}\dd x$}{$x+\ln\abs{x-1}-\ln \abs{x+2}+C$}

\exercise{$\ds \int \frac{x^3}{x^2-x-20}\dd x$}{$\frac{x^2}{2}+x+\frac{125}{9} \ln\abs{x-5}+\frac{64}{9} \ln\abs{x+4}+C$}

\exercise{$\ds \int \frac{2x^2-4x+6}{x^2-2x+3}\dd x$}{$2x+C$}

\exercise{$\ds \int \frac{1}{x^3+2x^2+3x}\dd x$}{$\frac{1}{6} \left(-\ln\abs{x^2+2 x+3}+2 \ln\abs{x}-\sqrt{2} \tan ^{-1}\left(\frac{x+1}{\sqrt{2}}\right)\right)+C$}

\exercise{$\ds\int \frac{\dd x}{x^4-x^2}$}{$\frac{1}{x} + \frac{1}{2} \ln\abs{\frac{x-1}{x+1}}+ C$}

\exercise{$\ds \int \frac{x^2+x+5}{x^2+4x+10}\dd x$}{$-\frac{3}{2} \ln \abs{x^2+4 x+10}+x+\frac{\tan^{-1}\left(\frac{x+2}{\sqrt{6}}\right)}{\sqrt{6}}+C$}

\exercise{$\ds \int \frac{12x^2+21x+3}{(x+1)(3x^2+5x-1)}\dd x$}{$\ln\abs{3 x^2+5 x-1}+2 \ln\abs{x+1}+C$}

\exercise{$\ds \int \frac{6x^2+8x-4}{(x-3)(x^2+6x+10)}\dd x$}{$2\ln\abs{x-3}+2\ln\abs{x^2+6x+10}-4\tan^{-1}(x+3) +C$}

\exercise{$\ds\int \frac{1-x+2x^2-x^3}{x(x^2+1)^2}\dd x$}{$\ln\abs x-\frac{1}{2} \ln (x^2+1) - \tan^{-1} x -\frac{1}{2(x^2 +1)} + C$}

\exercise{$\ds \int \frac{2x^2+x+1}{(x+1)(x^2+9)}\dd x$}{$\frac{9}{10} \ln\abs{x^2+9}+\frac{1}{5} \ln\abs{x+1}-\frac{4}{15} \tan ^{-1}\left(\frac{x}{3}\right) +C$}

\exercise{$\ds \int \frac{x^2-20x-69}{(x-7)(x^2+2x+17)}\dd x$}{$\frac{1}{2} \left(3 \ln \abs{x^2+2 x+17}-4 \ln\abs{x-7}+\tan^{-1}\left(\frac{x+1}{4}\right)\right) +C$}

\exercise{$\ds\int \frac{x^3+x^2+2x+1}{(x^2+1)(x^2+2)}\dd x$}{$\frac{1}{2} \ln (x^2+1) - \frac{1}{\sqrt 2}\tan^{-1} \left(\frac{x}{\sqrt 2}\right) + C$}

\exercise{$\ds\int \frac{x}{x^4+4x^2+3}\dd x$}{$-\frac{1}{4} \ln (x^2+3) + \frac{1}{4} \ln (x^2+1) + C =  \frac{1}{4} \ln  \frac{x^2+1}{x^2+3} + C$}

\exercise{$\ds\int \frac{x-3}{(x^2+2x+4)^2}\dd x$}{$\frac{-1}{2(x^2+2x+4)} - \frac{2\sqrt 3}{9} \tan^{-1} \left(\frac{x+1}{\sqrt 3}\right) -\frac{2(x+1)}{3(x^2+2x+4)} + C$}

\exercise{$\ds \int \frac{9x^2-60x+33}{(x-9)(x^2-2x+11)}\dd x$}{$3 \left(\ln\abs{x^2-2 x+11}+\ln\abs{x-9}\right)+3\sqrt{\frac{2}{5}} \tan ^{-1}\left(\frac{x-1}{\sqrt{10}}\right) +C$}

\exercise{$\ds \int \frac{6x^2+45x+121}{(x+2)(x^2+10x+27)}\dd x$}{$\frac{1}{2} \ln \abs{x^2+10 x+27}+5 \ln\abs{x+2}-6 \sqrt{2} \tan^{-1}\left(\frac{x+5}{\sqrt{2}}\right) +C$}

\exercise{$\ds \int \frac1{x^4-16}\dd x$}{$\frac{1}{32}\ln\abs{x-2}-\frac1{32}\ln\abs{x+2}-\frac{1}{16}\tan^{-1}(x/2)+C$}

\exercise{$\ds \int \frac1{x^2+x}\dd x$}{$\ln x- \ln\abs{x+1}+C$}

\exercise{$\ds \int \frac1{x(x^2+1)^2}\dd x$}{$\ln x-\frac12\ln(x^2+1)+\frac12\frac1{x^2+1}+C$}

\exercise{$\ds\int\frac{2x^2}{(x^2+1)^2}\dd x$}{$\tan^{-1}x-\frac x{x^2+1}+C$}

\end{exerciseset}
