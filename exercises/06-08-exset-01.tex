\begin{exerciseset}{In Exercises}{, apply Trigonometric Substitution to evaluate the indefinite integrals.}

\exercise{$\ds \int \sqrt{x^2+1}\dd x$\label{06_08_ex_05}}{$\frac12\left(x\sqrt{x^2+1}+\ln\abs{\sqrt{x^2+1}+x}\right) + C$}

\exercise{$\ds \int \sqrt{x^2-1}\dd x$}{$\frac12x\sqrt{x^2-1}-\frac12\ln\abs{x+\sqrt{x^2-1}}+C$}

\exercise{$\ds \int \sqrt{4x^2+1}\dd x$}{$x\sqrt{x^2+1/4}+\frac14\ln\abs{2\sqrt{x^2+1/4}+2x} + C
= \frac12x\sqrt{4x^2+1}+\frac14\ln\abs{\sqrt{4x^2+1}+2x}+C$}

\exercise{$\ds \int \sqrt{1-9x^2}\dd x$}{$\frac16\sin^{-1}(3x)+\frac32\sqrt{1/9-x^2}+C=\frac16\sin^{-1}(3x)+\frac12\sqrt{1-9x^2}+C$}

\exercise{$\ds \int \sqrt{16x^2-1}\dd x$}{$4\left(\frac12x\sqrt{x^2-1/16}-\frac1{32}\ln\abs{4x+4\sqrt{x^2-1/16}}\right)+C = \frac12x\sqrt{16x^2-1}-\frac18\ln\abs{4x+\sqrt{16x^2-1}}+C$}

\exercise{$\ds \int \frac8{\sqrt{x^2+2}}\dd x$}{$8\ln\abs{\frac{\sqrt{x^2+2}}{\sqrt{2}}+\frac x{\sqrt{2}}}+C$; with \autoref{sec:hyperbolic}, we can state the answer as $8\sinh^{-1}(x/\sqrt{2})+C$.}

\exercise{$\ds \int \frac3{\sqrt{7-x^2}}\dd x$}{$3\sin^{-1}\left(\frac{x}{\sqrt{7}}\right)+C$ (Trig. Subst. is not needed)}

\exercise{$\ds \int \frac5{\sqrt{x^2-8}}\dd x$\label{06_08_ex_16}}{$5\ln\abs{\frac{x}{\sqrt{8}}+\frac{\sqrt{x^2-8}}{\sqrt{8}}}+C$}
