\exercise{$\ds \int_2^6 x\sqrt{x-2} dx $}{$352/15$}

\exercise{$\ds \int_{-\pi/2}^{\pi/2} \sin^2x \cos x\ dx $}{$2/3$}

\exercise{$\ds \int_{0}^{1} 2x(1-x^2)^4\ dx $}{$1/5$}

\exercise{$\ds \int_{-2}^{-1} (x+1)e^{x^2+2x+1}\ dx $}{$(1-e)/2$}

\exercise{$\ds\int_0^{\pi/4} e^{\tan x}\sec^2x\ dx$}{$e-1$}

\exercise{$\ds\int_{-1}^1 \frac{x}{1+x^2}\ dx$}{$0$}

\exercise{$\ds\int_1^{\ln 3} \frac{e^x}{1+e^x}\ dx$}{$\ln\left(\frac{4}{1+e}\right)$}
