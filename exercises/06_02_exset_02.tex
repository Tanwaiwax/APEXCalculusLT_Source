\exerciseset{In Exercises}{, evaluate the indefinite integral after first making a substitution.}{

\exercise{$\ds \int \sin(\ln x)\ dx$}{$1/2x\big(\sin(\ln x)-\cos (\ln x)\big)+C$
}

\exercise{$\ds \int \sin(\sqrt{x})\ dx$}{$2 \sin \left(\sqrt{x}\right)-2 \sqrt{x} \cos \left(\sqrt{x}\right)+C$
}

\exercise{$\ds \int \ln(\sqrt{x})\ dx$}{$\frac{1}{2} x \ln |x|-\frac{x}{2}+C$
}

\exercise{$\ds \int e^{\sqrt{x}}\ dx$}{$2\sqrt{x} e^{\sqrt{x}} -2 e^{\sqrt{x}}+C$
}

\exercise{$\ds \int e^{\ln x}\ dx$}{$1/2x^2+C$
}

\exercise{$\ds \int x^3 e^{x^2}\ dx$}{$\frac12 e^{x^2} (x^2-1)+C$
}
}
