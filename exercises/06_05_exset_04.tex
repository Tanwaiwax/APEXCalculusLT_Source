\exerciseset{In Exercises}{, evaluate the given indefinite integral.}{

\exercise{$\ds \int \tanh (2x)\ dx$}{$1/2\ln (\cosh(2x))+C$}

\exercise{$\ds \int \cosh (3x-7)\ dx$}{$1/3\sinh(3x-7)+C$}

\exercise{$\ds \int \sinh x\cosh x\ dx$}{$1/2\sinh^2x+C$ or $1/2\cosh^2x+C$}

\exercise{$\ds \int x\cosh x\ dx$}{$x \sinh (x)-\cosh (x)+C$
}

\exercise{$\ds \int x\sinh x\ dx$}{$x \cosh (x)-\sinh (x)+C$
}

\exercise{$\ds \int \frac{1}{9-x^2}\ dx$}{$\left\{\begin{array}{ccc} \frac13\tanh^{-1}\left(\frac x3\right)+C & & x^2<9 \\ \\
\frac13\coth^{-1}\left(\frac x3\right)+C & & 9<x^2 \end{array}\right. = \frac12\ln |x+1| - \frac12\ln |x-1|+C$}

\exercise{$\ds \int \frac{2x}{\sqrt{x^4-4}}\ dx$}{$\cosh^{-1} (x^2/2) + C = \ln (x^2+\sqrt{x^4-4})+C$}

\exercise{$\ds \int \frac{\sqrt{x}}{\sqrt{1+x^3}}\ dx$}{$2/3\sinh^{-1} x^{3/2} + C = 2/3\ln (x^{3/2}+\sqrt{x^3+1})+C$}

\exercise{$\ds \int \frac{1}{x^4-16}\ dx$}{$\frac{1}{16}\tan^{-1}(x/2)+\frac{1}{32}\ln |x-2|+\frac1{32}\ln|x+2|+C$}

\exercise{$\ds \int \frac{1}{x^2+x}\ dx$}{$\ln x- \ln|x+1|+C$}

\exercise{$\ds \int \frac{e^x}{e^{2x}+1}\ dx$}{$\tan^{-1}(e^x)+C$}

\exercise{$\ds \int \sinh^{-1}x \ dx$}{$x\sinh^{-1}x-\sqrt{x^2+1}+C$}

\exercise{$\ds \int \tanh^{-1}x \ dx$}{$x\tanh^{-1}x+1/2\ln|x^2-1|+C$}

\exercise{$\ds \int \sech x \ dx$ \quad(Hint: mutiply by $\frac{\cosh x}{\cosh x}$; set $u = \sinh x$.)}{$\tan^{-1}(\sinh x)+C$}
}
