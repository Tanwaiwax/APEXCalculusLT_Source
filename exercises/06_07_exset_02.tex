\begin{exerciseset}{In Exercises}{, use the Direct Comparison Test or the Limit Comparison Test to determine whether the given definite integral converges or diverges. Clearly state what test is being used and what function the integrand is being compared to.}

\exercise{$\ds \int_{10}^\infty \frac{3}{\sqrt{3x^2+2x-5}}\dd x$}{diverges; Limit Comparison Test with $1/x$.}

\exercise{$\ds \int_{2}^\infty \frac{4}{\sqrt{7x^3-x}}\dd x$}{converges; Limit Comparison Test with $1/x^{3/2}$.}

\exercise{$\ds \int_{0}^\infty \frac{\sqrt{x+3}}{\sqrt{x^3-x^2+x+1}}\dd x$}{diverges; Limit Comparison Test with $1/x$.}

\exercise{$\ds \int_{1}^\infty e^{-x}\ln x\dd x$}{converges; Direct Comparison Test with $xe^{-x}$.}

\exercise{$\ds \int_{5}^\infty e^{-x^2+3x+1}\dd x$}{converges; Direct Comparison Test with $e^{-x}$.}

\exercise{$\ds \int_{0}^\infty \frac{\sqrt{x}}{e^x}\dd x$}{converges; Direct Comparison Test with $xe^{-x}$.}

\exercise{$\ds \int_{2}^\infty \frac{1}{x^2+\sin x}\dd x$}{converges; Direct Comparison Test with $1/(x^2-1)$.}

\exercise{$\ds \int_{0}^\infty \frac{x}{x^2+\cos x}\dd x$}{diverges; Direct Comparison Test with $x/(x^2+1)$.}

\exercise{$\ds \int_{0}^\infty \frac{1}{x+e^x}\dd x$}{converges; Direct Comparison Test with $1/e^x$.}

\exercise{$\ds \int_{0}^\infty \frac{1}{e^x-x}\dd x$}{converges; Limit Comparison Test with $1/e^x$.}

\end{exerciseset}
