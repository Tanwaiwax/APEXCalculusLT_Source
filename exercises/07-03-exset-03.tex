\exercise{Region bounded by: $y=\sqrt{x}$, $y=0$ and $x=1$.\\
Rotate about:
\begin{multicols}{2}
\begin{enumext}[start=1]
\item		the $y$-axis
\item		$x=1$
\item		the $x$-axis
\item		$y=1$
\end{enumext}
\end{multicols}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
\item $4\pi/5$
\item $8\pi/15$
\item $\pi/2$
\item $5\pi/6$
\end{enumext}}}
