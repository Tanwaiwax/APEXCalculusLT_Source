\exercise{Region bounded by $y=x^4$, $y=0$, and $x=1$.\\
Rotate about:\\
\begin{minipage}{.5\linewidth}
 \begin{enumext}[start=1]
  \item the $y$-axis
 \end{enumext}
\end{minipage}%
\begin{minipage}{.5\linewidth}
 \begin{enumext}[start=1]\stepcounter{enumii}
  \item the $x$-axis
 \end{enumext}
\end{minipage}}{\noindent
(a) Disk: $\pi\int_0^1\left[1^2-(\sqrt[4]y)^2\right]\dd y=\frac\pi3$\\
Shell: $2\pi\int_0^1 x\cdot x^4\dd x=\frac\pi3$\\
(b) Disk: $\pi\int_0^1(x^4)^2\dd x=\frac\pi9$\\
Shell: $2\pi\int_0^1 y(1-\sqrt[4]y)\dd y=\frac\pi9$.}

\exercise{Region bounded by $y=x^3+1$, $x=0$, and $y=2$.\\
Rotate about:\\
\begin{minipage}{.5\linewidth}
 \begin{enumext}[start=1]
  \item the $y$-axis
 \end{enumext}
\end{minipage}%
\begin{minipage}{.5\linewidth}
 \begin{enumext}[start=1]\stepcounter{enumii}
  \item $y=2$
 \end{enumext}
\end{minipage}}{\noindent
(a) Disk: $\pi \int_1^2(\sqrt[3]{y-1})^2\dd y=\frac{3\pi}5$\\
Shell: $2\pi \int_0^1 x(2-(x^3+1))\dd x=\frac{3\pi}5$\\
(b) Disk: $\pi \int_0^1\left[2-(x^3+1)\right]^2\dd x=\frac{9\pi}{14}$\\
Shell: $2\pi \int_1^2 (2-y)\sqrt[3]{y-1}\dd y=\frac{9\pi}{14}$.}

%\exercise{$y=x^3+1$, $x=0$, and $y=2$ rotated about $y=2$.}{\noindent Disk: $\pi \int_0^1\left[2-(x^3+1)\right]^2\dd x=\frac{9\pi}{14}$\\

\exercise{Region bounded by $y=4x^2$ and $4x+y=8$.\\
Rotate about\\
\begin{minipage}[t]{.5\linewidth}
 \begin{enumext}[start=1]
  \item the $x$-axis
  \item $x=1$
 \end{enumext}
\end{minipage}%
\begin{minipage}[t]{.5\linewidth}
 \begin{enumext}[start=1]\addtocounter{enumii}{2}
  \item $y=16$
 \end{enumext}
\end{minipage}}{\noindent
(a) Disk: $\pi \int_{-2}^1\left[(-4x+8)^2-(4x^2)^2\right]\dd x=\frac{1152\pi}{5}$\\
Shell: $2\pi \int_0^4 y\sqrt y\dd y+2\pi \int_4^{16}y\left[\left(2-\frac y4\right)+\frac{\sqrt y}2\right]\dd y
=\frac{128\pi}5+\frac{1024\pi}5$\\
(b) Disk:
$\pi \int_0^4\left[1+\frac{\sqrt y}2\right]^2-\left[1-\frac{\sqrt y}2\right]^2\dd y
+\pi \int_4^{16}\left[1+\frac{\sqrt y}2\right]^2-\left[1-\left(2-\frac y4\right)\right]^2\dd y
=\frac{32\pi}{3}+\frac{130\pi}3$\\
Shell: $2\pi \int_{-2}^1 (1-x)\left[(-4x+8)-4x^2\right]\dd x=54\pi$\\
(c) Disk: $\pi \int_{-2}^1\left[(16-4x^2)^2-(16-(-4x+8))^2\right]\dd x
=\frac{1728\pi}{5}$\\
Shell: $2\pi \int_0^4 (16-y)\left[\sqrt y\right]\dd y
+2\pi \int_4^{16} (16-y)\left[\left(2-\frac y4\right)+\frac{\sqrt y}2\right]\dd y
=\frac{2176\pi}{15}+\frac{3008\pi}{15}$.}
