\exerciseset{In Exercises}{, find the arc length of the function on the given interval.}{

\exercise{$\ds f(x) = x$ on $[0, 1]$.}{$\sqrt{2}$}

\exercise{$\ds f(x) = \sqrt{8}x$ on $[-1, 1]$.}{$6$}

\exercise{$\ds f(x) = \frac13x^{3/2}-x^{1/2}$ on $[0,1]$.}{$4/3$}

\exercise{$\ds f(x) = \frac1{12}x^{3}+\frac1x$ on $[1,4]$.}{$6$}

\exercise{$\ds f(x) = 2x^{3/2}-\frac16\sqrt{x}$ on $[0,9]$.}{$109/2$}

\exercise{$\ds f(x) = \cosh x$ on $[-\ln 2, \ln 2]$.}{$3/2$}

\exercise{$\ds f(x) = \frac12\bigl(e^x+e^{-x}\bigr)$ on $[0, \ln 5]$.}{$12/5$}

\exercise{$\ds f(x) = \frac1{12}x^5+\frac1{5x^3}$ on $[.1, 1]$.}{$79953333/400000 \approx 199.883$}

\exercise{$\ds f(x) = \ln \bigl(\sin x\bigr)$ on $[\pi/6, \pi/2]$.}{$-\ln (2-\sqrt{3}) \approx 1.31696$}

\exercise{$\ds f(x) = \ln \bigl(\cos x\bigr)$ on $[0, \pi/4]$.}{$\sinh^{-1} 1$}

}
