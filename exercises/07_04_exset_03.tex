\exerciseset{In Exercises}{, use Simpson's Rule, with $n=4$, to approximate the arc length of the function on the given interval. Note: these are the same problems as in Exercises \ref{ex_07_04_ex_13}--\ref{ex_07_04_ex_20}.}{

\exercise{$\ds f(x) = x^2$ on $[0,1]$.}{$1.4790$}

\exercise{$\ds f(x) = x^{10}$ on $[0,1]$.}{$1.8377$}

\exercise{$\ds f(x) = \sqrt{x}$ on $[0,1]$. (Note: $\fp(x)$ is not defined at $x=0$.)}{Simpson's Rule fails, as it requires one to divide by 0. However, recognize the answer should be the same as for $y=x^2$; why?}

\exercise{$\ds f(x) = \ln x$ on $[1, e]$.}{$2.1300$}

\exercise{$\ds f(x) = \sqrt{1-x^2}$ on $[-1, 1]$. (Note: $\fp(x)$ is not defined at the endpoints.)}{Simpson's Rule fails.}

\exercise{$\ds f(x) = \sqrt{1-x^2/9}$ on $[-3, 3]$. (Note: $\fp(x)$ is not defined at the endpoints.)}{Simpson's Rule fails.}

\exercise{$\ds f(x) = \frac1x$ on $[1,2]$.}{$1.4058$}

\exercise{$\ds f(x) = \sec x$ on $[-\pi/4,\pi/4]$.}{$1.7625$}
