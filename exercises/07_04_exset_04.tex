\exerciseset{In Exercises}{, find the surface area of the described solid of revolution.}{

\exercise{The solid formed by revolving $y=2x$ on $[0,1]$ about the $x$-axis.}{$2\pi\int_0^1 2x\sqrt{5}\ dx = 2\pi\sqrt{5}$}

\exercise{The solid formed by revolving $y=x^2$ on $[0,1]$ about the $y$-axis.}{$2\pi\int_0^1 x\sqrt{1+4x^2}\ dx = \pi/6(5\sqrt{5}-1)$}

\exercise{The solid formed by revolving $y=x^3$ on $[0,1]$ about the $x$-axis.}{$2\pi\int_0^1 x^3\sqrt{1+9x^4}\ dx = \pi/27(10\sqrt{10}-1)$}

\exercise{The solid formed by revolving $y=\sqrt{x}$ on $[0,1]$ about the $x$-axis.}{$2\pi\int_0^1 \sqrt{x}\sqrt{1+1/(4x)}\ dx = \pi/6(5\sqrt{5}-1)$}

\exercise{The sphere formed by revolving $y=\sqrt{1-x^2}$ on $[-1,1]$ about the $x$-axis.}{$2\pi\int_0^1 \sqrt{1-x^2}\sqrt{1+x/(1-x^2)}\ dx = 4\pi$}
