\begin{exerciseset}{In Exercises}{, determine whether the sequence is monotonically increasing or decreasing. If it is not, determine if there is an $m$ such that it is monotonic for all $n\geq m$.}

\exercise{$\ds\{a_n\} = \left\{\frac{n}{n+2}\right\}$}{monotonically increasing}

\exercise{$\ds\{a_n\} = \left\{\frac{n^2-6n+9}{n}\right\}$}{monotonically increasing for $n\geq 3$}

\exercise{$\ds\{a_n\} = \left\{(-1)^n\frac{1}{n^3}\right\}$}{never monotonic}

\exercise{$\ds\{a_n\} = \left\{\frac{n^2}{2^n}\right\}$}{monotonically decreasing for $n\geq 3$}

\exercise{$\ds\{a_n\}=\left\{\cos \biggl(\frac{n\pi}{2}\biggr)\right\}$}{never monotonic}

\exercise{$\{a_n\}=\{ne^{-n}\}$}{monotonically decreasing}

\end{exerciseset}
