\exercisesetinstructions{, a series $\ds\sum_{n=1}^\infty a_n$ is given.
\begin{enumext}
\item Give the first 5 partial sums of the series.
\item Give a graph of the first 5 terms of $a_n$ and $S_n$ on the same axes.
\end{enumext}}

\exercise{$\ds \sum_{n=1}^\infty \frac{(-1)^n}{n}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
\item	$-1,-\frac{1}{2},-\frac{5}{6},-\frac{7}{12},-\frac{47}{60}$
\item	Plot omitted
\end{enumext}}}
