\exercisesetinstructions{, use the Limit Comparison Test to determine the convergence of the given series; state what series is used for comparison.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n^2 -3n+5}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac 1{n^2}$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{4^n-n^2}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac 1{4^n}$.}

\exercise{$\ds \sum_{n=4}^\infty \frac{\ln n}{n-3}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {\ln n}{n}$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{\sqrt{n^2+n}}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n}$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n+\sqrt{n}}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n}$.}

\exercise{$\ds\sum_{n=1}^\infty\sin\big(1/n\big)$}{Diverges; compare to $\ds\sum_{n=1}^\infty\frac1n$. Just as $\ds\lim_{n\to0}\frac{\sin n}n=1$, $\ds\lim_{n\to\infty}\frac{\sin(1/n)}{1/n}=1$.}

\exercisesetend
