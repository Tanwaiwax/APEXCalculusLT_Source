\exercisesetinstructions{, determine the convergence of the given series. State the test used; more than one test may be appropriate.}

\exercise{$\ds \sum_{n=1}^\infty \frac{n^2}{2^n}$}{Converges; Integral Test}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{(2n+5)^3}$}{Converges; Integral Test, $p$-Series Test, Direct \& Limit Comparison Tests can all be used.}

\exercise{$\ds \sum_{n=1}^\infty \frac{n!}{10^n}$}{Diverges; the $n^\text{th}$ Term Test and Direct Comparison Test can be used.}

\exercise{$\ds \sum_{n=1}^\infty \frac{\ln n}{n!}$}{Converges; the Direct Comparison Test can be used with sequence $1/(n-1)!$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{3^n+n}$}{Converges; the Direct Comparison Test can be used with sequence $1/3^n$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{n-2}{10n+5}$}{Diverges; the $n^\text{th}$ Term Test can be used, along with the Limit Comparison Test (compare with 1/10).}

\exercise{$\ds \sum_{n=1}^\infty \frac{3^n}{n^3}$}{Diverges; the $n^\text{th}$ Term Test can be used, along with the Integral Test.}

\exercise{$\ds \sum_{n=1}^\infty \frac{\cos (1/n)}{\sqrt{n}}$}{Diverges; the Limit Comparison Test can be used with sequence $1/\sqrt{n}$.}

\exercisesetend
