\begin{exerciseset}{In Exercises}{, use the Direct Comparison Test or the Limit Comparison Test to determine the convergence of the given series.  State which series is used for comparison.}

% moved to set 1
%\exercise{$\ds \sum_{n=5}^\infty \frac{1}{\sqrt{n}-2}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac{1}{\sqrt n}$, as $1/\sqrt n \leq 1/(\sqrt{n}-2)$ for all $n\geq 5$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{n^2+n+1}{n^3-5}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n}$:
\[\frac 1n = \frac{n^2}{n^3} < \frac{n^2+n+1}{n^3} < \frac{n^2+n+1}{n^3-5},\]
for all $n\geq 1$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{n-10}{n^2+10n+10}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n}$.}

% moved to set 2
%\exercise{$\ds \sum_{n=1}^\infty \sin\big(1/n\big)$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n}$. Just as $\ds \lim_{n\to0}\frac{\sin n}{n} = 1$, $\ds\lim_{n\to\infty}\frac{\sin(1/n)}{1/n} = 1$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{2^n}{5^n+10}$}{Converges; compare to $\ds \sum_{n=1}^\infty \left(\frac{2}{5}\right)^n$, as $2^n/(5^n+10) < 2^n/5^n$ for all $n\geq 1$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{n+5}{n^3-5}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n^2}$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{n}{n^4+1}$}{Converges by Comparison Test with $\sum \frac{1}{n^3}$}

\exercise{$\ds \sum_{n=1}^\infty \frac{n-1}{n4^n}$}{Converges by Comparison Test with $\sum \frac{1}{4^n}$}

\exercise{$\ds \sum_{n=2}^\infty \frac{n}{n^2-1}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac 1n$. Note that
\[\frac{n}{n^2-1} = \frac{n^2}{n^2-1}\cdot\frac1n > \frac 1n,\]
as $\frac{n^2}{n^2-1}>1$, for all $n\geq 2$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{\sqrt{n}+100}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n^{1/2}}$. }

\exercise{$\ds \sum_{n=2}^\infty \frac{1}{n^2\ln n}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac 1{n^2}$, as $1/(n^2\ln n) \leq 1/n^2$ for all $n\geq 3$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{\sqrt{n}+3}{n^2+17}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n^{3/2}}$. }
