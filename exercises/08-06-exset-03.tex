\exercisesetinstructions{, a function $\ds f(x) = \sum_{n=0}^\infty a_nx^n$ is given.
\begin{enumerate}
	\item Give a power series for $\fp(x)$ and its interval of convergence.
	\item Give a power series for $\int f(x)\dd x$ and its interval of convergence.
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty nx^n$}{\begin{enumerate}
        \item $\ds \fp(x) =  \sum_{n=1}^\infty n^2x^{n-1}$; \quad $(-1,1)$
        \item   $\ds \int f(x)\dd x = C+\sum_{n=0}^\infty \frac{n}{n+1}x^{n+1}$; \quad $(-1,1)$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{x^n}{n}$}{\begin{enumerate}
        \item $\ds \fp(x) = \sum_{n=1}^\infty x^{n-1}$; \quad $(-1,1)$
        \item   $\ds \int f(x)\dd x = C+\sum_{n=1}^\infty \frac{1}{n(n+1)}x^{n+1}$; \quad $[-1,1]$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \left(\frac{x}{2}\right)^n$}{\begin{enumerate}
        \item $\ds \fp(x) = \sum_{n=1}^\infty \frac{n}{2^n}x^{n-1}$; \quad $(-2,2)$
        \item   $\ds \int f(x)\dd x = C+\sum_{n=0}^\infty \frac{1}{(n+1)2^n}x^{n+1}$; \quad $[-2,2)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty (-3x)^n$}{\begin{enumerate}

        \item   $\ds \int f(x)\dd x = C+\sum_{n=0}^\infty \frac{(-3)^n}{n+1}x^{n+1}$; \quad $(-1/3,1/3]$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{(-1)^nx^{2n}}{(2n)!}$}{\begin{enumerate}
        \item $\ds \fp(x) = \sum_{n=1}^\infty \frac{(-1)^nx^{2n-1}}{(2n-1)!} =\sum_{n=0}^\infty \frac{(-1)^{n+1}x^{2n+1}}{(2n+1)!}$; \quad $(-\infty,\infty)$
        \item   $\ds \int f(x)\dd x = C+\sum_{n=0}^\infty \frac{(-1)^nx^{2n+1}}{(2n+1)!}$; \quad $(-\infty,\infty)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{(-1)^nx^{n}}{n!}$}{\begin{enumerate}
        \item $\ds \fp(x) = \sum_{n=1}^\infty \frac{(-1)^nx^{n-1}}{(n-1)!} =\sum_{n=0}^\infty \frac{(-1)^{n+1}x^{n}}{n!}$; \quad $(-\infty,\infty)$
        \item   $\ds \int f(x)\dd x = C+\sum_{n=0}^\infty \frac{(-1)^{n}x^{n+1}}{(n+1)!}$; \quad $(-\infty,\infty)$
\end{enumerate}}
