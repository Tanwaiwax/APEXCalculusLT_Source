\begin{exerciseset}{In Exercises}{, find the Maclaurin polynomial of degree $n$ for the given function.}

\exercise{$f(x) = e^{-x}, \quad n=3$}{$p_3(x) = 1-x+\frac12x^3-\frac16x^3$}

\exercise{$f(x) = \sin x, \quad n=8$}{$p_8(x) = x-\frac16x^3+\frac1{120}x^5-\frac1{5040}x^7$}

\exercise{$f(x) = x\cdot e^x, \quad n=5$}{$p_8(x) = x+x^2+\frac12x^3+\frac16x^4+\frac1{24}x^5$}

\exercise{$f(x) = \tan x, \quad n=6$}{$p_6(x) = \frac{2 x^5}{15}+\frac{x^3}{3}+x$}

\exercise{$f(x) = e^{2x}, \quad n=4$}{$p_4(x) = \frac{2 x^4}{3}+\frac{4 x^3}{3}+2 x^2+2 x+1$}

\exercise{$\ds f(x) = \frac1{1-x}, \quad n=4$}{$p_4(x) = x^4+x^3+x^2+x+1$}

\exercise{$\ds f(x) = \frac1{1+x}, \quad n=4$}{$p_4(x) = x^4-x^3+x^2-x+1$}

\exercise{$\ds f(x) = \frac1{1+x}, \quad n=7$}{$p_7(x) = -\frac{x^7}{7}+\frac{x^5}{5}-\frac{x^3}{3}+x$}

\end{exerciseset}
