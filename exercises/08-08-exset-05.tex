\exercisesetinstructions{, use the Taylor series  given in \autoref{idea:common_taylor} to create the Taylor series of the given functions.}

\exercise{$f(x) = \cos \bigl(x^2\bigr)$}{$\ds \sum_{n=0}^\infty (-1)^n\frac{(x^2)^{2n}}{(2n)!} = \sum_{n=0}^\infty (-1)^n\frac{x^{4n}}{(2n)!}.$}

\exercise{$f(x) = e^{-x}$}{$\ds \sum_{n=0}^\infty \frac{(-x)^n}{n!}.$}

\exercise{$f(x) = \sin\bigl(2x+3\bigr)$}{$\ds \sum_{n=0}^\infty (-1)^n\frac{(2x+3)^{2n+1}}{(2n+1)!}.$}

\exercise{$f(x) = \tan^{-1}\bigl(x/2\bigr)$}{$\ds \sum_{n=0}^\infty (-1)^n\frac{(x/2)^{2n+1}}{(2n+1)}.$}

\exercise{$f(x) = e^x\sin x$\quad (only find the first non-zero 4 terms)}{$\ds x+x^2+\frac{x^3}{3}-\frac{x^5}{30}$}

\exercise{$f(x) = (1+x)^{1/2}\cos x$\quad (only find the first non-zero 4 terms)}{$\ds 1+\frac x2-\frac{5x^2}{8}-\frac{3x^3}{16}$}

\exercisesetend
