\exercisesetinstructions{, approximate the value of the given definite integral by using the first 4 nonzero terms of the integrand's Taylor series.}

\exercise{$\ds \int_0^{\sqrt{\pi}} \sin \bigl(x^2\bigr)\dd x$}{$\ds \int_0^{\sqrt{\pi}} \sin \big(x^2\big)\dd x \approx \int_0^{\sqrt{\pi}} \left(x^2-\frac{x^6}6+\frac{x^{10}}{120}-\frac{x^{14}}{5040}\right) \dd x
= 0.8877$}

\exercise{$\ds \int_0^{\pi^2/4} \cos \bigl(\sqrt{x}\bigr)\dd x$}{$\ds \int_0^{\pi^2/4} \cos \big(\sqrt{x}\big)\dd x \approx \int_0^{\pi^2/4} \left(1-\frac x2+\frac{x^2}{24}-\frac{x^3}{720}\right)\dd x
= 1.1412$. (Actual answer: $\pi-2$)}
