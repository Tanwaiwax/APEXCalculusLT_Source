\exerciseset{In Exercises}{, a series is given.
\begin{enumerate}
\item		[(a)] Find a formula for $S_n$, the $n^\text{th}$ partial sum of the series.
\item		[(b)] Determine whether the series converges or diverges. If it converges, state what it converges to.
\end{enumerate}
}{

\exercise{$\ds \sum_{n=0}^\infty \frac{1}{4^n}$
}{\begin{enumerate}
\item	$S_n = \frac{1-(1/4)^n}{3/4}$
\item	Converges to $4/3$.
\end{enumerate}
}

\exercise{$\ds 1^3+2^3+3^3+4^3+\cdots$
}{\begin{enumerate}
\item	$S_n = \left(\frac{n(n+1)}{2}\right)^2$
\item	Diverges
\end{enumerate}
}

\exercise{$\ds \sum_{n=1}^\infty (-1)^n n$
}{\begin{enumerate}
\item	$S_n = \begin{cases}\frac{n+1}{2} & $n$ \text{ is odd }\\
	-\frac{n}{2} & $n$ \text{ is even}
	\end{cases}$
\item	Diverges
\end{enumerate}
}

\exercise{$\ds \sum_{n=0}^\infty \frac{5}{2^n}$
}{\begin{enumerate}
\item	$S_n = 5\frac{1-1/2^n}{1/2}$
\item	Converges to 10.
\end{enumerate}
}

\exercise{$\ds \sum_{n=1}^\infty e^{-n}$
}{\begin{enumerate}
\item		$S_n = \frac{1-(1/e)^{n+1}}{1-1/e}$.
\item		Converges to $1/(1-1/e) = e/(e-1)$.
\end{enumerate}
}
\exercise{$\ds 1-\frac13+\frac19-\frac{1}{27}+\frac{1}{81}+\cdots$
}{\begin{enumerate}
\item	$S_n = \frac{1-(-1/3)^n}{4/3}$
\item	Converges to $3/4$.
\end{enumerate}
}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n(n+1)}$
}{\begin{enumerate}
\item		With partial fractions, $a_n = \frac1n-\frac1{n+1}$.\\
Thus $S_n = 1-\frac1{n+1}$.
\item		Converges to 1.
\end{enumerate}
}
\exercise{$\ds \sum_{n=1}^\infty \frac{3}{n(n+2)}$
}{\begin{enumerate}
\item		With partial fractions, $a_n = \frac32\left(\frac1n-\frac1{n+2}\right)$.\\
Thus $S_n = \frac32\left(\frac32-\frac1{n+1}-\frac1{n+2}\right)$.
\item		Converges to 9/4
\end{enumerate}
}
\exercise{$\ds \sum_{n=1}^\infty \frac{1}{(2n-1)(2n+1)}$
}{\begin{enumerate}
\item	Use partial fraction decomposition to recognize the telescoping series: $S_n = \frac12\left(\frac{1}{2n-1}-\frac{1}{2n+1}\right) = \frac{n}{2n+1}$.
\item	Converges to $1/2$.
\end{enumerate}
}

\exercise{$\ds \sum_{n=1}^\infty \ln \left(\frac{n}{n+1}\right)$
}{\begin{enumerate}
\item	$S_n = \ln \big(1/(n+1)\big)$
\item	Diverges (to $-\infty$).
\end{enumerate}
}

\exercise{$\ds \sum_{n=1}^\infty \frac{2n+1}{n^2(n+1)^2}$
}{\begin{enumerate}
\item	$S_n = 1-\frac{1}{(n+1)^2}$
\item	Converges to 1.
\end{enumerate}
}

\exercise{$\ds \frac{1}{1\cdot 4}+\frac1{2\cdot5}+\frac1{3\cdot6}+\frac1{4\cdot7}+\cdots$
}{\begin{enumerate}
\item	$a_n = \frac1{n(n+3)}$; using partial fractions,\\
the resulting telescoping sum reduces to\\
$S_n = \frac13\left(1+\frac12+\frac13-\frac1{n+1}-\frac1{n+2}-\frac1{n+3}\right)$
\item	Converges to $11/18$.
\end{enumerate}
}

\exercise{$\ds 2+\left(\frac12+\frac13\right) + \left(\frac14+\frac19\right) + \left(\frac18+\frac1{27}\right)+\cdots$
}{\begin{enumerate}
\item	$a_n = 1/2^n+1/3^n$ for $n\geq 0$. Thus $S_n = \frac{1-1/2^2}{1/2}+\frac{1-1/3^n}{2/3}$.
\item		Converges to $2+3/2 = 7/2$.
\end{enumerate}
}

\exercise{$\ds \sum_{n=2}^\infty \frac{1}{n^2-1}$
}{\begin{enumerate}
\item		With partial fractions, $a_n = \frac12\left(\frac1{n-1}-\frac1{n+1}\right)$. Thus $S_n = \frac12\left(3/2-\frac1n-\frac{1}{n+1}\right)$.
\item		Converges to 3/4.
\end{enumerate}
}
\exercise{$\ds \sum_{n=0}^\infty \big(\sin 1\big)^n$
}{\begin{enumerate}
\item		$S_n=\frac{1-(\sin 1)^{n+1}}{1-\sin 1}$
\item		Converges to $\frac{1}{1-\sin 1}$..
\end{enumerate}
}
\exercise{$\ds \sum_{n=1}^\infty \biggl(\frac{2}{n(n+2)} +\frac{5}{4^n}\biggr)$
}{\begin{enumerate}
\item		$S_n=1+\frac12-\frac{1}{n+1}-\frac{1}{n+2}+\frac{\frac54(1-(\frac14)^n)}{1-\frac14}$
\item		Converges to $\frac{19}{6}$.
\end{enumerate}}

}
