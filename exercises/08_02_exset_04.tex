\exerciseset{In Exercises}{, state whether the given series converges or diverges and provide justification for your conclusion.}{

%\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n^5}$}{Converges}

\exercise{$\ds \sum_{n=0}^\infty \frac{1}{5^n}$}{Converges because it is a geometric series with $r=\frac15$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{3n^2}{n(n+2)}$}{$\ds \lim_{n\to\infty}a_n = 3$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=0}^\infty \frac{6^n}{5^n}$}{Diverges by \autoref{thm:series_nth_term}}

%\exercise{$\ds \sum_{n=1}^\infty n^{-4}$}{Converges}

\exercise{$\ds \sum_{n=1}^\infty \frac{2^n}{n^2}$}{$\ds \lim_{n\to\infty}a_n = \infty$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^\infty \sqrt{n}$}{Diverges}

\exercise{$\ds\sum_{n=0}^\infty\ln(4n+2)-\ln(7n+5)$}{$\ds\lim_{n\to\infty}a_n=\ln\frac47$; by \autoref{thm:series_nth_term} the series diverges.}

%\exercise{$\ds \sum_{n=1}^\infty \frac{10}{n!}$}{Converges}

\exercise{$\ds \sum_{n=1}^\infty \frac{5^n-n^5}{5^n+n^5}$}{$\ds \lim_{n\to\infty}a_n = 1$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^\infty \left(\frac{1}{n!}+\frac1n\right)$}{Diverges}

%\exercise{$\ds \sum_{n=1}^\infty \frac{2}{(2n+8)^2}$}{Converges}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{2n}$}{Diverges}

\exercise{$\ds \sum_{n=1}^\infty \frac{2^n+1}{2^{n+1}}$}{$\ds \lim_{n\to\infty}a_n = 1/2$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{2n-1}$}{Diverges}

\exercise{$\ds \sum_{n=1}^\infty \sqrt[n]{3}$}{Diverges by the Test for Divergence}

\exercise{$\ds \sum_{n=1}^\infty \left(1+\frac1n\right)^n$}{$\ds \lim_{n\to\infty}a_n = e$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^{\infty} \frac{\pi^n}{3^{n+1}}$}{Diverges by \autoref{thm:series_nth_term}}

\exercise{$\ds \sum_{n=1}^{\infty} \frac{3^n+2^n}{6^{n}}$}{Converges}%{$3/2$}

\exercise{$\ds \sum_{n=1}^{\infty} \frac{4^n+2^n}{6^{n}}$}{Converges}

\exercise{$\ds \sum_{n=1}^{\infty} \frac{4^n+5^n}{6^{n}}$}{Converges}

\exercise{$\ds \sum_{n=1}^{\infty} \frac{3}{n(n+1)}+\frac{5}{4^n}$}{Converges}%{$14/3$}

}
