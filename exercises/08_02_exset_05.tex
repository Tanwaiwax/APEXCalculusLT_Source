\begin{exerciseset}{In Exercises}{, find the values of $x$ for which the series converges.}

\exercise{$\ds \sum_{n=1}^\infty \frac{x^n}{3^n}$}{$(-3,3)$}

\exercise{$\ds \sum_{n=1}^\infty \frac{(x+3)^n}{2^n}$}{$(-5,-1)$}

\exercise{$\ds \sum_{n=1}^\infty \frac{4^n}{x^n}$}{$(-\infty,-4)\cup (4,\infty)$}

\exercise{$\ds \sum_{n=1}^\infty (x+2)^n$}{$(-3,-1)$}

\end{exerciseset}
