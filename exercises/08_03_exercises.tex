\begin{exerciseset}{In Exercises}{, find the value(s) of $p$ for which the series is convergent.}

\exercise{$\ds \sum_{n=2}^{\infty} \frac{1}{n(\ln n)^p}$}{$p>1$}

\exercise{$\ds \sum_{n=1}^{\infty} n(1+n^2)^p$}{$p<-1$}

\exercise{$\ds \sum_{n=1}^{\infty} \frac{\ln n}{n^p}$}{$p>1$}

\exercise{$\ds \sum_{n=3}^{\infty} \frac{1}{n\ln n[\ln(\ln n)]^p}$}{$p>1$}

\end{exerciseset}
