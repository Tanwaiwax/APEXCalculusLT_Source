\exerciseset{In Exercises}{, use the Direct Comparison Test to determine the convergence of the given series; state what series is used for comparison.}{

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n^2+3n-5}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n^2}$, as $1/(n^2+3n-5) \leq 1/n^2$ for all $n>1$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{4^n+n^2-n}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac{1}{4^n}$, as $1/(4^n+n^2-n) \leq 1/4^n$ for all $n\geq 1$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{\ln n}{n}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n}$, as $1/n \leq \ln n/n$ for all $n\geq 2$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n!+n}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n!}$, as $1/(n!+n) \leq 1/n!$ for all $n\geq 1$. }

\exercise{$\ds \sum_{n=2}^\infty \frac{1}{\sqrt{n^2-1}}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n}$. Since $n=\sqrt{n^2} > \sqrt{n^2-1}$, $1/n \leq 1/\sqrt{n^2-1}$ for all $n\geq 2$. }

\exercise{$\ds\sum_{n=5}^\infty\frac1{\sqrt{n}-2}$}{Diverges; compare to $\ds \sum_{n=1}^\infty\frac1{\sqrt n}$, as $1/\sqrt n\le1/(\sqrt{n}-2)$ for all $n\ge5$.}

}
