\exerciseset{In Exercises}{, determine the convergence of the given series using the Ratio Test. If the Ratio Test is inconclusive, state so and determine convergence with another test.}{

\exercise{$\ds\sum_{n=0}^\infty \frac{2n}{n!}$}{Converges}

\exercise{$\ds\sum_{n=0}^\infty \frac{5^n-3n}{4^n}$}{Diverges}

\exercise{$\ds\sum_{n=0}^\infty \frac{n!10^n}{(2n)!}$}{Converges}

\exercise{$\ds\sum_{n=1}^\infty \frac{5^n+n^4}{7^n+n^2}$}{Converges}

\exercise{$\ds\sum_{n=1}^\infty \frac{1}{n}$}{The Ratio Test is inconclusive; the $p$-Series Test states it diverges.}

\exercise{$\ds\sum_{n=1}^\infty \frac{1}{3n^3+7}$}{The Ratio Test is inconclusive; the Direct Comparison Test with $1/n^3$ shows it converges.}

\exercise{$\ds\sum_{n=1}^\infty \frac{10\cdot5^n}{7^n-3}$}{Converges}

\exercise{$\ds\sum_{n=1}^\infty n\cdot\left(\frac35\right)^n$}{Converges}

\exercise{$\ds\sum_{n=1}^\infty \frac{2\cdot4\cdot6\cdot8\dotsm 2n}{3\cdot6\cdot9\cdot12\dotsm 3n}$}{Converges; note the summation can be rewritten as $\ds\sum_{n=1}^\infty \frac{2^nn!}{3^nn!}$, from which the Ratio Test can be applied.}

\exercise{$\ds\sum_{n=1}^\infty \frac{n!}{5\cdot10\cdot15\dotsm (5n)}$}{Converges; rewrite the summation as $\ds\sum_{n=1}^\infty \frac{n!}{5^nn!}$ then apply the Ratio Test.}

\exercise{$\ds\sum_{n=1}^\infty e^{-n}n!$}{Diverges}

\exercise{$\ds \sum_{n=1}^\infty \frac{e^{1/n}}{n^3}$}{Converges}

}
