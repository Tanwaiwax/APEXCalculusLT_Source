\exerciseset{In Exercises}{, determine the convergence of the given series. State the test used; more than one test may be appropriate.
}{

\exercise{$\ds\sum_{n=1}^\infty \frac{n^2+4n-2}{n^3+4n^2-3n+7}$
}{Diverges; Limit Comparison Test
}
\exercise{$\ds\sum_{n=1}^\infty \frac{n^44^n}{n!}$
}{Converges; Ratio Test
}
\exercise{$\ds\sum_{n=1}^\infty \frac{n^2}{3^n+n}$
}{Converges; Ratio Test or Limit Comparison Test with $1/3^n$.
}
\exercise{$\ds\sum_{n=1}^\infty \frac{3^n}{n^n}$
}{Converges; Root Test
}
\exercise{$\ds\sum_{n=1}^\infty \frac{n}{\sqrt{n^2+4n+1}}$
}{Diverges; $n^\text{th}$-Term Test or Limit Comparison Test with 1.
}
\exercise{$\ds\sum_{n=1}^\infty \frac{n!n!n!}{(3n)!}$
}{Converges; Ratio Test
}
\exercise{$\ds\sum_{n=1}^\infty \frac{1}{\ln n}$
}{Diverges; Direct Comparison Test with $1/n$
}
\exercise{$\ds\sum_{n=1}^\infty \left(\frac{n+2}{n+1}\right)^n$
}{Diverges; $n^\text{th}$-Term Test ($n^\text{th}$ term approaches $e$.)
}
\exercise{$\ds\sum_{n=1}^\infty \frac{n^3}{\big(\ln n\big)^n}$
}{Converges; Root Test
}
\exercise{$\ds\sum_{n=1}^\infty \left(\frac1n-\frac1{n+2}\right)$
}{Converges; Limit Comparison Test with $1/n^2$ (get common denominator first). It is also a Telescoping Series.
}}
