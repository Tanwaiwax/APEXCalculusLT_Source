\exerciseset{In Exercises}{, a power series is given.
\begin{enumerate}
	\item [(a)] Find the radius of convergence.
	\item [(b)] Find the interval of convergence.
\end{enumerate}}{

\exercise{$\ds\sum_{n=0}^\infty \frac{(-1)^{n+1}}{n!}x^{n}$}{\begin{enumerate}
        \item $R=\infty$
        \item   $(-\infty,\infty)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty nx^{n}$}{\begin{enumerate}
        \item $R=1$
        \item   $(-1,1)$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{(-1)^n(x-3)^{n}}{n}$}{\begin{enumerate}
        \item $R=1$
        \item   $(2,4]$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{(x+4)^{n}}{n!}$}{\begin{enumerate}
        \item $R=\infty$
        \item   $(-\infty,\infty)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{x^{n}}{2^n}$}{\begin{enumerate}
        \item $R=2$
        \item   $(-2,2)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{(-1)^n(x-5)^{n}}{10^n}$}{\begin{enumerate}
        \item $R=10$
        \item   $(-5,15)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty 5^n(x-1)^{n}$}{\begin{enumerate}
        \item $R=1/5$
        \item   $(4/5,6/5)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty (-2)^nx^{n}$}{\begin{enumerate}
        \item $R=1/2$
        \item   $(-1/2,1/2)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \sqrt{n}x^{n}$}{\begin{enumerate}
        \item $R=1$
        \item   $(-1,1)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{n}{3^n}x^{n}$}{\begin{enumerate}
        \item $R=3$
        \item   $(-3,3)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty \frac{3^n}{n!}(x-5)^{n}$}{\begin{enumerate}
        \item $R=\infty$
        \item   $(-\infty,\infty)$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty (-1)^nn!(x-10)^{n}$}{\begin{enumerate}
        \item $R=0$
        \item   $x=10$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{x^n}{n^2}$}{\begin{enumerate}
        \item $R=1$
        \item   $[-1,1]$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{(x+2)^n}{n^3}$}{\begin{enumerate}
        \item $R=1$
        \item   $[-3,-1]$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty n!\left(\frac x{10}\right)^n$}{\begin{enumerate}
        \item $R=0$
        \item   $x=0$
\end{enumerate}}

\exercise{$\ds\sum_{n=0}^\infty n^2\left(\frac{x+4}{4}\right)^n$}{\begin{enumerate}
        \item $R=4$
        \item   $x=(-8,0)$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{(3x-2)^n}{n3^n}$}{\begin{enumerate}
        \item $R=\frac19$
        \item $x=[\frac59,\frac79]$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{x^n}{5^n n^5}$}{\begin{enumerate}
        \item $R=5$
        \item $x=[-5,5]$
\end{enumerate}}

\exercise{$\ds\sum_{n=2}^\infty \frac{x^n}{(\ln n)^n}$}{\begin{enumerate}
        \item $R=\infty$
        \item $x=(-\infty,\infty)$
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty (-1)^n\frac{x^{2n+1}}{(2n+1)!}$}{\begin{enumerate}
        \item $R=\infty$
        \item $x=(-\infty,\infty)$
\end{enumerate}}
