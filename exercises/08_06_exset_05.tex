\begin{exerciseset}{In Exercises}{, find a power series representation for the function and determine the radius of convergence.}

\exercise{$f(x)=\ln(3-x)$}{$\ds\ln3-\sum_{n=1}^\infty 3^{-n}x^n/n$, $R=3$}

\exercise{$f(x)=\dfrac{x}{(1+9x)^2}$}{$\ds\sum_{n=1}^\infty n(-9)^{n-1}x^n$, $R=1/9$}

\exercise{$f(x)=\ln\left(\dfrac{1+x}{1-x}\right)$}{$\ds\sum_{n=0}^\infty\frac2{2n+1}x^{2n+1}$, $R=1$}

\exercise{$f(x)=\tan^{-1}x$}{$\ds\sum_{n=0}^\infty(-1)^n\frac{x^{2n+1}}{2n+1}$; $R=1$}

\exercise{$f(x)=x^2\tan^{-1}(x^3)$}{$\ds\sum_{n=0}^\infty(-1)^n \frac{x^{6n+5}}{2n+1}$; $R=1$}

\exercise{$f(x)=\dfrac{1+x}{(1-x)^2}$}{$\ds\sum_{n=0}^\infty(2n+1)x^n$; $R=1$}

\exercise{$f(x)=\left(\dfrac{x}{2-x}\right)^3$}{$\ds\sum_{n=0}^\infty\frac{(n+2)(n+1)}{2^{n+4}}x^{n+3}$; $R=2$}

\end{exerciseset}
