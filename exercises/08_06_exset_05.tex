\exerciseset{In Exercises}{, write the following functions as a power series and give the radius of convergence.}{

\exercise{$\ds f(x)=\frac{x}{1-8x}$}{$\ds\sum_{n=0}^\infty 8^{n}x^{n+1}$, $R=1/8$}

\exercise{$\ds f(x)=\frac{x^7}{8+x^3}$}{$\ds\frac18\sum_{n=0}^\infty (-2)^{-3n}x^{3n+7}$, $R=2$}

\exercise{$\ds f(x)=\frac{6}{1+7x^4}$}{$\ds \sum_{n=0}^\infty 6(-7)^nx^{4n}$, $R=1/\sqrt[4]7$}

\exercise{$\ds f(x)=\frac{x^3}{3-x^2}$}{$\ds \sum_{n=0}^\infty \frac{x^{2n+3}}{3^{n+1}}$, $R=\sqrt3$}

\exercise{$\ds f(x)=\frac{3x^2}{5-2\sqrt[3]x}$}{$\ds \sum_{n=0}^\infty \frac{3\cdot2^n x^{n/3+2}}{5^{n+1}}$, $R=125/8$}

\exercise{$\ds f(x)=\frac{1}{(1+x)^2}$}{$\ds\sum_{n=0}^\infty (n+1)(-1)^n x^n$, $R=1$}

\exercise{$\ds f(x)=\frac{1}{(1+x)^3}$}{$\ds\frac12\sum_{n=0}^\infty (n+2)(n+1)(-1)^n x^n$, $R=1$}

\exercise{$\ds f(x)=\ln(3-x)$}{$\ds\ln3-\sum_{n=1}^\infty 3^{-n}x^n/n$, $R=3$}
