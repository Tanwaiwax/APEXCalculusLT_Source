\exercise{\mbox{}\\[-2\baselineskip]\begin{enumerate}
\item Use differentiation to find a power series representation for $f(x)=\dfrac1{(1+x)^2}$. What is the radius of convergence?
\item Use part (a) to find a power series for $f(x)=\dfrac1{(1+x)^3}$.
\item Use part (b) to find a power series for $f(x)=\dfrac{x^2}{(1+x)^3}$.
\end{enumerate}}{\mbox{}\\[-2\baselineskip]\begin{enumerate}
\item $\sum_{n=0}^\infty x^n(-1)^n(1+n)$; $R=1$
\item $\sum_{n=1}^\infty x^{n-1}(-1)^{n-1}(1+n)n/2\\
=\sum_{n=0}^\infty x^n(-1)^n(2+n)(1+n)/2$
\item $\sum_{n=1}^\infty x^{n+1}(-1)^{n-1}(1+n)n/2\\
=\sum_{n=0}^\infty x^{n+2}(-1)^n(2+n)(1+n)/2\\
=\sum_{n=2}^\infty x^n(-1)^n n(-1+n)/2$
\end{enumerate}}
