\begin{exerciseset}{In Exercises}{, write out the first 5 terms of the Binomial series with the given $k$-value.}

\exercise{$k=1/2$}{$\ds 1+\frac x2-\frac{x^2}{8}+\frac{x^3}{16}-\frac{5x^4}{128}$}

\exercise{$k=-1/2$}{$\ds 1-\frac x2+\frac{3x^2}{8}-\frac{5x^3}{16}+\frac{35x^4}{128}$}

\exercise{$k=1/3$}{$\ds 1+\frac x3-\frac{x^2}{9}+\frac{5x^3}{81}-\frac{10x^4}{243}$}

\exercise{$k=4$}{$\ds 1+4x+6x^2+4x^3+x^4$ (note the series is finite, and the formula still applies)}

\end{exerciseset}
