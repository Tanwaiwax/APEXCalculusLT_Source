\exerciseset{In Exercises}{, approximate the value of the given definite integral by using the first 4 nonzero terms of the integrand's Taylor series. }{

\exercise{$\ds \int_0^{\sqrt{\pi}} \sin \bigl(x^2\bigr)\ dx$}{$\ds \int_0^{\sqrt{\pi}} \sin \big(x^2\big)\ dx \approx \int_0^{\sqrt{\pi}} \left(x^2-\frac{x^6}6+\frac{x^{10}}{120}-\frac{x^{14}}{5040}\right) dx\\
= 0.8877$}

\exercise{$\ds \int_0^{\pi^2/4} \cos \bigl(\sqrt{x}\bigr)\ dx$}{$\ds \int_0^{\pi^2/4} \cos \big(\sqrt{x}\big)\ dx \approx \int_0^{\pi^2/4} \left(1-\frac x2+\frac{x^2}{24}-\frac{x^3}{720}\right)\ dx\\
= 1.1412$. (Actual answer: $\pi-2$)}

}
