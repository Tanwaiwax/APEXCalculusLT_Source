\exerciseset{In Exercises}{, determine whether the series converges absolutely, converges conditionally, or diverges.}{

\exercise{$\ds \sum_{n=1}^\infty \frac{3^n}{n^3}$}{Diverges; the $n^\text{th}$ Term Test can be used, along with the Integral Test.}

\exercise{$\ds \sum_{n=1}^\infty \cos (\pi n)$}{\begin{enumerate}
\item	$-1,0,-1,0,-1$
\item	Plot omitted
\end{enumerate}}

\exercise{$\ds \frac{1}{1\cdot 4}+\frac1{2\cdot5}+\frac1{3\cdot6}+\frac1{4\cdot7}+\dotsb$}{\begin{enumerate}
\item	$a_n = \frac1{n(n+3)}$; using partial fractions, the resulting telescoping sum reduces to $S_n=\frac13\left(1+\frac12+\frac13-\frac1{n+1}-\frac1{n+2}-\frac1{n+3}\right)$
\item	Converges to $11/18$.
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n!}$}{\begin{enumerate}
\item	$1,\frac{3}{2},\frac{5}{3},\frac{41}{24},\frac{103}{60}$
\item	Plot omitted
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{\sqrt{n}+3}{n^2+17}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n^{3/2}}$. }

\exercise{$\ds\sum_{n=1}^\infty \left(\frac1n-\frac1{n+2}\right)$}{Converges; Limit Comparison Test with $1/n^2$ (get common denominator first). It is also a Telescoping Series.}

\exercise{$\ds \sum_{n=0}^\infty \left(-\frac23\right)^n$}{\begin{enumerate}
	\item converges
	\item	converges (Geometric Series with $r=2/3$)
	\item absolute
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty (-1)^n n$}{\begin{enumerate}
\item	$S_n = \begin{cases}\frac{n+1}{2} & $n$ \text{ is odd }\\
	-\frac{n}{2} & $n$ \text{ is even}
	\end{cases}$
\item	Diverges
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{5^n-n^5}{5^n+n^5}$}{$\ds \lim_{n\to\infty}a_n = 1$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^\infty \left(-\frac{9}{10}\right)^n$}{\begin{enumerate}
\item	$-0.9,-0.09,-0.819,-0.1629,-0.75339$
\item	Plot omitted
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{n^2+4n-2}{n^3+4n^2-3n+7}$}{Diverges; Limit Comparison Test}

\exercise{$\ds \sum_{n=0}^\infty (-e)^{-n}$}{\begin{enumerate}
	\item converges
	\item	converges (Geometric Series with $r=1/e$)
	\item absolute
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n^2+3n-5}$}{Converges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n^2}$, as $1/(n^2+3n-5) \leq 1/n^2$ for all $n>1$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n^4}$}{Converges}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{3^n}$}{\begin{enumerate}
\item	$\frac{1}{3},\frac{4}{9},\frac{13}{27},\frac{40}{81},\frac{121}{243}$
\item	Plot omitted
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n(n+1)}$}{\begin{enumerate}
\item		With partial fractions, $a_n = \frac1n-\frac1{n+1}$. Thus $S_n = 1-\frac1{n+1}$.
\item		Converges to 1.
\end{enumerate}}

\exercise{$\ds \sum_{n=0}^\infty \frac{5}{2^n}$}{\begin{enumerate}
\item	$S_n = 5\frac{1-1/2^n}{1/2}$
\item	Converges to 10.
\end{enumerate}}

\exercise{$\ds\sum_{n=1}^\infty \frac{5^n+n^4}{7^n+n^2}$}{Converges}

\exercise{$\ds \sum_{n=1}^\infty \frac{n!}{10^n}$}{Diverges; the $n^\text{th}$ Term Test and Direct Comparison Test can be used.}

\exercise{$\ds \sum_{n=1}^\infty e^{-n}$}{\begin{enumerate}
\item		$S_n = \frac{1-(1/e)^{n+1}}{1-1/e}$.
\item		Converges to $1/(1-1/e) = e/(e-1)$.
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{n+\sqrt{n}}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n}$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{n!}{10^n}$}{$\ds \lim_{n\to\infty}a_n = \infty$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^\infty \frac{10}{n!}$}{Converges}

\exercise{$\ds \sum_{n=1}^\infty \frac{(-1)^n}{n}$}{\begin{enumerate}
\item	$-1,-\frac{1}{2},-\frac{5}{6},-\frac{7}{12},-\frac{47}{60}$
\item	Plot omitted
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{\ln n}{n}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac{1}{n}$, as $1/n \leq \ln n/n$ for all $n\geq 2$. }

\exercise{$\ds \sum_{n=1}^\infty \frac{2^n}{5^n+10}$}{Converges; compare to $\ds \sum_{n=1}^\infty \left(\frac{2}{5}\right)^n$, as $2^n/(5^n+10) < 2^n/5^n$ for all $n\geq 1$. }

\exercise{$\ds\sum_{n=1}^\infty \frac{n}{\sqrt{n^2+4n+1}}$}{Diverges; $n^\text{th}$-Term Test or Limit Comparison Test with 1.}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{2n}$}{Diverges}

\exercise{$\ds \sum_{n=0}^\infty \frac{(-1)^nn^2}{n!}$}{\begin{enumerate}
	\item converges
	\item	converges (Ratio Test)
	\item absolute
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{\sqrt{n^2+n}}$}{Diverges; compare to $\ds \sum_{n=1}^\infty \frac {1}{n}$.}

\exercise{$\ds \sum_{n=1}^\infty \frac{(-1)^n}{\sqrt{n}}$}{\begin{enumerate}
	\item converges
	\item	diverges ($p$-Series Test with $p=1/2$)
	\item conditional
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{\sin\big((n+1/2)\pi\big)}{n\ln n}$}{\begin{enumerate}
	\item converges
	\item	diverges (Integral Test)
	\item conditional
\end{enumerate}}

\exercise{$\ds \sum_{n=0}^\infty (-1)^{n+1}\frac{3n+5}{n^2-3n+1}$}{\begin{enumerate}
	\item converges
	\item	diverges (Limit Comparison Test with $1/n$)
	\item conditional
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{2n+1}{n^2(n+1)^2}$}{\begin{enumerate}
\item	$S_n = 1-\frac{1}{(n+1)^2}$
\item	Converges to 1.
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \frac{1}{3^n+n}$}{Converges; the Direct Comparison Test can be used with sequence $1/3^n$. }

\exercise{$\ds \sum_{n=0}^\infty (-1)^n2^{-n^2}$}{\begin{enumerate}
	\item converges
	\item	converges (Ratio Test)
	\item absolute
\end{enumerate}}

\exercise{$\ds \sum_{n=1}^\infty \left(1+\frac1n\right)^n$}{$\ds \lim_{n\to\infty}a_n = e$; by \autoref{thm:series_nth_term} the series diverges.}

\exercise{$\ds \sum_{n=1}^\infty \left(\frac{1}{n!}+\frac1n\right)$}{Diverges}

\exercise{$\ds \sum_{n=1}^\infty \frac{n}{n^2+1}$}{Diverges}

\exercise{$\ds \sum_{n=2}^\infty \frac{1}{n\ln n}$}{Diverges}

}
