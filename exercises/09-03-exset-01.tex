\exercisesetinstructions{, parametric equations for a curve are given.
\begin{enumerate}
	\item Find $\dfrac{\dd y}{\dd x}$.
	\item Find the equations of the tangent and normal line(s) at the point(s) given.
	\item Sketch the graph of the parametric functions along with the found tangent and normal lines.
\end{enumerate}}

\exercise{$x=t$, $y=t^2$;\quad $t=1$\label{ex:09_03_ex_05}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = 2t$
	\item	Tangent line: $y= 2(x-1)+1$; normal line: $y = -1/2(x-1)+1$
\end{enumerate}}}

\exercise{$x=\sqrt{t}$, $y=5t+2$;\quad $t=4$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = 10\sqrt{t}$
	\item	Tangent line: $y= 20(x-2)+22$; normal line: $y = -1/20(x-2)+22$
\end{enumerate}}}

\exercise{$x=t^2-t$, $y=t^2+t$;\quad $t=1$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = \frac{2 t+1}{2 t-1}$
	\item	Tangent line: $y= 3x+2$; normal line: $y = -1/3x+2$
\end{enumerate}}}

\exercise{$x=t^2-1$, $y=t^3-t$;\quad $t=0$ and $t=1$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = \frac{3t^2-1}{2 t}$
	\item	$t=0$: Tangent line: $x=-1$; normal line: $y = 0$
	$t=1$: Tangent line: $y=x$; normal line: $y=-x$
\end{enumerate}}}

\exercise{$x=\sec t$, $y=\tan t$ on $(-\pi/2,\pi/2)$;\quad $t=\pi/4$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = \csc t$
	\item	$t=\pi/4$: Tangent line: $y=\sqrt{2}(x-\sqrt{2})+1$; normal line: $y = -1/\sqrt{2}(x-\sqrt{2})+1$
\end{enumerate}}}

\exercise{$x=\cos t$, $y=\sin(2t)$ on $[0,2\pi]$;\quad $t=\pi/4$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = -2\cos(2t)\csc t$
	\item	$t=\pi/4$: Tangent line: $y=1$; normal line: $x=\sqrt{2}/2$
\end{enumerate}}}

\exercise{$x=\cos t\sin(2t)$, $y=\sin t\sin(2t)$ on $[0,2\pi]$; \quad $t=3\pi/4$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = \frac{\cos t\sin(2t)+2\sin t\cos(2t)}{-\sin t\sin(2t)+2\cos t\cos(2t)}$
	\item	Tangent line: $y=x-\sqrt{2}$; normal line: $y=-x$
\end{enumerate}}}

\exercise{$x=e^{t/10}\cos t$, $y=e^{t/10}\sin t$; \quad $t=\pi/2$\label{ex:09_03_ex_12}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
	\item $\frac{\dd y}{\dd x} = \frac{\sin (t)+10 \cos (t)}{\cos (t)-10 \sin (t)}$
	\item	Tangent line: $y=-x/10+e^{\pi/20}$; normal line: $y=10x+e^{\pi/20}$
\end{enumerate}}}

\exercisesetend
