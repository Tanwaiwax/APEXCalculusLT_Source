\exerciseset{In Exercises}{, find $t$-values where the curve defined by the given parametric equations has a horizontal tangent line. Note: these are the same equations as in Exercises \ref{ex:09_03_ex_05} -- \ref{ex:09_03_ex_12}.}{

\exercise{$x=t$, $y=t^2$}{$t=0$}

\exercise{$x=\sqrt{t}$, $y=5t+2$}{$t=0$ (though this uses a one--sided limit, as $x(t)$ is not defined for $t<0$.}

\exercise{$x=t^2-t$, $y=t^2+t$}{$t=-1/2$}

\exercise{$x=t^2-1$, $y=t^3-t$}{$t=\pm 1/\sqrt{3}$}

\exercise{$x=\sec t$, $y=\tan t$ on $(-\pi/2,\pi/2)$}{The graph does not have a horizontal tangent line.}

\exercise{$x=\cos t$, $y=\sin(2t)$ on $[0,2\pi]$}{$t=\pi/4, 3\pi/4,5\pi/4, 7\pi/4$}

\exercise{$x=\cos t\sin(2t)$, $y=\sin t\sin(2t)$ on $[0,2\pi]$}{The solution is non-trivial; use identities $\sin(2t)=2\sin t\cos t$ and $\cos(2t)=\cos^2t-\sin^2t$ to rewrite $g'(t) = 2\sin t(2\cos^2t-\sin^2t)$. On $[0,2\pi]$, $\sin t= 0$ when $t=0,\pi,2\pi$, and $2\cos^2t-\sin^2t=0$ when $t=\tan^{-1}(\sqrt{2}),\ \pi \pm \tan^{-1}(\sqrt{2}),\ 2\pi-\tan^{-1}(\sqrt{2})$.}

\exercise{$x=e^{t/10}\cos t$, $y=e^{t/10}\sin t$}{$t=\tan^{-1}(-10),\ \tan^{-1}(-10)+\pi$}

}
