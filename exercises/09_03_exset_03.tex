\begin{exerciseset}{In Exercises}{, find $t=t_0$ where the graph of the given parametric equations is not smooth, then find $\ds \lim_{t\to t_0}\frac{\dd y}{\dd x}$.}

\exercise{$\ds x=\frac{1}{t^2+1}$,\quad $y=t^3$}{$t_0=0$; $\lim_{t\to 0} \frac{\dd y}{\dd x} = 0$.}

\exercise{$x=-t^3+7t^2-16t+13$, \quad$y=t^3-5t^2+8t-2$}{$t_0=2$; $\lim_{t\to 2} \frac{\dd y}{\dd x} = 1$.}

\exercise{$x=t^3-3t^2+3t-1$,\quad $y=t^2-2t+1$}{$t_0=1$; $\lim_{t\to 1} \frac{\dd y}{\dd x} = \infty$.}

\exercise{$\ds x=\cos^2t$,\quad $y=1-\sin^2t$}{$t_0=\dotsc,-\pi/2,0,\pi/2,\pi,\dotsc$; $\lim_{t\to 0} \frac{\dd y}{\dd x} = 1$.}

\end{exerciseset}
