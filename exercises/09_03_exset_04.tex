\begin{exerciseset}{In Exercises}{, parametric equations for a curve are given. Find $\frac{\dd^2y}{\dd x^2}$, then determine the intervals on which the graph of the curve is concave up/down. Note: these are the same equations as in Exercises~\ref{ex:09_03_ex_05}--\ref{ex:09_03_ex_12}.}

\exercise{$x=t$,\quad  $y=t^2$}{$\frac{\dd^2y}{\dd x^2}=2$; always concave up}

\exercise{$x=\sqrt{t}$,\quad  $y=5t+2$}{$\frac{\dd^2y}{\dd x^2}=10$; always concave up}

\exercise{$x=t^2-t$,\quad  $y=t^2+t$}{$\frac{\dd^2y}{\dd x^2}=-\frac{4}{(2t-1)^3}$; concave up on $(-\infty,1/2)$; concave down on $(1/2,\infty)$.}

\exercise{$x=t^2-1$,\quad  $y=t^3-t$}{$\frac{\dd^2y}{\dd x^2}=\frac{3t^2+1}{4t^3}$; concave down on $(-\infty,0)$; concave up on $(0,\infty)$.}

\exercise{$x=\sec t$,\quad  $y=\tan t$ on $(-\pi/2,\pi/2)$}{$\frac{\dd^2y}{\dd x^2}=-\cot^3 t$; concave up on $(-\pi/2,0)$; concave down on $(0,\pi/2)$.}

\exercise{$x=\cos t$,\quad  $y=\sin(2t)$ on $[0,2\pi]$}{$\frac{\dd^2y}{\dd x^2}=\frac{\cos t\sin (2t)+2\sin t\cos(2t)}{(-\sin t\sin(2t)+2\cos t\cos(2t))^2}$; concavity switches at\\
$t=\tan^{-1}(\frac1{\sqrt{2}}),\ \frac\pi2,\ \pi-\tan^{-1}(\frac1{\sqrt{2}}),
\pi+\tan^{-1}(\frac1{\sqrt{2}}),\ \frac{3\pi}2,\ 2\pi-\tan^{-1}(\frac1{\sqrt{2}})$}

\exercise{$\ds x=\cos t\sin(2t)$,\quad $y=\sin t\sin (2t)$ on $[-\pi/2,\pi/2]$}{$\frac{\dd^2y}{\dd x^2} = \frac{4(13+3\cos(4t))}{(\cos t+3\cos(3t))^3}$, obtained with a computer algebra system; concave up on $\big(-\tan^{-1}(\frac1{\sqrt{2}}),\tan^{-1}(\frac1{\sqrt{2}})\big)$, concave down on $\big(-\frac\pi2,-\tan^{-1}(\frac1{\sqrt{2}})\big)$; $\big(\tan^{-1}(\frac1{\sqrt{2}}),\frac\pi2\big)$}

\exercise{$x=e^{t/10}\cos t$,\quad  $y=e^{t/10}\sin t$}{$\frac{\dd^2y}{\dd x^2}=\frac{1010}{e^{t/10}\big(\cos t-10\sin t\big)^3}$; concavity switches at\\
$t=\tan^{-1}(1/10)+n\pi$, where $n$ is an integer.}

\end{exerciseset}
