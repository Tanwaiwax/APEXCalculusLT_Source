\exerciseset{In Exercises}{, find the arc length of the graph of the parametric equations on the given interval(s).}{

\exercise{$x=-3\sin(2t)$,\quad  $y=3\cos(2t)$ on $[0,\pi]$}{$L=6\pi$ }

\exercise{$x=e^{t/10}\cos t$,\quad  $y=e^{t/10}\sin t$ on $[0,2\pi]$ and $[2\pi,4\pi]$}{On $[0,2\pi]$, arc length is $L = \sqrt{101}(e^{\pi/5}-1)$; on $[2\pi,4\pi]$, $L =\sqrt{101}(e^{2\pi/5}-1)$. }

\exercise{$x=5t+2$,\quad  $y=1-3t$ on $[-1,1]$}{$L=2\sqrt{34}$ }

\exercise{$x=2t^{3/2}$,\quad  $y=3t$ on $[0,1]$}{$L=4\sqrt{2}-2$ }

\exercise{$x=\cos t$,\quad $y=\sin t$ on $[0,2\pi]$}{$2\pi$}

\exercise{$x=1+3t^2$,\quad $y=4+2t^3$ on $[0,1]$}{$4\sqrt{2} -2$}

\exercise{$x=\dfrac{t}{1+t}$,\quad $y=\ln (1+t)$ on $[0,2]$}{$-\dfrac{\sqrt{10}}{3} + \ln(3+\sqrt {10}) + \sqrt 2 - \ln (1+\sqrt 2)$}

\exercise{$x=e^t-t$,\quad $y=4e^{t/2}$ on $[-8,3]$}{$e^3 + 11 - e^{-8}$}

}
