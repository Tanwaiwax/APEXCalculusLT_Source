\exerciseset{In Exercises}{, a solid of revolution is described. Find or approximate its surface area as specified. }{

\exercise{Find the surface area of the sphere formed by rotating the circle $x=2\cos t$,\quad $y=2\sin t$ about:
\begin{enumerate}
	\item	the $x$-axis and
	\item	the $y$-axis.
\end{enumerate}}{The answer is $16\pi$ for both (of course), but the integrals are different.}

\exercise{Find the surface area of the torus (or ``donut'') formed by rotating the circle $x=\cos t+2$,\quad $y=\sin t$ about the $y$-axis.}{$8\pi^2$.}

\exercise{Find the surface area of the solid formed by rotating the curve $x=a\cos^3 \theta$,\quad $y=a\sin^3 \theta$ on $[0,\pi/2]$ about the $x-$axis}{$\frac{6\pi a^2}{5}$}

\exercise{Find the surface area of the solid formed by rotating the curve $x=3t^2$,\quad $y=2t^3$ on $[0,5]$ about the $y-$axis}{$\frac{24\pi (949\sqrt{26}+1)}{5}$}

\exercise{Approximate the surface area of the solid formed by rotating the ``upper right half'' of the bow tie curve $x=\cos t$,\quad $y=\sin(2t)$ on $[0,\pi/2]$ about the $x$-axis, using Simpson's Rule and $n=4$.}{$SA\approx 8.50101$ (actual value $SA = 8.02851$}
