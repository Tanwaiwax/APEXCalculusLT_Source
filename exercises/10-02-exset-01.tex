\begin{exerciseset}{In Exercises}{, points $P$ and $Q$ are given. Write the vector $\vv{PQ}$ in component form and using the standard unit vectors.}

\exercise{$P=(2,-1)$,\quad $Q = (3,5)$}{$\vv{PQ} =\bracket{1,6}= 1\veci + 6\vecj$}

\exercise{$P=(3,2)$,\quad $Q = (7,-2)$}{$\vv{PQ} =\bracket{4,-4}= 4\veci - 4\vecj$}

\exercise{$P=(0,3,-1)$,\quad $Q = (6,2,5)$}{$\vv{PQ} =\bracket{6,-1,6}= 6\veci -\vecj+6\vec k$}

\exercise{$P=(2,1,2)$,\quad $Q = (4,3,2)$}{$\vv{PQ} =\bracket{2,2,0}= 2\veci +2\vecj$}

\end{exerciseset}
