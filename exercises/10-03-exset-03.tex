\exercisesetinstructions{, a vector $\vec v$ is given. Give two vectors that are orthogonal to $\vec v$.}

\exercise{$\vec v =\bracket{4,7}$}{Answers will vary; two possible answers are $\bracket{-7,4}$ and $\bracket{14,-8}$.}

\exercise{$\vec v =\bracket{-3,5}$}{Answers will vary; two possible answers are $\bracket{5,3}$ and $\bracket{-15,-9}$.}

\exercise{$\vec v =\bracket{1,1,1}$}{Answers will vary; two possible answers are $\bracket{1,0,-1}$ and $\bracket{4,5,-9}$.}

\exercise{$\vec v =\bracket{1,-2,3}$}{Answers will vary; two possible answers are $\bracket{2,1,0}$ and $\bracket{1,1,1/3}$.}

\exercisesetend
