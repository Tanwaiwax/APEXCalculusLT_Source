\exercise{$\vec u =\bracket{-3,2}$, $\vec v =\bracket{1,1}$}{$\proj uv =\bracket{-1/2,-1/2}$.}

\exercise{$\vec u =\bracket{-3,2}$, $\vec v =\bracket{2,3}$}{$\proj uv =\bracket{0,0}$.}

\exercise{$\vec u =\bracket{1,5,1}$, $\vec v =\bracket{1,2,3}$}{$\proj uv =\bracket{1,2,3}$.}
