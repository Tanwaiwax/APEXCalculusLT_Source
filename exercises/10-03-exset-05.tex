\exercisesetinstructions{, vectors $\vec u$ and $\vec v$ are given. Write $\vec u$ as the sum of two vectors, one of which is parallel to $\vec v$ and one of which is perpendicular to $\vec v$. Note: these are the same pairs of vectors as found in Exercises~\ref{ex:10_03_ex_21}--\ref{ex:10_03_ex_24}.}

\exercise{$\vec u =\bracket{1,2}$, $\vec v =\bracket{-1,3}$}{$\vec u =\bracket{-1/2,3/2}+\bracket{3/2,1/2}$.}

\exercise{$\vec u =\bracket{5,5}$, $\vec v =\bracket{1,3}$}{$\vec u =\bracket{2,6}+\bracket{3,-1}$.}

\exercise{$\vec u =\bracket{-3,2}$, $\vec v =\bracket{1,1}$}{$\vec u =\bracket{-1/2,-1/2}+\bracket{-5/2,5/2}$.}

\exercise{$\vec u =\bracket{-3,2}$, $\vec v =\bracket{2,3}$}{$\vec u =\bracket{0,0}+\bracket{-3,2}$.}

\exercise{$\vec u =\bracket{1,5,1}$, $\vec v =\bracket{1,2,3}$}{$\vec u =\bracket{1,2,3}+\bracket{0,3,-2}$.}

\exercise{$\vec u =\bracket{3,-1,2}$, $\vec v =\bracket{2,2,1}$}{$\vec u =\bracket{4/3,4/3,2/3}+\bracket{5/3,-7/3,4/3}$.}

\exercisesetend
