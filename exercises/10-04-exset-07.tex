\exercisesetinstructions{, find a unit vector orthogonal to both $\vec u$ and $\vec v$.}

\exercise{$\vec u =\bracket{1,1,1}$,\quad $\vec v=\bracket{2,0,1}$}{$\pm\frac{ 1}{\sqrt{6}}\bracket{1,1,-2}$}

\exercise{$\vec u =\bracket{1,-2,1}$,\quad $\vec v=\bracket{3,2,1}$}{$\pm\frac{ 1}{\sqrt{21}}\bracket{-2,1,4}$}

\exercise{$\vec u =\bracket{5,0,2}$,\quad $\vec v=\bracket{-3,0,7}$}{$\bracket{0,\pm 1,0}$}

\exercise{$\vec u =\bracket{1,-2,1}$,\quad $\vec v=\bracket{-2,4,-2}$}{any unit vector orthogonal to $\vec u$ works (such as $\frac{1}{\sqrt{2}}\bracket{1,0,-1}$).}

\exercisesetend
