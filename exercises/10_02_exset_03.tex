\exerciseset{In Exercises}{, find $\norm{\vec u}$, $\norm{\vec v}$, $\norm{\vec u+\vec v}$ and $\norm{\vec u-\vec v}$.
}{

\exercise{$\vec u=\la 2,1\ra$,\quad $\vec v = \la 3,-2\ra$
}{$\norm{\vec u} = \sqrt{5}$, $\norm{\vec v} = \sqrt{13}$, $\norm{\vec u+\vec v} = \sqrt{26}$, $\norm{\vec u-\vec v} = \sqrt{10}$
}

\exercise{$\vec u=\la -3,2,2\ra$,\quad $\vec v = \la 1,-1,1\ra$
}{$\norm{\vec u} = \sqrt{17}$, $\norm{\vec v} = \sqrt{3}$, $\norm{\vec u+\vec v} = \sqrt{14}$, $\norm{\vec u-\vec v} = \sqrt{26}$
}

\exercise{$\vec u=\la 1,2\ra$,\quad $\vec v = \la -3,-6\ra$
}{$\norm{\vec u} = \sqrt{5}$, $\norm{\vec v} = 3\sqrt{5}$, $\norm{\vec u+\vec v} = 2\sqrt{5}$, $\norm{\vec u-\vec v} = 4\sqrt{5}$
}

\exercise{$\vec u=\la 2,-3,6\ra$,\quad $\vec v = \la 10,-15,30\ra$
}{$\norm{\vec u} = 7$, $\norm{\vec v} = 35$, $\norm{\vec u+\vec v} = 42$, $\norm{\vec u-\vec v} = 28$
}
}
