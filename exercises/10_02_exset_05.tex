\exerciseset{A weight of 100lb is suspended from two chains, making angles with the vertical of $\theta$ and $\varphi$ as shown in the figure below.
\begin{center}
\myincludegraphics{figures/fig10_02_exset_05}
\end{center}
In Exercises}{,  angles $\theta$ and $\varphi$ are given. Find the force applied to each chain.
}{

\exercise{$\theta = 30^\circ$,\quad $\varphi=30^\circ$
}{The force on each chain is  $100/\sqrt{3}\approx 57.735$lb.
}

\exercise{$\theta = 60^\circ$,\quad $\varphi=60^\circ$
}{The force on each chain is  $100$lb.
}

\exercise{$\theta = 20^\circ$,\quad $\varphi=15^\circ$
}{The force on the chain with angle $\theta$ is approx. $45.124$lb; the force on the chain with angle $\varphi$ is approx. $59.629$lb.
}

\exercise{$\theta = 0^\circ$,\quad $\varphi=0^\circ$
}{The force on each chain is 50lb.
}
}
