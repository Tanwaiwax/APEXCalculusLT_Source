\exerciseset{In Exercises}{, find the measure of the angle between the two vectors in both radians and degrees.
}{

\exercise{$\vec u = \la 1,1\ra$, $\vec v = \la 1,2\ra$
}{$\theta = 0.3218 \approx 18.43^\circ$
}

\exercise{$\vec u = \la -2,1\ra$, $\vec v = \la 3,5\ra$
}{$\theta = 1.6476 \approx 94.4^\circ$
}

\exercise{$\vec u = \la 8,1,-4\ra$, $\vec v = \la 2,2,0\ra$
}{$\theta = \pi/4 = 45^\circ$
}

\exercise{$\vec u = \la 1,7,2\ra$, $\vec v = \la 4,-2,5\ra$
}{$\theta = \pi/2 = 90^\circ$
}
