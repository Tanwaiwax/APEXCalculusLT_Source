\begin{exerciseset}{In Exercises}{, write the vector, parametric and symmetric equations of the lines described.}

\exercise{Passes through $P=(2,-4,1)$, parallel to $\vec d=\bracket{9,2,5}$.}{vector: $\ell(t) =\bracket{2,-4,1}+ t\bracket{9,2,5}$\\
parametric: $x= 2+9t$, $y=-4+2t$, $z = 1+5t$\\
symmetric: $(x-2)/9 =(y+4)/2 = (z-1)/5$}

\exercise{Passes through $P=(6,1,7)$, parallel to $\vec d=\bracket{-3,2,5}$.}{vector: $\ell(t) =\bracket{6,1,7}+ t\bracket{-3,2,5}$\\
parametric: $x= 6-3t$, $y=1+2t$, $z = 7+5t$\\
symmetric: $-(x-6)/3 =(y-1)/2 = (z-7)/5$}

\exercise{Passes through $P=(2,1,5)$ and $Q = (7,-2,4)$.}{Answers can vary:
vector: $\ell(t) =\bracket{2,1,5}+ t\bracket{5,-3,-1}$\\
parametric: $x= 2+5t$, $y=1-3t$, $z = 5-t$\\
symmetric: $(x-2)/5 =-(y-1)/3 = -(z-5)$}

\exercise{Passes through $P=(1,-2,3)$ and $Q = (5,5,5)$.}{Answers can vary:
vector: $\ell(t) =\bracket{1,-2,3}+ t\bracket{4,7,2}$\\
parametric: $x= 1+4t$, $y=-2+7t$, $z = 3+2t$\\
symmetric: $(x-1)/4 =(y+2)/7 = (z-3)/2$}

\exercise{Passes through $P=(0,1,2)$ and orthogonal to both\\ $\vec d_1=\bracket{2,-1,7}$ and $\vec d_2=\bracket{7,1,3}$.}{Answers can vary; here the direction is given by $\vec d_1\times \vec d_2$:
vector: $\ell(t) =\bracket{0,1,2}+ t\bracket{-10,43,9}$\\
parametric: $x= -10t$, $y=1+43t$, $z = 2+9t$\\
symmetric: $-x/10 =(y-1)/43 = (z-2)/9$}

\exercise{Passes through $P=(5,1,9)$ and orthogonal to both\\ $\vec d_1=\bracket{1,0,1}$ and $\vec d_2=\bracket{2,0,3}$.}{Answers can vary; here the direction is given by $\vec d_1\times \vec d_2$:
vector: $\ell(t) =\bracket{5,1,9}+ t\bracket{0,-1,0}$\\
parametric: $x= 5$, $y=1-t$, $z = 9$\\
symmetric: not defined, as some components of the direction are 0.}

\exercise{Passes through the point of intersection of $\vec\ell_1(t)$ and $\vec\ell_2(t)$ and orthogonal to both lines, where\\
$\vec\ell_1(t) =\bracket{2,1,1}+ t\bracket{5,1,-2}$ and \\$\vec\ell_2(t) =\bracket{-2,-1,2}+t\bracket{3,1,-1}$.}{Answers can vary; here the direction is given by $\vec d_1\times \vec d_2$:
vector: $\ell(t) =\bracket{7,2,-1}+ t\bracket{1,-1,2}$\\
parametric: $x= 7+t$, $y=2-t$, $z = -1+2t$\\
symmetric: $x-7=2-y=(z+1)/2$}

\exercise{Passes through the point of intersection of $\ell_1(t)$ and $\ell_2(t)$ and orthogonal to both lines, where\\
$\ell_1 = \begin{cases}x=t\\y=-2+2t\\z=1+t\end{cases}$ \quad and \quad 
$\ell_2 = \begin{cases}x=2+t\\y=2-t\\z=3+2t\end{cases}$.}{Answers can vary; here the direction is given by $\vec d_1\times \vec d_2$:
vector: $\ell(t) =\bracket{2,2,3}+ t\bracket{5,-1,-3}$\\
parametric: $x= 2+5t$, $y=2-t$, $z = 3-3t$\\
symmetric: $(x-2)/5=-(y-2)=-(z-3)/3$}

\exercise{Passes through $P=(1,1)$, parallel to $\vec d =\bracket{2,3}$.}{vector: $\ell(t) =\bracket{1,1}+ t\bracket{2,3}$\\
parametric: $x= 1+2t$, $y=1+3t$\\
symmetric: $(x-1)/2=(y-1)/3$}

\exercise{Passes through $P=(-2,5)$, parallel to $\vec d =\bracket{0,1}$.}{vector: $\ell(t) =\bracket{-2,5}+ t\bracket{0,1}$\\
parametric: $x= -2$, $y=5+t$\\
symmetric: not defined}

\end{exerciseset}
