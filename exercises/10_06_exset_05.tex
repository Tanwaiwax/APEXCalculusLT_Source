}{$(-3,-7,-5)$
}

\exercise{line: $\la 4,1,0\ra + t\la 1,0,-1\ra$,\\

}{$(3,1,1)$
}

\exercise{line: $\la 1,2,3\ra + t\la 3,5,-1\ra$,\\
plane: $3x-2y-z=4$
}{No point of intersection; the plane and line are parallel.
}
