\begin{exerciseset}{In Exercises}{, give the equation of the line tangent to the graph of $\vec r(t)$ at the given $t$ value.}

\exercise{$\ds \vec r(t) =\bracket{t^2+t, t^2-t}$ at $t=1$.}{$\ell(t) =\bracket{2,0}+ t\bracket{3,1}$}

\exercise{$\ds \vec r(t) =\bracket{3\cos t,\sin t}$ at $t=\pi/4$.}{$\ell(t) =\bracket{3\sqrt{2}/2,\sqrt{2}/2}+ t\bracket{-3\sqrt{2}/2,\sqrt{2}/2}$}

\exercise{$\ds \vec r(t) =\bracket{3\cos t,3\sin t,t}$ at $t=\pi$.}{$\ell(t) =\bracket{-3,0,\pi}+ t\bracket{0,-3,1}$}

\exercise{$\ds \vec r(t) =\bracket{e^t,\tan t,t}$ at $t=0$.}{$\ell(t) =\bracket{1,0,0}+ t\bracket{1,1,1}$}

\end{exerciseset}
