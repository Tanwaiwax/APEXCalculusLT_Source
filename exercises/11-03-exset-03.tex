\exercisesetinstructions{, a position function $\vrt$ of an object is given. Find the speed of the object in terms of $t$, and find where the speed is minimized/maximized on the indicated interval.}

\exercise{$\ds \vrt =\bracket{t^2,t}$ on $[-1,1]$}{$\norm{\vvt} = \sqrt{4t^2+1}$. \\
Min at $t=0$; Max at $t=\pm 1$.}

\exercise{$\ds \vrt =\bracket{t^2,t^2-t^3}$ on $[-1,1]$}{$\norm{\vvt} =\abs t\sqrt{9t^2-12t+8}$. \\
min: $t=0$; max: $t=-1$}

\exercise{$\ds \vrt =\bracket{5\cos t,5\sin t}$ on $[0,2\pi]$}{$\norm{\vvt} = 5$. \\
Speed is constant, so there is no difference between min/max}

\exercise{$\ds \vrt =\bracket{2\cos t,5\sin t}$ on $[0,2\pi]$}{$\norm{\vvt} = \sqrt{4\sin^2t+25\cos^2t}$. \\
min: $t=\pi/2,\ 3\pi/2$; max: $t=0,\ 2\pi$}

\exercise{$\ds \vrt =\bracket{\sec t,\tan t}$ on $[0,\pi/4]$}{$\norm{\vvt} =\abs{\sec t}\sqrt{\tan^2t+\sec^2t}$. \\
min: $t=0$; max: $t=\pi/4$}

\exercise{$\ds \vrt =\bracket{t+\cos t,1-\sin t}$ on $[0,2\pi]$}{$\norm{\vvt} = \sqrt{2-2\sin t}$. \\
min: $t=\pi/2$; max: $t=3\pi/2$}

\exercise{$\ds \vrt =\bracket{12t,5\cos t,5\sin t}$ on $[0,4\pi]$}{$\norm{\vvt} = 13$. \\
speed is constant, so there is no difference between min/max}

\exercise{$\ds \vrt =\bracket{t^2-t,t^2+t,t}$ on $[0,1]$}{$\norm{\vvt} = \sqrt{8t^2+3}$. \\
min: $t=0$; max: $t=1$}

\exercise{$\ds \vrt =\bracket{t,t^2,\sqrt{1-t^2}}$ on $[-1,1]$}{$\norm{\vvt} = \sqrt{4t^2+1+t^2/(1-t^2)}$. \\
min: $t=0$; max: there is no max; speed approaches $\infty$ as $t\to\pm 1$}

\exercise{\textbf{Projectile Motion:}
$\ds \vrt =\\
\bracket{(v_0\cos \theta)t,-\frac12gt^2+(v_0\sin\theta)t}$ on $\ds \left[0,\frac{2v_0\sin\theta}g\right]$}{$\norm{\vvt} = \sqrt{g^2t^2-(2gv_0\sin\theta)t+v_0^2}$. \\
min: $t=(v_0\sin\theta)/g$; max: $t=0$, $t=(2v_0\sin\theta)/g$}

\exercisesetend
