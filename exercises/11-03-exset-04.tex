\exercisesetinstructions{, position functions $\vec r_1(t)$ and $\vec r_2(s)$ for two objects are given that follow the same path on the respective intervals.
	\begin{enumerate}
		\item Show that the positions are the same at the indicated $t_0$ and $s_0$ values; i.e., show $\vec r_1(t_0) = \vec r_2(s_0).$
		\item Find the velocity, speed and acceleration of the two objects at $t_0$ and $s_0$, respectively.
	\end{enumerate}}

\exercise{$\vec r_1(t) =\bracket{t,t^2}$ on $[0,1]$; $t_0 = 1$\\
$\vec r_2(s) =\bracket{s^2,s^4}$ on $[0,1]$; $s_0 = 1$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\vec r_1(1) =\bracket{1,1}$; $\vec r_2(1) =\bracket{1,1}$
	\item	$\vec v_1(1) =\bracket{1,2}$; $\norm{\vec v_1(1)} = \sqrt{5}$; $\vec a_1(1) =\bracket{0,2}$\\
			$\vec v_2(1) =\bracket{2,4}$; $\norm{\vec v_2(1)} = 2\sqrt{5}$; $\vec a_2(1) =\bracket{2,12}$
\end{enumext}}}

\exercise{$\vec r_1(t) =\bracket{3\cos t,3\sin t}$ on $[0,2\pi]$; $t_0 = \pi/2$\\
$\vec r_2(s) =\bracket{3\cos (4s),3\sin(4s)}$ on $[0,\pi/2]$; $s_0 = \pi/8$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\vec r_1(\pi/2) =\bracket{0,3}$; $\vec r_2(\pi/8) =\bracket{0,3}$
	\item	$\vec v_1(\pi/2) =\bracket{-3,0}$; $\norm{\vec v_1(\pi/2)} = 3$; $\vec a_1(\pi/2) =\bracket{0,-3}$\\
			$\vec v_2(\pi/8) =\bracket{-12,0}$; $\norm{\vec v_2(\pi/8)} = 12$; $\vec a_2(\pi/8) =\bracket{0,-48}$
\end{enumext}}}

\exercise{$\vec r_1(t) =\bracket{3t,2t}$ on $[0,2]$; $t_0 = 2$\\
$\vec r_2(s) =\bracket{6s-6,4s-4}$ on $[1,2]$; $s_0 = 2$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\vec r_1(2) =\bracket{6,4}$; $\vec r_2(2) =\bracket{6,4}$
	\item	$\vec v_1(2) =\bracket{3,2}$; $\norm{\vec v_1(2)} = \sqrt{13}$; $\vec a_1(2) =\bracket{0,0}$\\
			$\vec v_2(2) =\bracket{6,4}$; $\norm{\vec v_2(2)} = 2\sqrt{13}$; $\vec a_2(2) =\bracket{0,0}$
\end{enumext}}}

\exercise{$\vec r_1(t) =\bracket{t,\sqrt{t}}$ on $[0,1]$; $t_0 = 1$\\
$\vec r_2(s) =\bracket{\sin t,\sqrt{\sin t}}$ on $[0,\pi/2]$; $s_0 = \pi/2$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\vec r_1(1) =\bracket{1,1}$; $\vec r_2(\pi/2) =\bracket{1,1}$
	\item	$\vec v_1(1) =\bracket{1,1/2}$; $\norm{\vec v_1(1)} = \sqrt{5}/2$; $\vec a_1(1) =\bracket{0,-1/4}$\\
			$\vec v_2(\pi/2) =\bracket{0,0}$; $\norm{\vec v_2(\pi/2)} = 0$; $\vec a_2(\pi/2) =\bracket{-1,-1/2}$
\end{enumext}}}

\exercisesetend
