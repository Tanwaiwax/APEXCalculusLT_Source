\exercisesetinstructions{, find the position function of an object given its acceleration and initial velocity and position.}

\exercise{$\vat =\bracket{2,3}$;\quad $\vec v(0) =\bracket{1,2}$,\quad $\vec r(0) =\bracket{5,-2}$}{$\vvt =\bracket{2t+1,3t+2}$,\\
$\vrt =\bracket{t^2+t+5,3t^2/2+2t-2}$}

\exercise{$\vat =\bracket{2,3}$;\quad $\vec v(1) =\bracket{1,2}$,\quad $\vec r(1) =\bracket{5,-2}$}{$\vvt =\bracket{2t-1,3t-1}$,\\
$\vrt =\bracket{t^2-t+5,3t^2/2-t-5/2}$}

\exercise{$\vat =\bracket{\cos t,-\sin t}$;\quad $\vec v(0) =\bracket{0,1}$,\quad $\vec r(0) =\bracket{0,0}$}{$\vvt =\bracket{\sin t,\cos t}$, $\vrt =\bracket{1-\cos t,\sin t}$}

\exercise{$\vat =\bracket{0,-32}$;\quad $\vec v(0) =\bracket{10,50}$,\quad $\vec r(0) =\bracket{0,0}$}{$\vvt =\bracket{10,-32t+50}$, $\vrt =\bracket{10t,-16t^2+50t}$}
