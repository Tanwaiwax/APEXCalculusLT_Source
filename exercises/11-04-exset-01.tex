\exercisesetinstructions{, given $\vrt$, find $\vec T(t)$ and evaluate it at the indicated value of $t$.}

\exercise{$\vrt =\bracket{2t^2,t^2-t}$,\quad $t=1$\label{11_04_ex_05}}{$\vec T(t) =\bracket{\frac{4 t}{\sqrt{20 t^2-4t+1}},\frac{2 t-1}{\sqrt{20 t^2-4t+1}}}$;\\
$\vec T(1) =\bracket{4/\sqrt{17},1/\sqrt{17}}$}

\exercise{$\vrt =\bracket{t,\cos t}$,\quad $t=\pi/4$}{$\vec T(t) =\bracket{\frac{1}{\sqrt{1+\sin^2t}},-\frac{\sin t}{\sqrt{1+\sin^2t}}}$;\\
$\vec T(\pi/4) =\bracket{\sqrt{2/3}, -1/\sqrt{3}}$}

\exercise{$\vrt =\bracket{\cos^3t,\sin^3 t}$,\quad $t=\pi/4$}{$\vec T(t) = \frac{\cos t\sin t}{\sqrt{\cos^2t\sin^2t}}\bracket{-\cos t,\sin t}$. (Be careful; this cannot be simplified as $\bracket{-\cos t,\sin t}$ because $\sqrt{\cos^2t\sin^2t}\neq \cos t\sin t$, but rather $\abs{\cos t\sin t}$.)
$\vec T(\pi/4) =\bracket{-\sqrt{2}/2,\sqrt{2}/2}$}

\exercise{$\vrt =\bracket{\cos t, \sin t}$,\quad $t=\pi$ \label{11_04_ex_08}}{$\vec T(t) =\bracket{-\sin t,\cos t}$; $\vec T(\pi) =\bracket{0,-1}$}

\exercisesetend
