\begin{exerciseset}{In Exercises}{, find the equation of the osculating circle to the curve at the indicated $t$-value.}

\exercise{$\vrt =\bracket{t,t^2}$, at $t=0$}{$x^2+(y-1/2)^2 = 1/4$, or\\
$\vec c(t) =\bracket{1/2\cos t, 1/2\sin t+ 1/2}$}

\exercise{$\vrt =\bracket{3\cos t, \sin t}$, at $t=0$}{$(x-8/3)^2+y^2 = 1/9$, or $\vec c(t) =\bracket{\frac13\cos t+\frac83, \frac13\sin t}$}

\exercise{$\vrt =\bracket{3\cos t, \sin t}$, at $t=\pi/2$}{$x^2+(y+8)^2 = 81$, or $\vec c(t) =\bracket{9\cos t, 9\sin t-8}$}

\exercise{$\vrt =\bracket{t^2-t,t^2+t}$, at $t=0$}{$(x-1/2)^2+(y-1/2)^2 = 1/2$, or\\
$\vec c(t) =\bracket{\frac{\sqrt{2}}{2}\cos t+\frac12, \frac{\sqrt{2}}{2}\sin t+\frac12}$}

\end{exerciseset}
