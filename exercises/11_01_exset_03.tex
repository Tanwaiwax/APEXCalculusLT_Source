\exerciseset{In Exercises}{, find $\norm{\vec r(t)}$.
}{

\exercise{$\vec r(t) = \la t,t^2\ra$.
}{
$\norm {\vec r(t)} = \sqrt{t^2+t^4} = |t|\sqrt{t^2+1}$.
}

\exercise{$\vec r(t) = \la 5\cos t,3\sin t\ra$.
}{
$\norm {\vec r(t)} = \sqrt{25\cos^2t+9\sin^2 t}$.
}

\exercise{$\vec r(t) = \la 2\cos t,2\sin t,t\ra$.
}{
$\norm {\vec r(t)} = \sqrt{4\cos^2t+4\sin^2 t+t^2} = \sqrt{t^2+4}$.
}

\exercise{$\vec r(t) = \la \cos t,t,t^2\ra$.
}{
$\norm {\vec r(t)} = \sqrt{\cos^2t+t^2+t^4}$.
}
}
