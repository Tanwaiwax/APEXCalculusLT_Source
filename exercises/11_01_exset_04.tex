\exerciseset{In Exercises}{, create a vector--valued function whose graph matches the given description.}{

\exercise{A circle of radius 2, centered at $(1,2)$, traced counter--clockwise once on $[0,2\pi]$.}{Answers may vary, though most direct solution is\\
$\vec r(t) =\bracket{2\cos t+1,2\sin t+2}$.}

\exercise{A circle of radius 3, centered at $(5,5)$, traced clockwise once on $[0,2\pi]$.}{Answers may vary; three solutions are\\
$\vec r(t) =\bracket{3\sin t+5,3\cos t+5}$, \\
$\vec r(t) =\bracket{-3\cos t+5,3\sin t+5}$ and \\
$\vec r(t) =\bracket{3\cos t+5,-3\sin t+5}$.}

\exercise{An ellipse, centered at $(0,0)$ with vertical major axis of length 10 and minor axis of length 3, traced once counter--clockwise on $[0,2\pi]$.}{Answers may vary, though most direct solution is\\
$\vec r(t) =\bracket{1.5\cos t,5\sin t}$.}

\exercise{An ellipse, centered at $(3,-2)$ with horizontal major axis of length 6 and minor axis of length 4, traced once clockwise on $[0,2\pi]$.}{Answers may vary, though most direct solutions are\\
$\vec r(t) =\bracket{-3\cos t+3,2\sin t-2}$,\\
$\vec r(t) =\bracket{3\cos t+3,-2\sin t-2}$ and \\
$\vec r(t) =\bracket{3\sin t+3,2\cos t-2}$.}

\exercise{A line through $(2,3)$ with a slope of 5.}{Answers may vary, though most direct solutions are \\
$\vec r(t) =\bracket{t,5(t-2)+3}$ and \\
$\vec r(t) =\bracket{t+2,5t+3}$.}

\exercise{A line through $(1,5)$ with a slope of $-1/2$.}{Answers may vary, though most direct solutions are \\
$\vec r(t) =\bracket{t,-1/2(t-1)+5}$,\\
$\vec r(t) =\bracket{t+1,-1/2t+5}$, \\
$\vec r(t) =\bracket{-2t+1,t+5}$ and \\
$\vec r(t) =\bracket{2t+1,-t+5}$.}

\exercise{A vertically oriented helix with radius of 2 that starts at $(2,0,0)$ and ends at $(2,0,4\pi)$ after 1 revolution on $[0,2\pi]$.}{Answers may vary, though most direct solution is\\
$\vec r(t) =\bracket{2\cos t,2\sin t,2t}$.}

\exercise{A vertically oriented helix with radius of 3 that starts at $(3,0,0)$ and ends at $(3,0,3)$ after 2 revolutions on $[0,1]$.}{Answers may vary, though most direct solution is\\
$\vec r(t) =\bracket{3\cos (4\pi t),3\sin (4\pi t),3t}$.}

\exercise{A vertically oriented helix with radius $R$ which completes $N$ revolutions over a height of $h$ on $[0,1]$.}{Answers may vary, though most direct solution is $\vec r(t)=\bracket{R\cos(2\pi Nt),R\sin(2\pi Nt),h}$.}

\exercise{The intersection of the sphere $x^2 + y^2 + z^2 = 9$ and the plane $y = 2$ on $[0,1]$.}{Answers may vary, though most direct solution is $\vec r(t)=\bracket{\sqrt5\cos(2\pi t),2,\sqrt5\sin(2\pi t)}$.}

}
