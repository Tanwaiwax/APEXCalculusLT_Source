\exerciseset{In Exercises}{, find the value(s) of $t$ for which $\vec r(t)$ is not smooth.}{

\exercise{$\ds \vec r(t) =\bracket{\cos t,\sin t - t}$}{$t=2n\pi$, where $n$ is an integer;\\
so $t = \dotsc-4\pi,-2\pi,0,2\pi,4\pi,\dotsc$}

\exercise{$\ds \vec r(t) =\bracket{t^2-2t+1,t^3+t^2-5t+3}$}{$t=1$}

\exercise{$\ds \vec r(t) =\bracket{\cos t-\sin t, \sin t - \cos t,\cos(4t)}$}{$\vec r(t)$ is not smooth at $t=3\pi/4+n\pi$, where $n$ is an integer}

\exercise{$\ds \vec r(t) =\bracket{t^3-3t+2, -\cos(\pi t),\sin^2(\pi t)}$}{$t=\pm 1$}

}
