\exercise{Find $\vec r(t)$, given that $\vrp(t) = \la t,\sin t\ra$ and $\vec r(0) = \la 2,2\ra$.
}{
$\vec r(t) = \la \frac12t^2+2,-\cos t+3\ra$
}

\exercise{Find $\vec r(t)$, given that $\vrp(t) = \la 1/(t+1),\tan t\ra$ and \\ $\vec r(0) = \la 1,2\ra$.
}{
$\vec r(t) = \la \ln |t+1| + 1, -\ln |\cos t| + 2\ra$
}

\exercise{Find $\vec r(t)$, given that $\vrp'(t) = \la t^2,t,1\ra$,\\ $\vrp(0) = \la 1,2,3\ra$ and $\vec r(0) = \la 4,5,6\ra$.
}{
$\vec r(t) = \la t^4/12+t+4,\ t^3/6+2t+5,\ t^2/2+3t+6\ra$
}

\exercise{Find $\vec r(t)$, given that $\vrp'(t) = \la \cos t,\sin t,e^t\ra$,\\  $\vrp(0) = \la 0,0,0\ra$ and $\vec r(0) = \la 0,0,0\ra$.
}{
$\vec r(t) = \la-\cos t+1,t-\sin t,e^t-t-1 \ra$
}
