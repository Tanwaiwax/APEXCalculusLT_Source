\begin{exerciseset}{In Exercises}{, find the displacement, distance traveled, average velocity and average speed of the described object on the given interval.}

\exercise{An object with position function $\vrt =\bracket{2\cos t,2\sin t, 3t}$, where distances are measured in feet and time is in seconds, on $[0,2\pi]$.}{Displacement: $\bracket{0,0,6\pi}$; distance traveled: $2\sqrt{13}\pi \approx 22.65$ft; average velocity: $\bracket{0,0,3}$; average speed: $\sqrt{13} \approx 3.61$ft/s}

\exercise{An object with position function $\vrt =\bracket{5\cos t,-5\sin t}$, where distances are measured in feet and time is in seconds, on $[0,\pi]$.}{Displacement: $\bracket{-10,0}$; distance traveled: $5\pi \approx 15.71$ft; average velocity: $\bracket{-10/\pi,0}\approx\bracket{-3.18,0}$; average speed: $5$ft/s}

\exercise{An object with velocity function $\vvt =\bracket{\cos t,\sin t}$, where distances are measured in feet and time is in seconds, on $[0,2\pi]$.}{Displacement: $\bracket{0,0}$; distance traveled: $2\pi \approx 6.28$ft; average velocity: $\bracket{0,0}$; average speed: $1$ft/s}

\exercise{An object with velocity function $\vvt =\bracket{1,2,-1}$, where distances are measured in feet and time is in seconds, on $[0,10]$.}{Displacement: $\bracket{10,20,-20}$; distance traveled: $30$ft; average velocity: $\bracket{1,2,-2}$; average speed: $3$ft/s}

\end{exerciseset}
