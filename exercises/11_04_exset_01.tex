\exerciseset{In Exercises}{ , given $\vrt$, find $\vec T(t)$ and evaluate it at the indicated value of $t$.
}{

\exercise{$\vrt = \la 2t^2,t^2-t\ra$,\quad $t=1$\label{11_04_ex_05}
}{$\vec T(t) = \la\frac{4 t}{\sqrt{20 t^2-4t+1}},\frac{2 t-1}{\sqrt{20 t^2-4t+1}}\ra$;
$\vec T(1) = \la 4/\sqrt{17},1/\sqrt{17}\ra$
}
\exercise{$\vrt = \la t,\cos t\ra$,\quad $t=\pi/4$
}{$\vec T(t) = \la \frac{1}{\sqrt{1+\sin^2t}},-\frac{\sin t}{\sqrt{1+\sin^2t}} \ra$;
$\vec T(\pi/4) = \la \sqrt{2/3}, -1/\sqrt{3}\ra$
}
\exercise{$\vrt = \la \cos^3t,\sin^3 t\ra$,\quad $t=\pi/4$
}{$\vec T(t) = \frac{\cos t\sin t}{\sqrt{\cos^2t\sin^2t}}\la -\cos t,\sin t\ra$. (Be careful; this cannot be simplified as just $\la -\cos t,\sin t\ra$ as $\sqrt{\cos^2t\sin^2t}\neq \cos t\sin t$, but rather $|\cos t\sin t|$.)
$\vec T(\pi/4) = \la -\sqrt{2}/2,\sqrt{2}/2\ra$
}
\exercise{$\vrt = \la \cos t, \sin t \ra$,\quad $t=\pi$ \label{11_04_ex_08}
}{$\vec T(t) = \la -\sin t,\cos t\ra$; 
$\vec T(\pi) = \la 0,-1\ra$
}}
