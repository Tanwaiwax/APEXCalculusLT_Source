\exerciseset{In Exercises}{ , find the equation of the line tangent to the curve at the indicated $t$-value using the unit tangent vector. Note: these are the same problems as in Exercises \ref{11_04_ex_05} -- \ref{11_04_ex_08}.
}{

\exercise{$\vrt = \la 2t^2,t^2-t \ra$,\quad $t=1$
}{$\ell(t) = \la 2,0\ra + t\la 4/\sqrt{17},1/\sqrt{17}\ra$; in parametric form,

$\ell(t) = \left\{\begin{array}{ccc} x&=&2+4t/\sqrt{17} \\ y &=& t/\sqrt{17}\end{array}\right.$
}
\exercise{$\vrt = \la t,\cos t \ra$,\quad $t=\pi/4$
}{$\ell(t) = \la \pi/4,\sqrt{2}/{2}\ra + t\la \sqrt{2/3},-1/\sqrt{3}\ra$; in parametric form,

$\ell(t) = \left\{\begin{array}{ccc} x&=&\pi/4+\sqrt{2/3}t \\ y &=& \sqrt{2}/2-t/\sqrt{3}\end{array}\right.$
}
\exercise{$\vrt = \la \cos^3 t,\sin^3t \ra$,\quad $t=\pi/4$
}{$\ell(t) = \la \sqrt{2}/4,\sqrt{2}/4\ra + t\la -\sqrt{2}/2,\sqrt{2}/2\ra$; in parametric form,

$\ell(t) = \left\{\begin{array}{ccc} x&=&\sqrt{2}/4-\sqrt{2}t/2 \\ y &=& \sqrt{2}/4+\sqrt{2}t/2\end{array}\right.$
}
\exercise{$\vrt = \la \cos t,\sin t \ra$,\quad $t=\pi$
}{$\ell(t) = \la -1,0\ra + t\la 0,-1\ra$; in parametric form,

$\ell(t) = \left\{\begin{array}{ccc} x&=&-1 \\ y &=& -t\end{array}\right.$
}}
