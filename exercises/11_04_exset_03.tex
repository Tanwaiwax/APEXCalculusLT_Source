\exerciseset{In Exercises}{, find $\vec N(t)$ using \autoref{def:unit_normal}. Confirm the result using \autoref{thm:concave}.}{

\exercise{$\vrt =\bracket{3\cos t, 3\sin t}$}{$\vec T(t) =\bracket{-\sin t,\cos t}$; $\vec N(t) =\bracket{-\cos t,-\sin t}$}

\exercise{$\vrt =\bracket{t, t^2}$}{$\vec T(t) =\bracket{\frac{1}{\sqrt{1+4t^2}},\frac{2t}{\sqrt{1+4t^2}}}$; $\vec N(t) =\bracket{-\frac{2t}{\sqrt{1+4t^2}},\frac{1}{\sqrt{1+4t^2}}}$}

\exercise{$\vrt =\bracket{\cos t,2\sin t}$}{$\vec T(t) =\bracket{-\frac{\sin t}{\sqrt{4\cos^2t+\sin^2t}},\frac{2\cos t}{\sqrt{4\cos^2t+\sin^2t}}}$;\\
$\vec N(t) =\bracket{-\frac{2\cos t}{\sqrt{4\cos^2t+\sin^2t}},-\frac{\sin t}{\sqrt{4\cos^2t+\sin^2t}}}$}

\exercise{$\vrt =\bracket{e^t,e^{-t}}$}{$\vec T(t) =\bracket{\frac{e^t}{\sqrt{e^{2t}+e^{-2t}}},-\frac{e^{-t}}{\sqrt{e^{2t}+e^{-2t}}}}$;\\
$\vec N(t) =\bracket{\frac{e^{-t}}{\sqrt{e^{2t}+e^{-2t}}},\frac{e^{t}}{\sqrt{e^{2t}+e^{-2t}}}}$}

}
