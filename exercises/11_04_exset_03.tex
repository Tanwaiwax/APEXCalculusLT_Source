\exerciseset{In Exercises}{, find $\vec N(t)$ using \autoref{def:unit_normal}. Confirm the result using \autoref{thm:concave}.}{

\exercise{$\vrt = \la 3\cos t, 3\sin t \ra$}{$\vec T(t) = \la -\sin t,\cos t\ra$; $\vec N(t) = \la -\cos t,-\sin t\ra$}

\exercise{$\vrt = \la t, t^2 \ra$}{$\vec T(t) = \la \frac{1}{\sqrt{1+4t^2}},\frac{2t}{\sqrt{1+4t^2}}\ra$; $\vec N(t) = \la -\frac{2t}{\sqrt{1+4t^2}},\frac{1}{\sqrt{1+4t^2}}\ra$}

\exercise{$\vrt = \la \cos t,2\sin t \ra$}{$\vec T(t) = \la -\frac{\sin t}{\sqrt{4\cos^2t+\sin^2t}},\frac{2\cos t}{\sqrt{4\cos^2t+\sin^2t}}\ra$; $\vec N(t) = \la -\frac{2\cos t}{\sqrt{4\cos^2t+\sin^2t}},-\frac{\sin t}{\sqrt{4\cos^2t+\sin^2t}}\ra$}

\exercise{$\vrt = \la e^t,e^{-t} \ra$}{$\vec T(t) = \la \frac{e^t}{\sqrt{e^{2t}+e^{-2t}}},-\frac{e^{-t}}{\sqrt{e^{2t}+e^{-2t}}}\ra$; $\vec N(t) = \la \frac{e^{-t}}{\sqrt{e^{2t}+e^{-2t}}},\frac{e^{t}}{\sqrt{e^{2t}+e^{-2t}}}\ra$}
