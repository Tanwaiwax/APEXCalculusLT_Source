\exerciseset{In Exercises}{, find $\vec T(t)$ and $\vec N(t)$.}{

\exercise{$\vrt =\bracket{4t,2\sin t,2\cos t}$}{$\vec T(t) = \frac{1}{\sqrt{5}}\bracket{2,\cos t,-\sin t}$; $\vec N(t) =\bracket{0,-\sin t,-\cos t}$}

\exercise{$\vrt =\bracket{5\cos t,3\sin t,4\sin t}$}{$\vec T(t) =\bracket{-\sin t, 3/5\cos t, 4/5\cos t}$;\\
$\vec N(t) =\bracket{-\cos t,-3/5\sin t,-4/5\sin t}$}

\exercise{$\vrt =\bracket{a\cos t,a\sin t,b t}$; \quad $a>0$}{$\vec T(t) = \frac{1}{\sqrt{a^2+b^2}}\bracket{-a\sin t,a\cos t,b}$;\\
$\vec N(t) =\bracket{-\cos t,-\sin t,0}$}

\exercise{$\vrt =\bracket{\cos (at),\sin (at),t}$}{$\vec T(t) = \frac{1}{\sqrt{a^2+1}}\bracket{-a\sin (at),a\cos (at),1}$;\\
$\vec N(t) =\bracket{-\cos t,-\sin t,0}$}

}
