\exerciseset{In Exercises}{ , find $\vec N(t)$.
}{

\exercise{$\vrt = \la 4t,2\sin t,2\cos t \ra$
}{$\vec T(t) = \frac{1}{\sqrt{5}}\la 2,\cos t,-\sin t\ra$; $\vec N(t) = \la 0,-\sin t,-\cos t\ra$
}
\exercise{$\vrt = \la 5\cos t,3\sin t,4\sin t \ra$
}{$\vec T(t) = \la -\sin t, 3/5\cos t, 4/5\cos t\ra$; $\vec N(t) = \la -\cos t,-3/5\sin t,-4/5\sin t\ra$
}
\exercise{$\vrt = \la a\cos t,a\sin t,b t \ra$; \quad $a>0$
}{$\vec T(t) = \frac{1}{\sqrt{a^2+b^2}}\la -a\sin t,a\cos t,b \ra$; $\vec N(t) = \la  -\cos t,-\sin t,0\ra$
}
\exercise{$\vrt = \la \cos (at),\sin (at),t \ra$
}{$\vec T(t) = \frac{1}{\sqrt{a^2+1}}\la -a\sin (at),a\cos (at),1 \ra$; $\vec N(t) = \la  -\cos t,-\sin t,0\ra$
}}
