\exerciseset{In Exercises}{, a position function $\vrt$ is given along with its unit tangent vector $\vec T(t)$ evaluated at $t=a$, for some value of $a$.
\begin{enumerate}
	\item [(a)] Confirm that $\vec T(a)$ is as stated.
	\item	[(b)] Using a graph of \vrt\ and \autoref{thm:concave}, find $\vec N(a)$.
\end{enumerate}
}{

\exercise{$\vrt =\bracket{3\cos t,5\sin t}$;\quad $\ds \vec T(\pi/4) =\bracket{-\frac3{\sqrt{34}}, \frac5{\sqrt{34}}}$.
}{\begin{enumerate}
	\item Be sure to show work
	\item	$\vec N(\pi/4) =\bracket{-5/\sqrt{34}, -3/\sqrt{34}}$
\end{enumerate}
}
\exercise{$\ds \vrt =\bracket{t,\frac{1}{t^2+1}}$;\quad $\ds \vec T(1) =\bracket{\frac2{\sqrt{5}}, -\frac1{\sqrt{5}}}$.
}{\begin{enumerate}
	\item Be sure to show work
	\item	$\vec N(1) =\bracket{\frac1{\sqrt{5}}, \frac2{\sqrt{5}}}$
\end{enumerate}
}
\exercise{$\ds \vrt = (1+2\sin t)\bracket{\cos t,\sin t}$;\quad $\ds \vec T(0) =\bracket{\frac2{\sqrt{5}}, \frac1{\sqrt{5}}}$.
}{\begin{enumerate}
	\item Be sure to show work
	\item	$\vec N(0) =\bracket{-\frac1{\sqrt{5}}, \frac2{\sqrt{5}}}$
\end{enumerate}
}
\exercise{$\ds \vrt =\bracket{\cos^3 t,\sin^3 t}$;\quad $\ds \vec T(\pi/4) =\bracket{-\frac1{\sqrt{2}}, \frac1{\sqrt{2}}}$.
}{\begin{enumerate}
	\item Be sure to show work
	\item	$\vec N(\pi/4) =\bracket{\frac1{\sqrt{2}}, \frac1{\sqrt{2}}}$
\end{enumerate}
}
