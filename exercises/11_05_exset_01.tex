\exerciseset{In Exercises}{ , a position function \vrt\ is given, where $t=0$ corresponds to the initial position. Find the arc length parameter $s$, and rewrite \vrt\ in terms of $s$; that is, find $\vec r(s)$. 
}{

\exercise{$\vrt = \la 2t, t, -2t\ra$
}{$s = 3t$, so $\vec r(s)  = \la 2s/3, s/3, -2s/3\ra$

}
\exercise{$\vrt = \la 7\cos t,7\sin t\ra$
}{$s = 7t$, so $\vec r(s)  = \la 7\cos (s/7), 7\sin (s/7)\ra$

}
\exercise{$\vrt = \la 3\cos t,3\sin t, 2t\ra$
}{$s = \sqrt{13}t$, so $\vec r(s)  = \la 3\cos (s/\sqrt{13}), 3\sin (s/\sqrt{13}), 2s/\sqrt{13}\ra$
}
\exercise{$\vrt = \la 5\cos t,13\sin t, 12\cos t\ra$
}{$s = 13t$, so $\vec r(s)  = \la 5\cos (s/13), 13\sin (s/13), 12\cos (s/13)\ra$
}}
