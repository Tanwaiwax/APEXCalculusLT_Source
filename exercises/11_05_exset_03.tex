\exerciseset{In Exercises}{, find the value of $x$ or $t$ where curvature is maximized.}{

\exercise{$\ds y=\frac16x^3$ }{maximized at $x=\pm \frac{\sqrt{2}}{\sqrt[4]{5}}$}

\exercise{$\ds y=\sin x$ }{maximized at $x=\dotsc -3\pi/2, -\pi/2, \pi/2, \dotsc$}

\exercise{$\vrt =\bracket{t^2+2t,3t-t^2}$ }{maximized at $t=1/4$}

\exercise{$\vrt =\bracket{t, 4/t, 3/t}$ }{maximized at $t= \pm \sqrt{5}$}

}
