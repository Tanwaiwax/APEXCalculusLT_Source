\exercisesetinstructions{, give the domain and range of the multivariable function.}

\exercise{$f(x,y) = x^2+y^2+2$}{domain: $\mathbb{R}^2$\\
range: $z\geq 2$}

\exercise{$f(x,y) = x+2y$}{domain: $\mathbb{R}^2$\\
range: $\mathbb{R}$}

\exercise{$f(x,y) = x-2y$}{domain: $\mathbb{R}^2$\\
range: $\mathbb{R}$}

\exercise{$\ds f(x,y) = \frac{1}{x+2y}$}{domain: $x\neq 2y$; in set notation, $\{(x,y)\,|\, x\neq 2y\}$\\
range: $z\neq 0$}

\exercise{$\ds f(x,y) = \frac{1}{x^2+y^2+1}$}{domain: $\mathbb{R}^2$\\
range: $0<z\leq 1$}

\exercise{$\ds f(x,y) = \sin x\cos y$}{domain: $\mathbb{R}^2$\\
range: $-1\leq z\leq 1$}
\exercise{$\ds f(x,y) = \sqrt{9-x^2-y^2}$}{domain: $\{(x,y)\,|\, x^2+y^2\leq 9\}$, i.e., the domain is the circle and interior of a circle centered at the origin with radius 3.\\
range: $0\leq z\leq 3$}
\exercise{$\ds f(x,y) = \frac1{\sqrt{x^2+y^2-9}}$}{domain: $\{(x,y)\,|\, x^2+y^2\geq 9\}$, i.e., the domain is the exterior of the circle (not including the circle itself) centered at the origin with radius 3.\\
range: $0< z< \infty$, or $(0,\infty)$}

\exercisesetend
