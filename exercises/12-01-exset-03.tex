\exercisesetinstructions{, give the domain and range of the functions of three variables.}

\exercise{$\ds f(x,y,z) = \frac{x}{x+2y-4z}$}{domain: $x+2y-4z\neq 0$; the set of points in $\mathbb{R}^3$ NOT in the domain form a plane through the origin.\\
range: $\mathbb{R}$}

\exercise{$\ds f(x,y,z) = \frac{1}{1-x^2-y^2-z^2}$}{domain: $x^2+y^2+z^2\neq 1$; the set of points in $\mathbb{R}^3$ NOT in the domain form a sphere of radius 1.\\
range: $(-\infty,0)\cup[1,\infty)$}

\exercise{$\ds f(x,y,z) = \sqrt{z-x^2+y^2}$}{domain: $z\geq x^2-y^2$; the set of points in $\mathbb{R}^3$ above (and including) the hyperbolic paraboloid $z=x^2-y^2$.\\
range: $[0,\infty)$}

\exercise{$\ds f(x,y,z) = z^2\sin x\cos y$}{domain: $\mathbb{R}^3$\\
range: $\mathbb{R}$}

\exercisesetend
