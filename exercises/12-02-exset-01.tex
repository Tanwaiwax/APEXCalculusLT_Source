\exercisesetinstructions{, a set $S$ is given.
\begin{enumerate}
\renewcommand{\theenumi}{(\alph{enumi})}
\item Give one boundary point and one interior point, when possible, of $S$. 
\item State whether $S$ is open, closed, or neither.
\item State whether $S$ is bounded or unbounded.
\end{enumerate}}

\exercise{$\ds S = \left\{(x,y)\,\left| \, \frac{(x-1)^2}{4}+\frac{(y-3)^2}{9}\leq 1\right.\right\}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
\item Answers will vary.\\
interior point: $(1,3)$\\
boundary point: $(3,3)$
\item $S$ is a closed set
\item $S$ is bounded
\end{enumext}}}

\exercise{$\ds S = \left\{(x,y)\,| \, y\neq x^2\right\}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
\item Answers will vary.\\
interior point: $(-5,28)$\\
boundary point: $(3,9)$
\item $S$ is an open set
\item $S$ is unbounded
\end{enumext}}}

\exercise{$\ds S = \left\{(x,y)\,| \,  x^2+y^2=1\right\}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
\item Answers will vary.\\
interior point: none\\
boundary point: $(0,-1)$
\item $S$ is a closed set, consisting only of boundary points
\item $S$ is bounded
\end{enumext}}}

\exercise{$\ds S=\{ (x,y) | y>\sin x\}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
\item Answers will vary.\\
Interior point: $(0,1)$\\
Boundary point: $(0,0)$
\item $S$ is a closed set, containing all of its boundary points.
\item $S$ is unbounded.
\end{enumext}}}

\exercisesetend
