\exercisesetinstructions{: 
\begin{enumerate}
\item Find the domain $D$ of the given function.
\item State whether $D$ is an open or closed set.
\item State whether $D$ is bounded or unbounded.
\end{enumerate}}

\exercise{$\ds f(x,y) = \sqrt{9-x^2-y^2}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D = \left\{(x,y)\, |\, 9-x^2-y^2\geq 0\right\}$.
\item $D$ is a closed set.
\item	$D$ is bounded.
\end{enumerate}}}

\exercise{$\ds f(x,y) = \sqrt{y-x^2}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D = \left\{(x,y)\, |\, y\geq x^2\right\}$.
\item $D$ is a closed set.
\item	$D$ is unbounded.
\end{enumerate}}}

\exercise{$\ds f(x,y) = \frac{1}{\sqrt{y-x^2}}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D = \left\{(x,y)\, |\, y> x^2\right\}$.
\item $D$ is an open set.
\item $D$ is unbounded.
\end{enumerate}}}

\exercise{$\ds f(x,y) = \frac{x^2-y^2}{x^2+y^2}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D = \left\{(x,y)\, |\, (x,y)\neq (0,0) \right\}$.
\item $D$ is an open set.
\item $D$ is unbounded.
\end{enumerate}}}

\exercise{$f(x,y)=x^2+3xy+y^2$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D=\mathbb{R}^2$.
\item $D$ is an open set.
\item $D$ is unbounded.
\end{enumerate}}}

\exercise{$f(x,y)=\dfrac{x^2+y+1}{x-2y+3}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D=\{(x, y)\,|\,x-2y+3\neq0\}$. This is all points in the plane except those on the line $y=\frac x2-\frac32$.
\item $D$ is an open set.
\item $D$ is unbounded.
\end{enumerate}}}

\exercise{$f(x,y)=\dfrac1{\sqrt{1-x^2-y^2}}$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D=\{(x,y)\,|\,x^2+y^2<1\}$. This is the open disk of radius 1 centered at the origin.
\item $D$ is an open set.
\item $D$ is bounded.
\end{enumerate}}}

\exercise{$f(x,y)=\ln(xy)$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumerate}
\item $D=\{(x,y)\,|\, x,y>0\text{ or }x,y<0\}$. This is the first and third quadrants of the $xy$-plane without the axes.
\item $D$ is an open set.
\item $D$ is unbounded.
\end{enumerate}}}
