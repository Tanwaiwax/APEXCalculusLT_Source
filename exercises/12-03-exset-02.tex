\begin{exerciseset}{In Exercises}{, find $f_x$,\ \ $f_y$,\ \  $f_{xx}$,\ \  $f_{yy}$,\ \  $f_{xy}$ and $f_{yx}$.}

\exercise{$f(x,y) = x^2y+3x^2+4y-5$}{$f_x=2xy+6x$, $f_y=x^2+4$\\
$f_{xx}=2y+6$, $f_{yy}=0$\\
$f_{xy}=2x$, $f_{yx}=2x$}

\exercise{$f(x,y) = y^3+3xy^2+3x^2y+x^3$}{$f_x=3x^2+6xy+3y^2$, $f_y=3x^2+6xy+3y^2$\\
$f_{xx}=6x+6y$, $f_{yy}=6x+6y$\\
$f_{xy}=6x+6y$, $f_{yx}=6x+6y$}

\exercise{$\ds f(x,y) = \frac xy$}{$f_x=1/y$, $f_y=-x/y^2$\\
$f_{xx}=0$, $f_{yy}=2x/y^3$\\
$f_{xy}=-1/y^2$, $f_{yx}=-1/y^2$}

\exercise{$\ds f(x,y) = \frac 4{xy}$}{$f_x=-4/(x^2y)$, $f_y=-4/(xy^2)$\\
$f_{xx}=8/(x^3y)$, $f_{yy}=8/(xy^3)$\\
$f_{xy}=4/(x^2y^2)$, $f_{yx}=4/(x^2y^2)$}

\exercise{$\ds f(x,y) = e^{x^2+y^2}$}{$f_x=2xe^{x^2+y^2}$, $f_y=2ye^{x^2+y^2}$\\
$f_{xx}=2e^{x^2+y^2}+4x^2e^{x^2+y^2}$, $f_{yy}=2e^{x^2+y^2}+4y^2e^{x^2+y^2}$\\
$f_{xy}=4xye^{x^2+y^2}$, $f_{yx}=4xye^{x^2+y^2}$}

\exercise{$\ds f(x,y) = e^{x+2y}$}{$f_x=e^{x+2y}$, $f_y=2e^{x+2y}$\\
$f_{xx}=e^{x+2y}$, $f_{yy}=4e^{x+2y}$\\
$f_{xy}=2e^{x+2y}$, $f_{yx}=2e^{x+2y}$}

\exercise{$\ds f(x,y) = \sin x\cos y$}{$f_x=\cos x\cos y$, $f_y=-\sin x\sin y$\\
$f_{xx}=-\sin x\cos y$, $f_{yy}=-\sin x\cos y$\\
$f_{xy}=-\sin y\cos x$, $f_{yx}=-\sin y\cos x$}

\exercise{$\ds f(x,y) = (x+y)^3$}{$f_x=3(x+y)^2$, $f_y=3(x+y)^2$\\
$f_{xx}=6(x+y)$, $f_{yy}=6(x+y)$\\
$f_{xy}=6(x+y)$, $f_{yx}=6(x+y)$}

\exercise{$f(x,y) = \cos(5xy^3)$}{$f_x=-5y^3\sin(5xy^3)$, $f_y=-15xy^2\sin(5xy^3)$\\
$f_{xx}=-25y^6\cos(5xy^3)$, $f_{yy}=-225x^2y^4\cos(5xy^3)-30xy\sin(5xy^3)$\\
$f_{xy}=-75xy^5\cos(5xy^3)-15y^2\sin(5xy^3)$, $f_{yx}=-75xy^5\cos(5xy^3)-15y^2\sin(5xy^3)$}

\exercise{$f(x,y) = \sin(5x^2+2y^3)$}{$f_x=10x\cos(5x^2+2y^3)$, $f_y=6y^2\cos(5x^2+2y^3)$\\
$f_{xx}=10\cos(5x^2+2y^3)-100x^2\sin(5x^2+2y^3)$, $f_{yy}=12y\cos(5x^2+2y^3)-36y^4\sin(5x^2+2y^3)$\\
$f_{xy}=-60xy^2\sin(5x^2+2y^3)$, $f_{yx}=-60xy^2\sin(5x^2+2y^3)$}

\exercise{$\ds f(x,y) = \sqrt{4xy^2+1}$}{$f_x=\frac{2y^2}{\sqrt{4xy^2+1}}$, $f_y=\frac{4xy}{\sqrt{4xy^2+1}}$\\
$f_{xx}=-\frac{4y^4}{\sqrt{4xy^2+1}^3}$, $f_{yy}=-\frac{16x^2y^2}{\sqrt{4xy^2+1}^3}+\frac{4x}{\sqrt{4xy^2+1}}$\\
$f_{xy}=-\frac{8xy^3}{\sqrt{4xy^2+1}^3}+\frac{4y}{\sqrt{4xy^2+1}}$, $f_{yx}=-\frac{8xy^3}{\sqrt{4xy^2+1}^3}+\frac{4y}{\sqrt{4xy^2+1}}$}

\exercise{$\ds f(x,y) =(2x+5y)\sqrt{y}$}{$f_x=2\sqrt{y}$, $f_y=5\sqrt{y}+\frac{2x+5y}{2\sqrt{y}}$\\
$f_{xx}=0$, $f_{yy}=\frac{5}{\sqrt{y}}-\frac{2x+5y}{4y^{3/2}}$\\
$f_{xy}=\frac{1}{\sqrt{y}}$, $f_{yx}=\frac{1}{\sqrt{y}}$}

\exercise{$\ds f(x,y) =\frac{1}{x^2+y^2+1}$}{$f_x=-\frac{2x}{(x^2+y^2+1)^2}$, $f_y=-\frac{2y}{(x^2+y^2+1)^2}$\\
$f_{xx}=\frac{8x^2}{(x^2+y^2+1)^3}-\frac{2}{(x^2+y^2+1)^2}$, $f_{yy}=\frac{8y^2}{(x^2+y^2+1)^3}-\frac{2}{(x^2+y^2+1)^2}$\\
$f_{xy}=\frac{8xy}{(x^2+y^2+1)^3}$, $f_{yx}=\frac{8xy}{(x^2+y^2+1)^3}$}

\exercise{$\ds f(x,y) =5x-17y$}{$f_x=5$, $f_y=-17$\\
$f_{xx}=0$, $f_{yy}=0$\\
$f_{xy}=0$, $f_{yx}=0$}

\exercise{$\ds f(x,y) =3x^2+1$}{$f_x=6x$, $f_y=0$\\
$f_{xx}=6$, $f_{yy}=0$\\
$f_{xy}=0$, $f_{yx}=0$}

\exercise{$\ds f(x,y) =\ln(x^2+y)$}{$f_x=\frac{2x}{(x^2+y)}$, $f_y=\frac{1}{(x^2+y)}$\\
$f_{xx}=-\frac{4x^2}{(x^2+y)^2}+\frac{2}{(x^2+y)}$, $f_{yy}=-\frac{1}{(x^2+y)^2}$\\
$f_{xy}=-\frac{2x}{(x^2+y)^2}$, $f_{yx}=-\frac{2x}{(x^2+y)^2}$}

\exercise{$\ds f(x,y) =\frac{\ln x}{4y}$}{$f_x=\frac{1}{4xy}$, $f_y=-\frac{\ln x}{4y^2}$\\
$f_{xx}=-\frac{1}{4x^2y}$, $f_{yy}=\frac{\ln x}{2y^3}$\\
$f_{xy}=-\frac{1}{4xy^2}$, $f_{yx}=-\frac{1}{4xy^2}$}

\exercise{$\ds f(x,y) =5e^x\sin y+9$}{$f_x=5e^x\sin y$, $f_y=5e^x\cos y$\\
$f_{xx}=5e^x\sin y$, $f_{yy}=-5e^x\sin y$\\
$f_{xy}=5e^x\cos y$, $f_{yx}=5e^x\cos y$}

\end{exerciseset}
