\begin{exerciseset}{In Exercises}{, a function $z=f(x,y)$ is given. Give the indicated approximation using the total differential.}

\exercise{$f(x,y) = \sqrt{x^2+y}$. Approximate $f(2.95,7.1)$ knowing $f(3,7) = 4$. }{$\dd z = \frac{x}{\sqrt{x^2+y}}\dd x + \frac{1}{2\sqrt{x^2+y}}\dd y$, with $\dd x = -0.05$ and $\dd y = .1$. At $(3,7)$, $\dd z = 3/4(-0.05) + 1/8(.1) = -0.025$, so $f(2.95,7.1) \approx -0.025+4 = 3.975$.}

\exercise{$f(x,y) = \sin x\cos y$. Approximate $f(0.1,-0.1)$ knowing $f(0,0) = 0$. }{$\dd z = (\cos x\cos y)\dd x - (\sin x\sin y)\dd y$, with $\dd x = 0.1$ and $\dd y = -0.1$. At $(0,0)$, $\dd z = 1(.1) - (0)(-0.1) = 0.1$, so $f(0.1,-0.1) \approx 0.1+0 = 0.1$.}

\exercise{$f(x,y) = x^2y-xy^2$. Approximate $f(2.04,3.06)$ knowing $f(2,3) = -6$. }{$\dd z = (2xy-y^2)\dd x + (x^2-2xy)\dd y$, with $\dd x = 0.04$ and $\dd y = 0.06$. At $(2,3)$, $\dd z = 3(0.04) + (-8)(0.06) = -0.36$, so $f(2.04,3.06) \approx -0.36-6 = -6.36$.}

\exercise{$f(x,y) = \ln(x-y)$. Approximate $f(5.1,3.98)$ knowing $f(5,4) = 0$. }{$\dd z = \frac1{x-y}\dd x -\frac{1}{x-y}\dd y$, with $\dd x = 0.1$ and $\dd y = -0.02$. At $(5,4)$, $\dd z = 1(0.1) + (-1)(-0.02) = 0.12$, so $f(5.1,3.98) \approx 0.12+0 = 0.12$.}

\end{exerciseset}
