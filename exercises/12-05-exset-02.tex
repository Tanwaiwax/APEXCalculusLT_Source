\exercisesetinstructions{, a function $z=f(x,y)$ and a point $P$ are given. Find the directional derivative of $f$ in the indicated directions. Note: these are the same functions as in Exercises~\ref{12_05_ex_07}--\ref{12_05_ex_22}.}

\exercise{$f(x,y) = -x^2y+xy^2+xy$, $P= (2,1)$ \label{12_05_ex_11}
\begin{enumext}[start=1]
	\item In the direction of $\vec v =\bracket{3,4}$
	\item In the direction toward the point $Q = (1,-1)$.
\end{enumext}
}{$\nabla f =\bracket{-2xy+y^2+y, -x^2+2xy+x}$; $\nabla f(2,1) =\bracket{-2,2}$.
Be sure to change all directions to unit vectors.
\begin{enumext}[start=1]
	\item $2/5$ ($\vec u =\bracket{3/5,4/5}$)
	\item	$-2/\sqrt{5}$ ($\vec u =\bracket{-1/\sqrt{5},-2/\sqrt{5}}$)
\end{enumext}}

\exercise{$\ds f(x,y) = \sin x\cos y$, $\ds P = \left(\frac{\pi}{4},\frac{\pi}{3}\right)$
\begin{enumext}[start=1]
	\item In the direction of $\vec v=\bracket{1,1}$.
	\item	In the direction toward the point $Q = (0,0)$.
\end{enumext}
}{$\nabla f =\bracket{\cos x\cos y, -\sin x\sin y}$;\\
$\nabla f(\frac{\pi}{4},\frac{\pi}{3}) =\bracket{\frac{1}{2\sqrt{2}},-\frac12\sqrt{\frac32}}$.\\
Be sure to change all directions to unit vectors. 
\begin{enumext}[start=1]
	\item $\frac14(1-\sqrt{3})$ ($\vec u =\bracket{1/\sqrt{2},1/\sqrt{2}}$)
	\item	$\frac{4\sqrt{3}-1}{10\sqrt{2}}$ ($\vec u =\bracket{-3/5,-4/5}$)
\end{enumext}}

\exercise{$\ds f(x,y) = \frac{1}{x^2+y^2+1}$, $P = (1,1)$.
\begin{enumext}[start=1]
	\item In the direction of $\vec v =\bracket{1,-1}$.
	\item In the direction toward the point $Q = (-2,-2)$.
\end{enumext}%
}{$\nabla f =\bracket{\frac{-2x}{(x^2+y^2+1)^2}, \frac{-2y}{(x^2+y^2+1)^2}}$;\\
$\nabla f(1,1) =\bracket{-2/9,-2/9}$.\\
Be sure to change all directions to unit vectors.
\begin{enumext}[start=1]
	\item 0 ($\vec u =\bracket{1/\sqrt{2},-1/\sqrt{2}}$)
	\item	$2\sqrt{2}/9$ ($\vec u =\bracket{-1/\sqrt{2},-1/\sqrt{2}}$)
\end{enumext}}

\exercise{$\ds f(x,y) = -4x+3y$, $P = (5,2)$\label{12_05_ex_14}

	\item In the direction of $\vec v =\bracket{3,1}.$
	\item In the direction toward the point $Q = (2,7)$.
\end{enumext}
}{$\nabla f =\bracket{-4,3}$; $\nabla f(5,2) =\bracket{-4,3}$. Be sure to change all directions into unit vectors.
\begin{enumext}[start=1]

	\item	$27/\sqrt{34}$ ($\vec u =\bracket{-3/\sqrt{34},5/\sqrt{34}}$)
\end{enumext}}

\exercise{$\ds f(x,y) = x^2y^3-2x$, $P = (1,1)$\label{12_05_ex_23}
\begin{enumext}[start=1]
	\item In the direction of $\vec v =\bracket{3,3}$
	\item	In the direction toward the point $Q = (1,2)$.

	\begin{enumext}[start=1]
	\item	$3/\sqrt{2}$; ($\vec u =\bracket{1/\sqrt{2},1/\sqrt{2}}$)
	\item	$3$
\end{enumext}}
