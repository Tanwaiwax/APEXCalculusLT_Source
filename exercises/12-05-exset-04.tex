\exercisesetinstructions{, a function $w=F(x,y,z)$, a vector $\vec v$ and a point $P$ are given. 
\begin{enumext}
	\item Find $\nabla F(x,y,z)$.
	\item Find $D_{\vec u\,}F$ at $P$, where $\vec u$ is the unit vector in the direction of $\vec v$.
\end{enumext}}

\exercise{$\ds F(x,y,z) = 3x^2z^3+4xy-3z^2$, $\vec v =\bracket{1,1,1}$, $P = (3,2,1)$}{\begin{enumext}
	\item $\nabla F(x,y,z) =\bracket{6xz^3+4y, 4x, 9x^2z^2-6z}$
	\item	$113/\sqrt{3}$
\end{enumext}}

\exercise{$\ds F(x,y,z) = \sin(x)\cos(y)e^z$, $\vec v =\bracket{2,2,1}$, $P = (0,0,0)$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\nabla F(x,y,z)=\\\bracket{\cos x\cos ye^z, -\sin x\sin ye^z, \sin x\cos ye^z}$
	\item	$2/3$
\end{enumext}}}

\exercise{$\ds F(x,y,z) = x^2y^2-y^2z^2$, $\vec v =\bracket{-1,7,3}$, $P = (1,0,-1)$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\nabla F(x,y,z) =\bracket{2xy^2, 2y(x^2-z^2), -2y^2z}$
	\item	$0$
\end{enumext}}}

\exercise{$\ds F(x,y,z) = \frac{2}{x^2+y^2+z^2}$, $\vec v =\bracket{1,1,-2}$, $P = (1,1,1)$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\nabla F(x,y,z) =\\\bracket{-\frac{4x}{(x^2+y^2+z^2)^2},-\frac{4y}{(x^2+y^2+z^2)^2},-\frac{4z}{(x^2+y^2+z^2)^2}}$
	\item	$0$
\end{enumext}}}

\exercisesetend
