\exercisesetinstructions{, functions $z=f(x,y)$, $x=g(t)$ and $y=h(t)$ are given.
\begin{enumerate}
	\renewcommand{\theenumi}{(\alph{enumi})}
	\item Use the Multivariable Chain Rule to compute $\ds \frac{\dd z}{\dd t}$.
	\item Evaluate $\ds \frac{\dd z}{\dd t}$ at the indicated $t$-value.
\end{enumerate}}

\exercise{$z=3x+4y$,\qquad $x=t^2$,\qquad $y=2t$;\qquad $t=1$\label{12_08_ex_07}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
	\item $\frac{\dd z}{\dd t} = 3(2t)+4(2) = 6t+8$.
	\item At $t=1$, $\frac{\dd z}{\dd t} = 14$.
\end{enumext}}}

\exercise{$\ds z=x^2-y^2$,\qquad $x=t$,\qquad $y=t^2-1$;\qquad $t=1$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
	\item $\frac{\dd z}{\dd t} = 2x(1)-2y(2t) = 2x-4yt$
	\item At $t=1$, $x=1$, $y=0$ and $\frac{\dd z}{\dd t} = 2$.
\end{enumext}}}

\exercise{$\ds z=5x+2y$,\qquad $x=2\cos t+1$,\qquad $y=\sin t-3$;\qquad $t=\pi/4$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
	\item $\frac{\dd z}{\dd t} = 5(-2\sin t)+2(\cos t) = -10\sin t+2\cos t$
	\item At $t=\pi/4$, $\frac{\dd z}{\dd t} = -4\sqrt{2}$.
\end{enumext}}}

\exercise{$\ds z=\frac{x}{y^2+1}$,\qquad $x=\cos t$,\qquad $y=\sin t$;\qquad $t=\pi/2$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
	\item $\ds\frac{\dd z}{\dd t} = \frac{1}{1+y^2}(-\sin t)-\frac{2xy}{(y^2+1)^2}(\cos t)$.
	\item At $t=\pi/2$, $x=0$, $y=1$, and $\frac{\dd z}{\dd t} = -1/2$.
\end{enumext}}}

\exercise{$\ds z=x^2+2y^2$,\qquad $x=\sin t$,\qquad $y=3\sin t$;\qquad $t=\pi/4$\label{12_08_ex_10}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
	\item $\ds\frac{\dd z}{\dd t} = 2x(\cos t) + 4y(3\cos t)$.
	\item At $t=\pi/4$, $x=\sqrt{2}/2$, $y=3\sqrt{2}/2$, and $\frac{\dd z}{\dd t} = 19$.
\end{enumext}}}

\exercise{$z=\cos x\sin y$,\qquad $x=\pi t$,\qquad $y=2\pi t+\pi/2$;\qquad $t=3$\label{12_08_ex_08}}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}[start=1]
	\item $\frac{\dd z}{\dd t} = -\sin x\sin y(\pi) + \cos x\cos y(2\pi)$.
	\item At $t=3$, $x=3\pi$, $y=13\pi/2$, and $\frac{\dd z}{\dd t} = 0$.
\end{enumext}}}
