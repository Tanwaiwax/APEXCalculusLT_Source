\exercisesetinstructions{,  functions $z=f(x,y)$, $x=g(s,t)$ and $y=h(s,t)$ are given. 
\begin{enumext}
	\item Use the Multivariable Chain Rule to compute $\ds \frac{\partial z}{\partial s}$ and $\ds \frac{\partial z}{\partial t}$.
	\item Evaluate $\ds \frac{\partial z}{\partial s}$ and $\ds \frac{\partial z}{\partial t}$ at the indicated $s$ and $t$ values.
\end{enumext}}

\exercise{$\ds z=x^2y$,\qquad $x=s-t$,\qquad $y=2s+4t$;\qquad $s=1$, $t=0$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\frac{\partial z}{\partial s} = 2xy(1) + x^2(2) = 2xy+2x^2$;\\
	$\frac{\partial z}{\partial t} = 2xy(-1) + x^2(4) = -2xy+4x^2$
	\item		With $s=1$, $t=0$, $x=1$ and $y=2$. Thus $\frac{\partial z}{\partial s} = 6$ and $\frac{\partial z}{\partial t} = 0$
\end{enumext}}}

\exercise{$\ds z=\cos\big(\pi x+\frac{\pi}2y\big)$,\quad $x=st^2$,\quad $y=s^2t$;\quad $s=1$, $t=1$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\frac{\partial z}{\partial s} = -\pi\sin(\pi x+\pi y/2)(t^2) -\frac12\pi\sin(\pi x+\pi y/2)(2st) = -\pi\Big(t^2\sin(\pi x+ \pi y/2) + st\sin(\pi x+\pi y/2)\Big)$;\\
	$\frac{\partial z}{\partial t} = -\pi\sin(\pi x+\pi y/2)(2st) -\frac12\pi\sin(\pi x+\pi y/2)(s^2)= -\pi\Big(2st\sin(\pi x+\pi y/2) + \frac12s^2\sin(\pi x+\pi y/2)\Big)$
	\item		With $s=1$, $t=1$, $x=1$ and $y=1$. Thus $\frac{\partial z}{\partial s} = 2\pi$ and $\frac{\partial z}{\partial t} = 5\pi/2$
\end{enumext}}}

\exercise{$\ds z=x^2+y^2$,\quad $x=s\cos t$,\quad $y=s\sin t$;\quad $s=2$, $t=\pi/4$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
 \item $\frac{\partial z}{\partial s} = 2x(\cos t) + 2y(\sin t) = 2x\cos t+2y\sin t$;\\
	$\frac{\partial z}{\partial t} = 2x(-s\sin t)+2y(s\cos t) = -2xs\sin t+ 2ys\cos t$
	\item		With $s=2$, $t=\pi/4$, $x=\sqrt{2}$ and $y=\sqrt{2}$. Thus $\frac{\partial z}{\partial s} = 4$ and $\frac{\partial z}{\partial t} = 0$
\end{enumext}}}

\exercise{$\ds z=e^{-(x^2+y^2)}$,\quad $x=t$,\quad $y=st^2$;\quad $s=1$, $t=1$}{\mbox{}\\[-2\baselineskip]\parbox[t]{\linewidth}{\begin{enumext}
	\item $\frac{\partial z}{\partial s} = -2xe^{-(x^2+y^2)}(0) -2ye^{-(x^2+y^2)}(t^2) = -2yt^2e^{-(x^2+y^2)}$;\\
	$\frac{\partial z}{\partial t} = -2xe^{-(x^2+y^2)}(1)-2ye^{-(x^2+y^2)}(2st) = -2xe^{-(x^2+y^2)}-4s t y e^{-(x^2+y^2)}$
	\item		With $s=1$, $t=1$, $x=1$ and $y=1$. Thus $\frac{\partial z}{\partial s} = -2/e^2$ and $\frac{\partial z}{\partial t} = -6/e^2$
\end{enumext}}}

\exercisesetend
