\begin{exerciseset}{In Exercises}{, find the total differential $dz$.}

\exercise{$z = x\sin y + x^2$}{$dz = (\sin y + 2x)dx + (x\cos y)dy$}

\exercise{$z = (2x^2+3y)^2$}{$dz = 8x(2x^2+3y)dx + 6(2x^2+3y)dy$}

\exercise{$z = 5x-7y$}{$dz = 5dx -7dy$}

\exercise{$z = xe^{x+y}$}{$dz = (e^{x+y}+xe^{x+y})dx +xe^{x+y}dy$}

\end{exerciseset}
