\exerciseset{In Exercises}{, a function $z=f(x,y)$ is given. Find $\nabla f$.}{

\exercise{$f(x,y) = -x^2y+xy^2+xy$\label{12_05_ex_07}
}{$\nabla f = \la -2xy+y^2+y, -x^2+2xy+x\ra$
}
\exercise{$\ds f(x,y) = \sin x\cos y$
}{$\nabla f = \la \cos x\cos y, -\sin x\sin y\ra$
}
\exercise{$\ds f(x,y) = \frac{1}{x^2+y^2+1}$
}{$\nabla f = \la \frac{-2x}{(x^2+y^2+1)^2}, \frac{-2y}{(x^2+y^2+1)^2}\ra$
}
\exercise{$\ds f(x,y) = -4x+3y$\label{12_05_ex_10}
}{$\nabla f = \la -4,3\ra$
}
\exercise{$\ds f(x,y) = x^2+2y^2-xy-7x$
}{$\nabla f = \la 2x-y-7,4y-x\ra$
}
\exercise{$\ds f(x,y) = x^2y^3-2x$\label{12_05_ex_22}
}{$\nabla f = \la 2xy^3-2,3x^2y^2\ra$
}}
