\exerciseset{In Exercises}{, a function $z=f(x,y)$ and a point $P$ are given. 
\begin{enumerate}
	\item [(a)] Find the direction of maximal increase of $f$ at $P$.
	\item [(b)] What is the maximal value of $D_{\vec u\,}f$ at $P$?
	\item [(c)] Find the direction of minimal increase of $f$ at $P$.
	\item	[(d)] Give a direction $\vec u$ such that $D_{\vec u\,}f=0$ at $P$.
\end{enumerate}
Note: these are the same functions and points as in Exercises~\ref{12_05_ex_11}--\ref{12_05_ex_23}.}{

\exercise{$f(x,y) = -x^2y+xy^2+xy$, $P= (2,1)$}{$\nabla f =\bracket{-2xy+y^2+y, -x^2+2xy+x}$
\begin{enumerate}
	\item $\nabla f(2,1) =\bracket{-2,2}$
	\item	$\norm{\nabla f(2,1)} = \norm{\bracket{-2,2}} = \sqrt{8}$
	\item	$\bracket{2,-2}$
	\item	$\bracket{1/\sqrt{2},1/\sqrt{2}}$
\end{enumerate}}

\exercise{$\ds f(x,y) = \sin x\cos y$, $\ds P = \left(\frac{\pi}{4},\frac{\pi}{3}\right)$}{$\nabla f =\bracket{\cos x\cos y, -\sin x\sin y}$
\begin{enumerate}
	\item $\nabla f(\frac{\pi}{4},\frac{\pi}{3}) =\bracket{\frac{1}{2\sqrt{2}},-\frac12\sqrt{\frac32}}$
	\item	$\norm{\nabla f(\frac{\pi}{4},\frac{\pi}{3})} = \norm{\bracket{\frac{1}{2\sqrt{2}},-\frac12\sqrt{\frac32}}} = 1/\sqrt{2} $
	\item $\bracket{-\frac{1}{2\sqrt{2}},\frac12\sqrt{\frac32}}$
	\item $\bracket{\frac12\sqrt{\frac32},\frac{1}{2\sqrt{2}}}$
\end{enumerate}}

\exercise{$\ds f(x,y) = \frac{1}{x^2+y^2+1}$, $P = (1,1)$.}{$\nabla f =\bracket{\frac{-2x}{(x^2+y^2+1)^2}, \frac{-2y}{(x^2+y^2+1)^2}}$
\begin{enumerate}
	\item  $\nabla f(1,1) =\bracket{-2/9,-2/9}$.
	\item		$\norm{\nabla f(1,1)} = \norm{\bracket{-2/9,-2/9}}=2\sqrt{2}/9$
	\item		$\bracket{2/9,2/9}$
	\item		$\bracket{1/\sqrt{2},-1/\sqrt{2}}$
\end{enumerate}}

\exercise{$\ds f(x,y) = -4x+3y$, $P = (5,4)$.}{$\nabla f =\bracket{-4,3}$
\begin{enumerate}
	\item  $\nabla f(5,4) =\bracket{-4,3}$.
	\item		$\norm{\nabla f(5,4)} = \norm{\bracket{-4,3}}=5$
	\item		$\bracket{4,-3}$
	\item		$\bracket{3/5,4/5}$
\end{enumerate}}

\exercise{$\ds f(x,y) = x^2+2y^2-xy-7x$, $P = (4,1)$}{$\nabla f =\bracket{2x-y-7,4y-x}$
\begin{enumerate}
	\item $\nabla f(4,1) =\bracket{0,0}$
	\item	0
	\item	$\bracket{0,0}$
	\item	All directions give a directional derivative of 0.
\end{enumerate}}

\exercise{$\ds f(x,y) = x^2y^3-2x$, $P = (1,1)$}{$\nabla f =\bracket{2xy^3-2,3x^2y^2}$
\begin{enumerate}
	\item $\nabla f(1,1) =\bracket{0,3}$
	\item	3
	\item	$\bracket{0,-3}$
	\item	$\vec u =\bracket{1,0}$
\end{enumerate}}

}
