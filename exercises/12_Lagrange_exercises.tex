\printproblems

\exercise{Find the constrained maxima and minima of $f(x,y) = 2x+y$ given that $x^2 + y^2 =4$.}{$\pm2\sqrt5$ at $(\pm4/\sqrt5,\pm2/\sqrt5)$}

\exercise{Find the constrained maxima and minima of $f(x,y) = xy$ given that $x^2 + 3y^2 =6$.}{min.\ $-\sqrt3$ at $(\pm\sqrt3,\mp1)$; max.\ $\sqrt3$ at $(\pm\sqrt3,\pm1)$}

\exercise{Find the points on the circle $x^2 + y^2 = 100$ which are closest to and farthest from the point $(2,3)$.}{$(\pm20/\sqrt{13},\pm30/\sqrt{13})$}

\exercise{Find the constrained maxima and minima of $f(x,y,z) = x+y^2 +2z$ given that $4x^2 + 9y^2 -36z^2 =36$.}{min.\ $(-9/\sqrt5,0,2/\sqrt5)$; max.\ $(9/8,\sqrt{59}/4,-1/4)$}

% from here are adapted from Whitman

\exercise{Find the maximum volume of a rectangular box inscribed in the unit sphere.}{$(2/\sqrt3)^3=8/3\sqrt3$}

\exercise{Find the minimum surface area of a box that holds 2 cubic meters.}{$6\sqrt[3]{4}$ square meters.}

\exercise{The \emph{girth} of a box is the perimeter of a cross section perpendicular to its length.  The US post office will accept packages whose combined length and girth are at most 130 inches.  Find the dimensions of the largest volume box that will be accepted.}{Length 130/3, height and width 65/3.}

\exercise{Using Lagrange multipliers, find the shortest distance from the point $(x_0,y_0,z_0)$ to the plane $ax+by+cz=d$.  (See also \autoref{idea:planes_dist}.)}{$\abs{ax_0+by_0+cz_0-d}/\sqrt{a^2+b^2+c^2}$}

\exercise{Find all points on the surface $xz-y^2+1=0$ that are closest to the origin.}{$(0,\pm1,0)$}

%6. The material for the bottom of an aquarium costs half as much as the high strength glass for the four sides. Find the shape of the cheapest aquarium that holds a given volume V . => \sqrt[3]{4V}\times\sqrt[3]{4V}\times\sqrt[3]{V/16}

\exercise{Find the three positive numbers whose sum is 60 and whose product is as large as possible.}{$20^3=8000$}

\exercise{Find all points on the plane $x+y+z=5$ in the first octant at which $f(x,y,z)=x^2yz^2$ has a maximum value.}{$(2,1,2)$}

\exercise{Find the points on the surface $z^2-xy=5$ that are closest to the origin.}{$(0,0,\pm\sqrt5)$}

\exercise{Find the maximum and minimum points of $f(x,y)=xy+\sqrt{9-x^2-y^2}$ when $x^2+y^2\le9$.}{Max: 5 at $\pm(2,2)$, min: $-9/2$ at $\pm(3/\sqrt2,-3/\sqrt2)$.}

\exercise{Find three real numbers whose sum is 12 and the sum of whose squares is a small as possible.}{$3\cdot4^2=48$.}

\exercise{The plane $x+y-z=1$ intersects the cylinder $x^2+y^2=1$ in an ellipse. Find the points on the ellipse closest to and farthest from the origin.}{Closest: $(1,0,0)$ and $(0,1,0)$.\\
Furthest: $(-1/\sqrt2,-1/\sqrt2,-1-\sqrt2)$.}
