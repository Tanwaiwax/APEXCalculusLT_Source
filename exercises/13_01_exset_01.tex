\exerciseset{In Exercises}{, evaluate the integral and subsequent iterated integral.}{

\exercise{\begin{enumerate}
	\item $\ds \int_2^{5} \big(6x^2+4xy-3y^2\big)\ dy$\\
	\item $\ds \int_{-3}^{-2} \int_2^{5} \big(6x^2+4xy-3y^2\big)\ dy\ dx$
\end{enumerate}
}{\begin{enumerate}
	\item $18x^2+42x-117$
	\item	$-108$
\end{enumerate}
}

\exercise{\begin{enumerate}
	\item $\ds \int_0^{\pi} \big(2x\cos y+ \sin x\big)\ dx$\\
	\item $\ds \int_{0}^{\pi/2} \int_0^{\pi} \big(2x\cos y+ \sin x\big)\ dx\ dy$
\end{enumerate}
}{\begin{enumerate}
	\item $2+\pi^2\cos y$
	\item	$\pi^2+\pi$
\end{enumerate}
}

\exercise{\begin{enumerate}
	\item $\ds \int_1^x \big(x^2y - y+2\big)\ dy$\\
	\item $\ds \int_0^2\int_1^x \big(x^2y - y+2\big)\ dy\ dx$
\end{enumerate}
}{\begin{enumerate}
	\item $x^4/2-x^2+2x-3/2$
	\item	$23/15$
\end{enumerate}
}

\exercise{\begin{enumerate}
	\item $\ds \int_y^{y^2} \big(x-y\big)\ dx$\\
	\item $\ds \int_{-1}^1\int_y^{y^2} \big(x-y\big)\ dx\ dy$
\end{enumerate}
}{\begin{enumerate}
	\item $y^4/2-y^3+y^2/2$
	\item	$8/15$
\end{enumerate}
}

\exercise{\begin{enumerate}
	\item $\ds \int_0^{y} \big(\cos x\sin y\big)\ dx$\\
	\item $\ds \int_0^\pi \int_0^{y} \big(\cos x\sin y\big)\ dx\ dy$
\end{enumerate}
}{\begin{enumerate}
	\item $\sin^2y$
	\item	$\pi/2$
\end{enumerate}
}

\exercise{\begin{enumerate}
	\item $\ds \int_0^{x} \left(\frac{1}{1+x^2}\right)\ dy$\\
	\item $\ds \int_1^2 \int_0^{x} \left(\frac{1}{1+x^2}\right)\ dy\ dx$
\end{enumerate}
}{\begin{enumerate}
	\item $x/(1+x^2)$
	\item	$\frac12\ln\left(\frac52\right)$
\end{enumerate}
}
}
