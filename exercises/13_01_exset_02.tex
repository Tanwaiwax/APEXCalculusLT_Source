\begin{exerciseset}{In Exercises}{, a graph of a planar region $R$ is given. Give the iterated integrals, with both orders of integration $dy\ dx$ and $dx\ dy$, that give the area of $R$. Evaluate one of the iterated integrals to find the area.}

\begin{tikzpicture}
\begin{axis}[width=1.16\marginparwidth,tick label style={font=\scriptsize},

\draw [very thick,draw={\colorone}] (axis cs:1,-2) -- (axis cs: 4,-2)-- (axis cs: 4,1) -- (axis cs: 1,1) -- cycle;

\draw (axis cs: 2.5,-1) node {$R$};

\end{axis}

\node [right] at (myplot.right of origin) {\scriptsize $x$};
\node [above] at (myplot.above origin) {\scriptsize $y$};
\end{tikzpicture}
\end{minipage}
}{$\ds \int_1^4\int_{-2}^1\ dy\ dx$ and $\ds\int_{-2}^1\int_1^4\ dx\ dy$.\\
area  of $R = 9\text{ units}^2$
}
