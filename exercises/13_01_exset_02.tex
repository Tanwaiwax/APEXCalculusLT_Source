\exerciseset{In Exercises}{, a graph of a planar region $R$ is given. Give the iterated integrals, with both orders of integration $dy\ dx$ and $dx\ dy$, that give the area of $R$. Evaluate one of the iterated integrals to find the area.}{

\exercise{\noindent \begin{minipage}{\linewidth}
\myincludegraphics{figures/fig13_01_ex_11}
\end{minipage}
}{$\ds \int_1^4\int_{-2}^1\ dy\ dx$ and $\ds\int_{-2}^1\int_1^4\ dx\ dy$.\\
area  of $R = 9u^2$
}

\exercise{\noindent \begin{minipage}{\linewidth}
\myincludegraphics{figures/fig13_01_ex_12}
\end{minipage}
}{$\ds\int_1^4\int_{1}^{\frac23x+\frac13}\ dy\ dx$ and $\ds\int_{1}^3\int_{\frac32y-\frac12}^4\ dx\ dy$.\\
area  of $R = 3u^2$
}

\exercise{\noindent \begin{minipage}{\linewidth}
\myincludegraphics{figures/fig13_01_ex_13}
\end{minipage}
}{$\ds\int_2^4\int_{x-1}^{7-x}\ dy\ dx$. The order $dx\ dy$ needs two iterated integrals as $x$ is bounded above by two different functions. This gives: 
$$\int_{1}^3\int_{2}^{y+1}\ dx\ dy+\int_{3}^5\int_{2}^{7-y}\ dx\ dy.$$\\
area  of $R = 4u^2$
}

}
