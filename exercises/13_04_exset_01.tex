\begin{exerciseset}{In Exercises}{, point masses are given along a line or in the plane. Find the center of mass $\overline{x}$ or $(\overline{x},\overline{y})$, as appropriate. (All masses are in grams and distances are in cm.)}

\exercise{$m_1=4$  at $x=1$; \quad $m_2=3$  at $x=3$; \quad $m_3=5$  at $x=10$}{$\overline{x}=5.25$}

\exercise{$m_1=2$  at $x=-3$; \quad $m_2=2$  at $x=-1$;\par $m_3=3$  at $x=0$;\quad $m_4=3$ at $x=7$}{$\overline{x}=1.3$}

\exercise{$m_1=2$  at $(-2,-2)$; \quad $m_2=2$  at $(2,-2)$;\par $m_3=20$  at $(0,4)$}{$(\overline{x},\overline{y}) = (0,3)$}

\exercise{$m_1=1$  at $(-1,-1)$; \quad $m_2=2$  at $(-1,1)$;\par $m_3=2$  at $(1,1)$;\quad $m_4=1$ at $(1,-1)$}{$(\overline{x},\overline{y}) = (0,1/3)$}

\end{exerciseset}
