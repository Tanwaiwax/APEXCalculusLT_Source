\exercise{$R$ is the rectangle with corners $(1,-3)$, $(1,2)$, $(7,2)$ and $(7,-3)$; $\delta(x,y) = 5$g/cm$^2$
\label{13_04_ex_11}}{$M = 150$g;  %$M_y=600$; $M_x=-75$; $(\overline{x},\overline{y}) = (4,-0.5)$
}

\exercise{$R$ is the rectangle with corners $(1,-3)$, $(1,2)$, $(7,2)$ and $(7,-3)$; $\delta(x,y) = (x+y^2)$g/cm$^2$}{$M = 190$g%; $M_y= 850$; $M_x = -315/2$; $(\overline{x},\overline{y}) = (4.47,-0.83)$
}

\exercise{$R$ is the triangle with corners $(-1,0)$, $(1,0)$, and $(0,1)$; $\delta(x,y) = 2$lb/in$^2$}{$M = 2$lb%; $M_y= 0$; $M_x = 2/3$; $(\overline{x},\overline{y}) = (0,1/3)$
}

\exercise{$R$ is the triangle with corners $(0,0)$, $(1,0)$, and $(0,1)$; $\delta(x,y) = (x^2+y^2+1)$lb/in$^2$}{$M = 2/3$lb%; $M_y= 7/30$; $M_x = 7/30$; $(\overline{x},\overline{y}) = (0.35,0.35)$
}

\exercise{$R$ is the disk centered at the origin with radius 2; $\delta(x,y) = (x+y+4)$kg/m$^2$}{$M = 16\pi\approx 50.27$kg%; $M_y= 4\pi$; $M_x = 4\pi$; $(\overline{x},\overline{y}) = (1/4,1/4)$
}

\exercise{$R$ is the circle sector bounded by $x^2+y^2=25$ in the first quadrant; $\delta(x,y) = (\sqrt{x^2+y^2}+1)$kg/m$^2$}{$M = 325\pi/12\approx 85$kg%; $M_y= 2375/12$; $M_x = 2375/12$; $(\overline{x},\overline{y}) = (2.33,2.33)$
}

\exercise{$R$ is the annulus in the first and second quadrants  bounded by $x^2+y^2=9$ and $x^2+y^2=36$; $\delta(x,y) = 4$lb/ft$^2$}{$M = 54\pi\approx 169.65$lb%; $M_y= 325/6$; $M_x = 325/6$; $(\overline{x},\overline{y}) = (1.97,1.97)$
}

\exercise{$R$ is the annulus in the first and second quadrants  bounded by $x^2+y^2=9$ and $x^2+y^2=36$; $\delta(x,y) = \sqrt{x^2+y^2}$lb/ft$^2$\label{13_04_ex_18}}{$M = 63\pi\approx 197.92$lb%; $M_y= 0$; $M_x = 504$; $(\overline{x},\overline{y}) = (0,2.97)$
