\exercise{$\ds \int_{0}^{1}\int_0^x\int_0^{x+y}\big(x+y+z\big)\ dz\ dy\ dx$}{$7/8$}

\exercise{$\ds \int_{\pi}^{\pi^2}\int_{x}^{x^3}\int_{-y^2}^{y^2}\left(z\frac{x^2y+y^2x}{e^{x^2+y^2}}\right)\ dz\ dy\ dx$}{$0$}
