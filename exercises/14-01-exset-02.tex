\begin{exerciseset}{In Exercises}{, a planar curve $C$ is given along with a surface $f$ that is defined over $C$. Set up the line integral $\ds \int_Cf(s)\dd s$, then approximate its value using technology.}

\exercise{$C$ is the portion of the parabola $y=2x^2+x+1$ on $[0,1]$; the surface is $f(x,y)=x^2+2y$.}{$\int_0^1(5t^2+_2t+2)\sqrt{(4t+1)^2+1}\dd t \approx 17.071$}

\exercise{$C$ is the portion of the curve $y=\sin x$ on $[0,\pi]$; the surface is $f(x,y)=x$.}{$\int_0^\pi t\sqrt{1+\cos^2t}\dd t \approx 6.001$}

\exercise{$C$ is the ellipse given by $\vec r(t) =\bracket{2\cos t,\sin t}$ on $[0,2\pi]$; the surface is $f(x,y)=10-x^2-y^2$.}{$\oint_0^{2\pi} \big(10-4\cos^2t-\sin^2t\big)\sqrt{\cos^2t+4\sin^2t}\dd t \approx 74.986$}

\exercise{$C$ is the portion of $y=x^3$ on $[-1,1]$; the surface is $f(x,y)=2x+3y+5$.}{$\int_{-1}^{1} \big(3t^3+2t+5\big)\sqrt{9t^4+1}\dd t \approx 15.479$}

\end{exerciseset}
