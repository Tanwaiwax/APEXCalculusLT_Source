\exercise{Let $\vec F$ be a vector field and let $C$ be a curve. \textit{Flow} is a measure of the amount of $\vec F$ going \underline{\hskip1in} $C$; \textit{flux} is a measure of the amount of $\vec F$ going \underline{\hskip1in} $C$.}{along, across}

\exercise{What is circulation?}{It is the measure of flow around the entirety of a closed curve $C$.}

\exercise{Green's Theorem states, informally, that the circulation around a closed curve that bounds a region $R$ is equal to the sum of \underline{\hskip1in} across $R$.}{the curl of $\vec F$, or $\curl \vec F$}

\exercise{The Divergence Theorem states, informally, that the outward flux across a closed curve that bounds a region $R$ is equal to the sum of \underline{\hskip1in} across $R$.}{the divergence of $\vec F$, or $\divv \vec F$}

\exercise{Let $\vec F$ be a vector field and let $C_1$ and $C_2$ be any nonintersecting paths except that each starts at point $A$ and ends at point $B$. If \underline{\hskip.5in}$=0$, then $\int_{C_1} \vec F\cdot \vec T\dd s = \int_{C_2} \vec F\cdot \vec T\dd s$.}{$\curl \vec F$}

\exercise{Let $\vec F$ be a vector field and let $C_1$ and $C_2$ be any nonintersecting paths except that each starts at point $A$ and ends at point $B$. If \underline{\hskip.5in}$=0$, then $\int_{C_1} \vec F\cdot \vec n\dd s = \int_{C_2} \vec F\cdot \vec n\dd s$.}{$\divv \vec F$}
