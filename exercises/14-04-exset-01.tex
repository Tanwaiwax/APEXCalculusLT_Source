\exercise{$\vec F =\bracket{x+y,x-y}$; $C$ is the curve with initial and terminal points $(3,-2)$ and $(3,2)$, respectively, parameterized by $\vec r(t) =\bracket{3,t}$ on $-2\leq t\leq 2$.}{$12$}

\exercise{$\vec F =\bracket{x^2,y+1}$; $C$ is line segment from $(0,0)$ to $(2,4)$.}{$-2/3$}

\exercise{$\vec F =\bracket{y,0}$; $C$ is the line segment from $(0,0)$ to $(1,1)$.}{$1/2$}
