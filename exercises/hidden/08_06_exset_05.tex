\exerciseset{In Exercises}{, write the following functions as a power series and give the radius of convergence.}{

\exercise{$\ds f(x)=\frac{x}{1-8x}$}{\mbox{}}

\exercise{$\ds f(x)=\frac{x^7}{8+x^3}$}{\mbox{}}

\exercise{$\ds f(x)=\frac{6}{1+7x^4}$}{$\ds \sum_{n=0}^\infty 6(-7)^nx^{4n}$, $1/\sqrt[4]7$}

\exercise{$\ds f(x)=\frac{x^3}{3-x^2}$}{$\ds \sum_{n=0}^\infty \frac{x^{2n+3}}{3^{n+1}}$, $\sqrt3$}

\exercise{$\ds f(x)=\frac{3x^2}{5-2\sqrt[3]x}$}{$\ds \sum_{n=0}^\infty \frac{3\cdot2^n x^{n/3+2}}{5^{n+1}}$, $125/8$}

\exercise{$\ds f(x)=\frac{1}{(1+x)^2}$}{\mbox{}}

\exercise{$\ds f(x)=\frac{1}{(1+x)^3}$}{\mbox{}}

\exercise{$\ds f(x)=\ln(3-x)$}{\mbox{}}

\exercise{$\ds f(x)=\frac {x}{(1+9x^2)^2}$}{\mbox{}}

\exercise{$\ds f(x)=\ln\biggl(\frac{1+x}{1-x}\biggr)$}{\mbox{}}
