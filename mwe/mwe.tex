\documentclass{article}

\usepackage{graphicx}
\usepackage{ragged2e}
\usepackage[justification=RaggedRight]{caption}
\usepackage{marginnote}

\newcommand*{\mnote}[1]{%
	\marginnote{%
		\begin{minipage}[c]{.8in}
			\captionsetup{type=figure}%
			\RaggedRight%
			#1%
		\end{minipage}%
	}%
}

\newcommand{\mfigure}[3]{%
	\marginnote{%
		\begin{minipage}[c]{.8in}
			\centering\includegraphics[width=.8in]{#3}%
			\captionsetup{type=figure}\caption{#1}\label{#2}%
		\end{minipage}%
	}%
}
