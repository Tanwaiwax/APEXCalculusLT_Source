\section*{11.1 \ \ More Problems}

\ 


\begin{enumerate}
\item  The points $A = (3, 2, 0)$, $B = (1, 0, 1)$, and $C = (-1, 1, 3)$ form a triangle in space.  Find the distance between each pair of points and determine if the triangle is isosceles. \\ % 7.5

\item  Explain why three points lie on a line if and only if the distance between two of the points is equal to the sum of the distances from each of these points to the third. \\ %7.6

\item  Determine whether or not the points $A = (2, 0, 3)$, $B = (0, -3, 4)$, and $C = (4, 3, 2)$ lie on a line. \\ % 7.7

\item  Determine whether or not the points $A = (4, 5, 3)$, $B = (6, 8, 7)$, and $C = (0, 0, -5)$ lie on a line. \\ % 7.8

\item  Let $P = (x_1, y_1, z_1)$ and $Q = (x_2, y_2, z_2)$ be points in space.  Show that the midpoint of $\overline{PQ}$ is 
\[
\left( \frac{x_1 + x_2}{2}, \frac{y_1 + y_2}{2}, \frac{z_1 + z_2}{2} \right).
\] \\ % 10.6

\item  Show that the point $A = (2, 1, 5)$ is inside the sphere given by 
\[
x^2 + y^2 + z^2 - 2x + 4y - 6z = 11.
\]
Is $A$ closer to the center of the sphere or to the surface of the sphere? \\ % 10.5

\item  Sketch the quadric surface 
\[
z = x^2 + y^2 - 2x - 4y + 4.
\] \\  % 32.5

\item  Sketch the quadric surface 
\[
\frac{(x - 2)^2}{4} + (y + 1)^2 + \frac{(z - 3)^2}{9} = 1.
\] \\ % 32.6
\end{enumerate}
