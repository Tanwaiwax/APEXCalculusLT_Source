\textbf{6.3 Additional Volume problems}\\
\vskip .5 truecm


\textbf{Choose the appropriate method (disk, washer, shell) to find the volume of the solid generated by revolving the bounded region about the given axes. Set-up the integral, but do NOT integrate}.\\ \\

\begin{enumerate}
\item $y =x^4, y = 0, x=1$ about 
\subitem $y$-axis  - Solutions: V = $\ds 2\pi \int_0^1 x \cdot x^4 ~dx = 2\pi \int_0^1 x^5 ~dx $ or $V = \ds \pi \int_0^1 \bigl[1^2 - (\root 4 \of y)^2 \bigr] ~dy$\\

\subitem $x$-axis - Solutions: V = $\ds 2\pi \int_0^1 y \bigl(1 - \root 4 \of y\bigr) ~dx$ or $V = \ds \pi \int_0^1 (x^4)^2 ~dy$\\

\item $y=x^3 + 1, x=0, y=2$ about\\
\subitem $y$-axis - Solutions: V = $\ds 2\pi \int_0^1 x \bigl(2 - (x^3+1)\bigr) ~dx$ or $V = \ds \pi \int_1^2 \bigl(\root 3 \of {y-1}\bigr)^2 ~dy$\\

\subitem $y=2$ - Solutions: V = $\ds 2\pi \int_1^2  (2-y)\root 3 \of {y-1} ~dy$ or $V = \ds \pi \int_0^1 \bigl[2 -(x^3 +1)\bigr]^2 ~dx$\\

\item $y=4x^2, 4x+y=8$ about
\subitem $x$-axis  - Solution: $V = \ds \pi \int_{-2}^1 \bigl[(-4x+8)^2 - (4x^2)^2\bigr] ~dx$\\
\subitem $x=1$ - Solution: V = $\ds 2\pi \int_{-2}^1  (1-x)\bigl[(-4x+8) - 4x^2\bigr]  ~dx$ \\
\subitem $y=16$ - Solution: $V = \ds \pi \int_{-2}^1 \bigl[(16 - 4x^2)^2-\bigl(16-(-4x+8)\bigr)^2  \bigr] ~dx$\\
\\
\end{enumerate}
